173     /**
174      * 見出し付きのセルの情報を取得する。
175      * @param anno マッピング情報が設定されているアノテーション
176      * @param processCase 処理ケース
177      * @return ラベルや値が設定されているセルの開始情報
178      */
179     public Optional<LabelInfo> handle(final Annotation anno, final ProcessCase processCase) {
180 
181         final AnnotationProxy annoProxy = new AnnotationProxy(anno);
182 
183         // ラベルの位置を取得する
184         final Optional<CellPosition> labelPosition = getLabelPosition(annoProxy);
185         if(!labelPosition.isPresent()) {
186             return Optional.empty();
187         }
188 
189         final int column = labelPosition.get().getColumn();
190         final int row = labelPosition.get().getRow();
191 
192         /*
193          * 見出しか結合している場合を考慮する場合
194          * ・結合サイズ分で補正する。
195          * ・考慮しない場合は、mergedXXXSizeの値は0のまま。
196          */
197         int mergedRowSize = 0;
198         int mergedColumnSize = 0;
199         if(annoProxy.labelMerged()) {
200             CellRangeAddress mergedRegion = POIUtils.getMergedRegion(sheet, row, column);
201             if(mergedRegion != null) {
202                 mergedRowSize = mergedRegion.getLastRow() - mergedRegion.getFirstRow();
203                 mergedColumnSize = mergedRegion.getLastColumn() - mergedRegion.getFirstColumn();
204             }
205         }
206 
207         int range = annoProxy.range();
208         if(range < 1){
209             range = 1;
210         }
211 
212         // 値が設定されているセルを検索する。
213         Point targetPosition = new Point();
214         Cell targetCell = null;
215         for(int i=0; i < range; i++){
216             final int index = annoProxy.skip() + i +1;
217             if(annoProxy.type() == LabelledCellType.Left) {
218                 targetPosition.x = column - index;
219                 targetPosition.y = row;
220                 targetCell = POIUtils.getCell(sheet, targetPosition);
221 
222             } else if(annoProxy.type() == LabelledCellType.Right) {
223                 targetPosition.x = column + index + mergedColumnSize;
224                 targetPosition.y = row;
225                 targetCell = POIUtils.getCell(sheet, targetPosition);
226 
227             } else if(annoProxy.type() == LabelledCellType.Bottom) {
228                 targetPosition.x = column;
229                 targetPosition.y = row + index + mergedRowSize;
230                 targetCell = POIUtils.getCell(sheet, targetPosition);
231 
232             }
233 
234             if(POIUtils.getCellContents(targetCell, config.getCellFormatter()).length() > 0){
235                 break;
236             }
237 
238             if(processCase.equals(ProcessCase.Save)) {
239                 /*
240                  * 書き込み時は、属性rangeの範囲を考慮しない。
241                  * テンプレートファイルの場合、値は空を設定しているため。
242                  */
243                 break;
244             }
245         }
246 
247         final LabelInfo info = new LabelInfo();
248         info.valueCell = targetCell;
249         info.valueAddress = CellPosition.of(targetPosition);
250         info.label = POIUtils.getCellContents(POIUtils.getCell(sheet, column, row), config.getCellFormatter());
251 
252         return Optional.of(info);
253     }

266     /**
267      * ラベルの位置を取得する。
268      * @param anno アノテーションの情報
269      * @return ラベルの位置。見つからない場合は、空を返す。
270      *         ただし、見つからない場合、設定により例外をスローする場合がある。
271      */
272     private Optional<CellPosition> getLabelPosition(final AnnotationProxy anno) {
273 
274         if(Utils.isNotEmpty(anno.label())) {
275             // 属性「label」によるラベルの指定がある場合
276             try {
277                 if(Utils.isNotEmpty(anno.headerLabel())){
278                     Cell headerCell = CellFinder.query(sheet, anno.headerLabel(), config).findWhenNotFoundException();
279                     Cell labelCell = CellFinder.query(sheet, anno.label(), config)
280                             .startPosition(headerCell.getColumnIndex(), headerCell.getRowIndex() + 1)
281                             .findWhenNotFoundException();
282                     return Optional.of(CellPosition.of(labelCell));
283 
284                 } else {
285                     Cell labelCell = CellFinder.query(sheet, anno.label(), config).findWhenNotFoundException();
286                     return Optional.of(CellPosition.of(labelCell));
287                 }
288             } catch(CellNotFoundException ex){
289                 if(anno.optional()){
290                     return Optional.empty();
291                 } else {
292                     throw ex;
293                 }
294             }
295 
296         } else {
297             /*
298              * 属性「labelRow」「labelColumn」によるアドレスを直接指定の場合
299              * これは、アノテーション @XlsIterateTables の中で指定しているときに設定される。
300              */
301             if(anno.labelRow() < 0) {
302                 throw new AnnotationInvalidException(anno.getTarget(), MessageBuilder.create("anno.attr.min")
303                         .var("property", field.getNameWithClass())
304                         .varWithAnno("anno", anno.annotationType())
305                         .var("attrName", "labelRow")
306                         .var("attrValue", anno.labelRow())
307                         .var("min", 0)
308                         .format());
309             }
310 
311             if(anno.labelColumn() < 0) {
312                 throw new AnnotationInvalidException(anno.getTarget(), MessageBuilder.create("anno.attr.min")
313                         .var("property", field.getNameWithClass())
314                         .varWithAnno("anno", anno.annotationType())
315                         .var("attrName", "labelColumn")
316                         .var("attrValue", anno.labelColumn())
317                         .var("min", 0)
318                         .format());
319 
320             }
321 
322             return Optional.of(CellPosition.of(anno.labelRow(), anno.labelColumn()));
323         }
