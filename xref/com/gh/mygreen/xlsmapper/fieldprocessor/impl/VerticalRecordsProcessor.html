85              final FieldAccessor accessor, final Configuration config, final LoadingWorkObject work) throws XlsMapperException {
86  
87          if(!accessor.isWritable()) {
88              // セルの値を書き込むメソッド／フィールドがない場合はスキップ
89              return;
90          }
91          
92          if(!Utils.isLoadCase(anno.cases())) {
93              return;
94          }
95  
96          final Class<?> clazz = accessor.getType();
97          if(Collection.class.isAssignableFrom(clazz)) {
98  
99              Class<?> recordClass = anno.recordClass();
100             if(recordClass == Object.class) {
101                 recordClass = accessor.getComponentType();
102             }
103 
104             final List<?> value = loadRecords(sheet, beansObj, anno, accessor, recordClass, config, work);
105             if(value != null) {
106                 @SuppressWarnings({"unchecked", "rawtypes"})
107                 Collection<?> collection = Utils.convertListToCollection(value, (Class<Collection>)clazz, config.getBeanFactory());
108                 accessor.setValue(beansObj, collection);
109             }
110         } else if(clazz.isArray()) {
111 
112             Class<?> recordClass = anno.recordClass();
113             if(recordClass == Object.class) {
114                 recordClass = accessor.getComponentType();
115             }
116 
117             final List<?> value = loadRecords(sheet, beansObj, anno, accessor, recordClass, config, work);
118             if(value != null) {
119                 final Object array = Array.newInstance(recordClass, value.size());
120                 for(int i=0; i < value.size(); i++) {
121                     Array.set(array, i, value.get(i));
122                 }
123 
124                 accessor.setValue(beansObj, array);
125             }
126 
127         } else {
128             throw new AnnotationInvalidException(anno, MessageBuilder.create("anno.notSupportType")
129                     .var("property", accessor.getNameWithClass())
130                     .varWithAnno("anno", XlsVerticalRecords.class)
131                     .varWithClass("actualType", clazz)
132                     .var("expectedType", "Collection(List/Set) or Array")
133                     .format());
134         }
135 
136     }

138    private List<?> loadRecords(final Sheet sheet, final Object beansObj, final XlsVerticalRecords anno, final FieldAccessor accessor,
139            final Class<?> recordClass, final Configuration config, final LoadingWorkObject work) throws XlsMapperException {
140 
141         // get table starting position
142         final Optional<CellPosition> initPosition = getHeaderPosition(sheet, anno, accessor, config);
143         if(!initPosition.isPresent()) {
144             return null;
145         }
146 
147         // ラベルの設定
148         if(Utils.isNotEmpty(anno.tableLabel())) {
149             final Optional<Cell> tableLabelCell = CellFinder.query(sheet, anno.tableLabel(), config).findOptional();
150             tableLabelCell.ifPresent(c -> {
151                 final String label = POIUtils.getCellContents(c, config.getCellFormatter());
152                 accessor.setLabel(beansObj, label);
153 
154             });
155         }
156 
157         final int initColumn = initPosition.get().getColumn();
158         final int initRow = initPosition.get().getRow();
159 
160         int hColumn = initColumn;
161         int hRow = initRow;
162 
163         // get header columns.
164         final List<RecordHeader> headers = new ArrayList<>();
165         int rangeCount = 1;
166         while(true){
167             try {
168                 Cell cell = POIUtils.getCell(sheet, hColumn, hRow);
169                 while(POIUtils.isEmptyCellContents(cell, config.getCellFormatter()) && rangeCount < anno.range()){
170                     cell = POIUtils.getCell(sheet, hColumn, hRow + rangeCount);
171                     rangeCount++;
172                 }
173 
174                 String cellValue = POIUtils.getCellContents(cell, config.getCellFormatter());
175                 if(Utils.isEmpty(cellValue)){
176                     break;
177                 } /*else {
178                     for(int j=hColumn; j > initColumn; j--){
179                         final Cell tmpCell = POIUtils.getCell(sheet, j, hRow);
180                         if(!POIUtils.isEmptyCellContents(tmpCell, config.getCellFormatter())){
181                             cell = tmpCell;
182                             break;
183                         }
184                     }
185                 }*/
186 
187                 headers.add(new RecordHeader(cellValue, cell.getRowIndex() - initRow));
188                 hRow = hRow + rangeCount;
189                 rangeCount = 1;
190 
191                 // 結合しているセルの場合は、はじめのセルだけ取得して、後は結合分スキップする。
192                 CellRangeAddress mergedRange = POIUtils.getMergedRegion(sheet, cell.getRowIndex(), cell.getColumnIndex());
193                 if(mergedRange != null) {
194                     hRow = hRow + (mergedRange.getLastRow() - mergedRange.getFirstRow());
195                 }
196 
197             } catch(ArrayIndexOutOfBoundsException ex){
198                 break;
199             }
200 
201             if(anno.headerLimit() > 0 && headers.size() >= anno.headerLimit()){
202                 break;
203             }
204         }
205 
206         // データ行の開始位置の調整
207         hColumn += anno.headerRight();
208         CellPosition startPosition = CellPosition.of(initRow, hColumn);
209 
210         // 独自の開始位置を指定する場合
211         final Optional<XlsRecordFinder> finderAnno = accessor.getAnnotation(XlsRecordFinder.class);
212         if(finderAnno.isPresent()) {
213             final RecordFinder finder = config.createBean(finderAnno.get().value());
214             startPosition = finder.find(ProcessCase.Load, finderAnno.get().args(), sheet, startPosition, beansObj, config);
215 
216         }
217 
218         return loadRecords(sheet, headers, anno, startPosition, 0, accessor, recordClass, config, work);
219    }

221    private List<?> loadRecords(final Sheet sheet, final List<RecordHeader> headers,
222            final XlsVerticalRecords anno,
223            final CellPosition initPosition, final int parentMergedSize,
224            final FieldAccessor accessor, final Class<?> recordClass,
225            final Configuration config, final LoadingWorkObject work) throws XlsMapperException {
226 
227         final List<Object> result = new ArrayList<>();
228 
229         final int initColumn = initPosition.getColumn();
230         final int initRow = initPosition.getRow();
231 
232         final int maxColumn = initColumn + parentMergedSize;
233         int hColumn = initColumn;
234 
235         // Check for columns
236         RecordsProcessorUtil.checkColumns(sheet, recordClass, headers, work.getAnnoReader(), config);
237         RecordsProcessorUtil.checkMapColumns(sheet, recordClass, headers, work.getAnnoReader(), config);
238         RecordsProcessorUtil.checkArrayColumns(sheet, recordClass, headers, work.getAnnoReader(), config);
239 
240         RecordTerminal terminal = anno.terminal();
241         if(terminal == null){
242             terminal = RecordTerminal.Empty;
243         }
244 
245         // 各種レコードのコールバック用メソッドを抽出する
246         final RecordMethodCache methodCache = new RecordMethodFacatory(work.getAnnoReader(), config)
247                 .create(recordClass, ProcessCase.Load);
248 
249         // レコードの見出しに対するカラム情報のキャッシュ
250         final Map<String, List<FieldAccessor>> propertiesCache = new HashMap<>();
251 
252         // カラムに対するConverterのキャッシュ
253         final Map<String, CellConverter<?>> converterCache = new HashMap<>();
254 
255         final int startHeaderIndex = getStartHeaderIndexForLoading(headers, recordClass, work.getAnnoReader(), config);
256 
257         // get records
258         while(hColumn < POIUtils.getColumns(sheet)){
259 
260             if(parentMergedSize > 0 && hColumn >= maxColumn) {
261                 // ネストしている処理のとき、最大の処理レコード数をチェックする。
262                 break;
263             }
264 
265             boolean emptyFlag = true;
266             // recordは、マッピング先のオブジェクトのインスタンス。
267             final Object record = config.createBean(recordClass);
268 
269             // パスの位置の変更
270             work.getErrors().pushNestedPath(accessor.getName(), result.size());
271 
272             // execute PreProcess listener
273             methodCache.getListenerClasses().forEach(listenerClass -> {
274                 listenerClass.getPreLoadMethods().forEach(method -> {
275                     Utils.invokeNeedProcessMethod(listenerClass.getObject(), method, record, sheet, config, work.getErrors(), ProcessCase.Load);
276                 });
277             });
278 
279             // execute PreProcess method
280             methodCache.getPreLoadMethods().forEach(method -> {
281                 Utils.invokeNeedProcessMethod(record, method, record, sheet, config, work.getErrors(), ProcessCase.Load);
282             });
283 
284             final List<MergedRecord> mergedRecords = new ArrayList<>();
285 
286             loadMapColumns(sheet, headers, mergedRecords, CellPosition.of(initRow, hColumn), recordClass, record, config, work);
287 
288             loadArrayColumns(sheet, headers, mergedRecords, CellPosition.of(initRow, hColumn), recordClass, record, config, work);
289 
290             for(int i=0; i < headers.size() && hColumn < POIUtils.getColumns(sheet); i++){
291                 final RecordHeader headerInfo = headers.get(i);
292                 int hRow = initRow + headerInfo.getInterval();
293                 final Cell cell = POIUtils.getCell(sheet, hColumn, hRow);
294 
295                 // find end of the table
296                 if(!POIUtils.isEmptyCellContents(cell, config.getCellFormatter())){
297                     emptyFlag = false;
298                 }
299 
300                 if(terminal==RecordTerminal.Border && i == startHeaderIndex){
301                     if(!POIUtils.getBorderTop(cell).equals(BorderStyle.NONE)){
302                         emptyFlag = false;
303                     } else {
304                         emptyFlag = true;
305                         break;
306                     }
307                 }
308 
309                 if(!anno.terminateLabel().equals("")){
310                     if(Utils.matches(POIUtils.getCellContents(cell, config.getCellFormatter()), anno.terminateLabel(), config)){
311                         emptyFlag = true;
312                         break;
313                     }
314                 }
315 
316                 // mapping from Excel columns to Object properties.
317                 final List<FieldAccessor> propeties = propertiesCache.computeIfAbsent(headerInfo.getLabel(), key -> {
318                     return FieldAccessorUtils.getColumnPropertiesByName(
319                             record.getClass(), work.getAnnoReader(), config, key)
320                             .stream()
321                             .filter(p -> p.isWritable())
322                             .collect(Collectors.toList());
323                 });
324 
325                 for(FieldAccessor property : propeties) {
326                     Cell valueCell = cell;
327                     final XlsColumn column = property.getAnnotationNullable(XlsColumn.class);
328 
329                     if(column.headerMerged() > 0){
330                         hRow = hRow + column.headerMerged();
331                         valueCell = POIUtils.getCell(sheet, hColumn, hRow);
332                     }
333 
334                     // for merged cell
335                     if(POIUtils.isEmptyCellContents(valueCell, config.getCellFormatter())){
336                         CellStyle valueCellFormat = valueCell.getCellStyle();
337                         if(column.merged() && POIUtils.getBorderRight(valueCell).equals(BorderStyle.NONE)){
338                             for(int k=hColumn; k > initColumn; k--){
339                                 final Cell tmpCell = POIUtils.getCell(sheet, k, hRow);
340                                 final CellStyle tmpCellFormat = tmpCell.getCellStyle();
341 
342                                 if(!POIUtils.getBorderLeft(tmpCell).equals(BorderStyle.NONE)){
343                                     break;
344                                 }
345 
346                                 if(!POIUtils.isEmptyCellContents(tmpCell, config.getCellFormatter())){
347                                     valueCell = tmpCell;
348                                     break;
349                                 }
350                             }
351                         }
352                     }
353 
354                     if(column.headerMerged() > 0){
355                         hRow = hRow - column.headerMerged();
356                     }
357 
358                     CellRangeAddress mergedRange = POIUtils.getMergedRegion(sheet, valueCell.getRowIndex(), valueCell.getColumnIndex());
359                     if(mergedRange != null) {
360                         int mergedSize =  mergedRange.getLastColumn() - mergedRange.getFirstColumn() + 1;
361                         mergedRecords.add(new MergedRecord(headerInfo, mergedRange, mergedSize));
362                     } else {
363                         mergedRecords.add(new MergedRecord(headerInfo, CellRangeAddress.valueOf(POIUtils.formatCellAddress(valueCell)), 1));
364                     }
365 
366                     if(!Utils.isLoadCase(column.cases())) {
367                         continue;
368                     }
369 
370                     // set for value
371                     property.setPosition(record, CellPosition.of(valueCell));
372                     property.setLabel(record, headerInfo.getLabel());
373 
374                     final Cell tempCommentCell = valueCell;
375                     property.getCommentSetter().ifPresent(setter -> 
376                             config.getCommentOperator().loadCellComment(setter, tempCommentCell, record, property, config));
377                     
378                     final CellConverter<?> converter = converterCache.computeIfAbsent(property.getName(), key -> getCellConverter(property, config));
379                     if(converter instanceof FieldFormatter) {
380                         work.getErrors().registerFieldFormatter(property.getName(), property.getType(), (FieldFormatter<?>)converter, true);
381                     }
382 
383                     try {
384                         final Object value = converter.toObject(valueCell);
385                         property.setValue(record, value);
386                     } catch(TypeBindException e) {
387                         work.addTypeBindError(e, valueCell, property.getName(), headerInfo.getLabel());
388                         if(!config.isContinueTypeBindFailure()) {
389                             throw e;
390                         }
391                     }
392                 }
393             }
394 
395             // execute nested record
396             final int skipSize = loadNestedRecords(sheet, headers, mergedRecords, anno, CellPosition.of(initRow, hColumn), record, config, work);
397             if(parentMergedSize > 0 && skipSize > 0 && (hColumn + skipSize) > maxColumn) {
398                 // check over merged cell.
399                 String message = String.format("Over merged size. In sheet '%s' with columnIndex=%d, over the columnIndex=%s.",
400                         sheet.getSheetName(), hColumn + skipSize, maxColumn);
401                 throw new NestedRecordMergedSizeException(sheet.getSheetName(), skipSize, message);
402             }
403 
404 
405             if(emptyFlag){
406                 // パスの位置の変更
407                 work.getErrors().popNestedPath();
408                 break;
409             }
410 
411             if(isAvailabledRecord(methodCache.getIgnoreableMethod(), record)) {
412                 // 有効なレコードのみ、処理を行う
413                 result.add(record);
414 
415                 // set PostProcess listener
416                 methodCache.getListenerClasses().forEach(listenerClass -> {
417                     listenerClass.getPostLoadMethods().forEach(method -> {
418                         work.addNeedPostProcess(new NeedProcess(record, listenerClass.getObject(), method));
419                     });
420                 });
421 
422                 // set PostProcess method
423                 methodCache.getPostLoadMethods().forEach(method -> {
424                     work.addNeedPostProcess(new NeedProcess(record, record, method));
425                 });
426 
427             }
428 
429             // パスの位置の変更
430             work.getErrors().popNestedPath();
431 
432             if(skipSize > 0) {
433                 hColumn += skipSize;
434             } else {
435                 hColumn++;
436             }
437 
438         }
439 
440         return result;
441     }

443     /**
444      * 表の開始位置（見出し）の位置情報を取得する。
445      *
446      * @param sheet
447      * @param anno
448      * @param accessor
449      * @param config
450      * @return 表の開始位置。指定したラベルが見つからない場合、設定によりnullを返す。
451      * @throws AnnotationInvalidException アノテーションの値が不正で、表の開始位置が位置が見つからない場合。
452      * @throws CellNotFoundException 指定したラベルが見つからない場合。
453      */
454     private Optional<CellPosition> getHeaderPosition(final Sheet sheet, final XlsVerticalRecords anno,
455             final FieldAccessor accessor, final Configuration config) throws AnnotationInvalidException, CellNotFoundException {
456 
457         if(Utils.isNotEmpty(anno.headerAddress())) {
458             try {
459                 return Optional.of(CellPosition.of(anno.headerAddress()));
460 
461             } catch(IllegalArgumentException e) {
462                 throw new AnnotationInvalidException(anno, MessageBuilder.create("anno.attr.invalidAddress")
463                         .var("property", accessor.getNameWithClass())
464                         .varWithAnno("anno", XlsVerticalRecords.class)
465                         .var("attrName", "headerAddress")
466                         .var("attrValue", anno.headerAddress())
467                         .format());
468 
469             }
470 
471         } else if(Utils.isNotEmpty(anno.tableLabel())) {
472             try {
473                 final Cell labelCell = CellFinder.query(sheet, anno.tableLabel(), config).findWhenNotFoundException();
474 
475                 if(anno.tableLabelAbove()) {
476                     // 表の見出しが上にある場合、左側に補正する
477                     int initColumn = labelCell.getColumnIndex() + anno.right()-1;
478                     int initRow = labelCell.getRowIndex() + anno.bottom();
479                     return Optional.of(CellPosition.of(initRow, initColumn));
480 
481                 } else {
482 
483                     int initColumn = labelCell.getColumnIndex() + anno.right();
484                     int initRow = labelCell.getRowIndex();
485                     return Optional.of(CellPosition.of(initRow, initColumn));
486 
487                 }
488 
489             } catch(CellNotFoundException ex) {
490                 if(anno.optional()) {
491                     return Optional.empty();
492                 } else {
493                     throw ex;
494                 }
495             }
496         } else {
497             if(anno.headerRow() < 0) {
498                 throw  new AnnotationInvalidException(anno, MessageBuilder.create("anno.attr.min")
499                         .var("property", accessor.getNameWithClass())
500                         .varWithAnno("anno", XlsVerticalRecords.class)
501                         .var("attrName", "headerRow")
502                         .var("attrValue", anno.headerRow())
503                         .var("min", 0)
504                         .format());
505             }
506 
507             if(anno.headerColumn() < 0) {
508                 throw new AnnotationInvalidException(anno, MessageBuilder.create("anno.attr.min")
509                         .var("property", accessor.getNameWithClass())
510                         .varWithAnno("anno", XlsVerticalRecords.class)
511                         .var("attrName", "column")
512                         .var("attrValue", anno.headerColumn())
513                         .var("min", 0)
514                         .format());
515 
516             }
517             return Optional.of(CellPosition.of(anno.headerRow(), anno.headerColumn()));
518         }
519     }

521     /**
522      * 表の見出しから、レコードのJavaクラスの定義にあるカラムの定義で初めて見つかるリストのインデックスを取得する。
523      * <p>カラムの定義とは、アノテーション「@XlsColumn」が付与されたもの。
524      * @param headers 表の見出し情報。
525      * @param recordClass アノテーション「@XlsColumn」が定義されたフィールドを持つレコード用のクラス。
526      * @param annoReader {@link AnnotationReader}
527      * @param config システム設定
528      * @return 引数「headers」の該当する要素のインデックス番号。不明な場合は0を返す。
529      */
530     private int getStartHeaderIndexForLoading(final List<RecordHeader> headers, Class<?> recordClass,
531             final AnnotationReader annoReader,  final Configuration config) {
532 
533         // レコードクラスが不明の場合、0を返す。
534         if((recordClass == null || recordClass.equals(Object.class))) {
535             return 0;
536         }
537 
538         for(int i=0; i < headers.size(); i++) {
539             RecordHeader headerInfo = headers.get(i);
540             final List<FieldAccessor> propeties = FieldAccessorUtils.getColumnPropertiesByName(
541                     recordClass, annoReader, config, headerInfo.getLabel())
542                     .stream()
543                     .filter(p -> p.isWritable())
544                     .collect(Collectors.toList());
545             if(!propeties.isEmpty()) {
546                 return i;
547             }
548         }
549 
550         return 0;
551 
552     }

554     private void loadMapColumns(final Sheet sheet, final List<RecordHeader> headers, final List<MergedRecord> mergedRecords,
555             final CellPosition beginPosition, final Class<?> recordClass, final Object record, final Configuration config, final LoadingWorkObject work) throws XlsMapperException {
556 
557         final List<FieldAccessor> properties = FieldAccessorUtils.getPropertiesWithAnnotation(
558                 recordClass, work.getAnnoReader(), XlsMapColumns.class)
559                 .stream()
560                 .filter(p -> p.isWritable())
561                 .collect(Collectors.toList());
562 
563         for(FieldAccessor property : properties) {
564             final XlsMapColumns mapAnno = property.getAnnotationNullable(XlsMapColumns.class);
565 
566             if(!Utils.isLoadCase(mapAnno.cases())) {
567                 continue;
568             }
569 
570             Class<?> valueClass = mapAnno.valueClass();
571             if(valueClass == Object.class) {
572                 valueClass = property.getComponentType();
573             }
574 
575             // get converter (map key class)
576             final CellConverter<?> converter = getCellConverter(valueClass, property, config);
577             if(converter instanceof FieldFormatter) {
578                 work.getErrors().registerFieldFormatter(property.getName(), valueClass, (FieldFormatter<?>)converter, true);
579             }
580 
581             boolean foundPreviousColumn = false;
582             final Map<String, Object> map = new LinkedHashMap<>();
583             for(RecordHeader headerInfo : headers){
584                 int hRow = beginPosition.getRow() + headerInfo.getInterval();
585                 if(Utils.matches(headerInfo.getLabel(), mapAnno.previousColumnName(), config)){
586                     foundPreviousColumn = true;
587                     hRow++;
588                     continue;
589                 }

1785      * @param valueCellPositions 結合する開始位置のセルのアドレス
1786      */
1787     private void processSavingNestedMergedRecord(final Sheet sheet, final int mergedSize,
1788             final List<CellPosition> valueCellPositions) {
1789 
1790         if(mergedSize <= 1) {
1791             return;
1792         }
1793 
1794         // ネストした場合、上のセルのスタイルをコピーして、結合する
1795         for(CellPosition position : valueCellPositions) {
1796             Cell valueCell = POIUtils.getCell(sheet, position);
1797             if(valueCell == null) {
1798                 continue;
1799             }
1800 
1801             final CellStyle style = valueCell.getCellStyle();
1802 
1803             // 結合するセルに対して、上のセルのスタイルをコピーする。
1804             // 列を挿入するときなどに必要になるため、スタイルを設定する。
1805             for(int i=1; i < mergedSize; i++) {
1806                 Cell mergedCell = POIUtils.getCell(sheet, position.getColumn() + i, position.getRow());
1807                 mergedCell.setCellStyle(style);
1808                 mergedCell.setCellType(CellType.BLANK);
1809             }
1810 
1811             final CellRangeAddress range = new CellRangeAddress(position.getRow(), position.getRow(),
1812                     position.getColumn(), position.getColumn() + mergedSize -1);
1813 
1814             // 既に結合済みのセルがある場合、外す。
1815             for(int colIdx=range.getFirstColumn(); colIdx <= range.getLastColumn(); colIdx++) {
1816                 CellRangeAddress r = POIUtils.getMergedRegion(sheet, position.getRow(), colIdx);
1817                 if(r != null) {
1818                     POIUtils.removeMergedRange(sheet, r);
1819                 }
1820             }
1821 
1822             sheet.addMergedRegion(range);
1823         }
1824 
1825     }

1853             for(CellRangeAddress range : region.getCellRangeAddresses()) {

1864                 }
