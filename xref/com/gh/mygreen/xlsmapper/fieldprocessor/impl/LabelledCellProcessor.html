33      @Override
34      public void loadProcess(final Sheet sheet, final Object beansObj, final XlsLabelledCell anno,
35              final FieldAccessor accessor, final Configuration config, final LoadingWorkObject work) throws XlsMapperException {
36  
37          if(!accessor.isWritable()) {
38              // セルの値を書き込むメソッド／フィールドがない場合はスキップ
39              return;
40          }
41          
42          if(!Utils.isLoadCase(anno.cases())) {
43              return;
44          }
45  
46          // マッピング対象のセル情報の取得
47          LabelledCellHandlermpl/LabelledCellHandler.html#LabelledCellHandler">LabelledCellHandler labelHandler = new LabelledCellHandler(accessor, sheet, config);
48          Optional<LabelInfo> labelInfo = labelHandler.handle(anno, ProcessCase.Load);
49  
50          if(!labelInfo.isPresent()) {
51              /*
52               * ラベル用のセルが見つからない場合
53               * optional=falseの場合は、例外がスローされここには到達しない。
54               */
55              return;
56          }
57  
58          accessor.setPosition(beansObj, labelInfo.get().valueAddress);
59          accessor.setLabel(beansObj, labelInfo.get().label);
60          
61          accessor.getCommentSetter().ifPresent(setter -> 
62                  config.getCommentOperator().loadCellComment(setter, labelInfo.get().valueCell, beansObj, accessor, config));
63  
64          final CellConverter<?> converter = getCellConverter(accessor, config);
65          if(converter instanceof FieldFormatter) {
66              work.getErrors().registerFieldFormatter(accessor.getName(), accessor.getType(), (FieldFormatter<?>)converter, true);
67          }
68  
69          try {
70              final Object value = converter.toObject(labelInfo.get().valueCell);
71              accessor.setValue(beansObj, value);
72          } catch(TypeBindException e) {
73              work.addTypeBindError(e, labelInfo.get().valueAddress, accessor.getName(), labelInfo.get().label);
74              if(!config.isContinueTypeBindFailure()) {
75                  throw e;
76              }
77          }
78      }

84  
85          if(!accessor.isReadable()) {
86              // セルの値を参照するメソッド／フィールドがない場合はスキップ
87              return;
88          }
89          
90          if(!Utils.isSaveCase(anno.cases())) {
91              return;
92          }
93  
94          // マッピング対象のセル情報の取得
95          LabelledCellHandlermpl/LabelledCellHandler.html#LabelledCellHandler">LabelledCellHandler labelHandler = new LabelledCellHandler(accessor, sheet, config);
96          Optional<LabelInfo> labelInfo = labelHandler.handle(anno, ProcessCase.Save);
97  
98          if(!labelInfo.isPresent()) {
99              /*
100              * ラベル用のセルが見つからない場合
101              * optional=falseの場合は、例外がスローされここには到達しない。
102              */
103             return;
104         }
105 
106         accessor.setPosition(targetObj, labelInfo.get().valueAddress);
107         accessor.setLabel(targetObj, labelInfo.get().label);
108         
109         accessor.getCommentGetter().ifPresent(getter -> 
110                     config.getCommentOperator().saveCellComment(getter, POIUtils.getCell(sheet, labelInfo.get().valueAddress), 
111                         targetObj, accessor, config));
112 
113         final CellConverter converter = getCellConverter(accessor, config);
114         if(converter instanceof FieldFormatter) {
115             work.getErrors().registerFieldFormatter(accessor.getName(), accessor.getType(), (FieldFormatter<?>)converter, true);
116         }
117 
118         try {
119             converter.toCell(accessor.getValue(targetObj), targetObj, sheet, labelInfo.get().valueAddress);
120 
121         } catch(TypeBindException e) {
122             work.addTypeBindError(e, labelInfo.get().valueAddress, accessor.getName(), labelInfo.get().label);
123             if(!config.isContinueTypeBindFailure()) {
124                 throw e;
125             }
126         }
