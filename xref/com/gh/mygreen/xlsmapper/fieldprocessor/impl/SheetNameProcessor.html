13  
14  
15  /**
16   * アノテーション {@link XlsSheetName} を処理する。
17   *
18   * @version 2.0
19   * @author Naoki Takezoe
20   * @author T.TSUCHIE
21   */
22  public class SheetNameProcessor extends AbstractFieldProcessor<XlsSheetName> {
23  
24      @Override
25      public void loadProcess(final Sheet sheet, final Object beansObj, final XlsSheetName anno, final FieldAccessor accessor,
26              final Configuration config, final LoadingWorkObject work) {
27  
28          if(!Utils.isLoadCase(anno.cases())) {
29              return;
30          }
31  
32          final String sheetName = sheet.getSheetName();
33          accessor.setValue(beansObj, sheetName);
34  
35      }
36  
37      @Override
38      public void saveProcess(final Sheet sheet, final Object targetObj, final XlsSheetName anno, final FieldAccessor accessor,
39              final Configuration config, final SavingWorkObject work) throws XlsMapperException {
40  
41          if(!Utils.isSaveCase(anno.cases())) {
42              return;
43          }
44  
45          final String sheetName = sheet.getSheetName();
46          accessor.setValue(targetObj, sheetName);
47  
48      }
49  }
