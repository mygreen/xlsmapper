30  /**
31   * {@link FieldProcessor}を管理するクラス。
32   * 
33   * @version 2.1
34   * @author Naoki Takezoe
35   * @author T.TSUCHIE
36   *
37   */
38  public class FieldProcessorRegistry {
39      
40      private Map<Class<? extends Annotation>, FieldProcessor<?>> pocessorMap = new ConcurrentHashMap<>();
41      
42      public FieldProcessorRegistry() {
43          this.pocessorMap = new ConcurrentHashMap<>();
44          
45          init();
46      }
47      
48      /**
49       * {@link FieldProcessor}の登録状態を初期値に戻します。
50       */
51      public void init() {
52          
53          pocessorMap.clear();
54          
55          //標準のフィールドプロセッサを登録する。
56          registerProcessor(XlsSheetName.class, new SheetNameProcessor());
57          registerProcessor(XlsCell.class, new CellProcessor());
58          registerProcessor(XlsLabelledCell.class, new LabelledCellProcessor());
59          registerProcessor(XlsHorizontalRecords.class, new HorizontalRecordsProcessor());
60          registerProcessor(XlsVerticalRecords.class, new VerticalRecordsProcessor());
61          registerProcessor(XlsIterateTables.class, new IterateTablesProcessor());
62          registerProcessor(XlsArrayCells.class, new ArrayCellsProcessor());
63          registerProcessor(XlsLabelledArrayCells.class, new LabelledArrayCellsProcessor());
64          registerProcessor(XlsComment.class, new CommentProcessor());
65          registerProcessor(XlsLabelledComment.class, new LabelledCommentProcessor());
66          
67      }
68      
69      /**
70       * アノテーションに対する{@link FieldProcessor}を取得する。
71       * @param annoClass  取得対象のアノテーションのクラスタイプ。
72       * @return 見つからない場合はnullを返す。
73       * @throws NullPointerException {@literal annoClass}
74       */
75      @SuppressWarnings("unchecked")
76      public <A extends Annotation> FieldProcessor<A> getProcessor(final Class<A> annoClass) {
77          ArgUtils.notNull(annoClass, "annoClass");
78          
79          return (FieldProcessor<A>) pocessorMap.get(annoClass);
80      }
81      
82      /**
83       * アノテーションに対する{@link FieldProcessor}を登録する。
84       * @param annoClass 登録対象のアノテーションのクラスタイプ。
85       * @param processor フィールドプロセッサーのインスタンス。{@link FieldProcessor}を実装している必要がある。
86       * @throws NullPointerException {@literal annoClass == null or processor == null.}
87       */
88      public <A extends Annotation> void registerProcessor(final Class<A> annoClass, final FieldProcessor<A> processor) {
89          ArgUtils.notNull(annoClass, "annoClass");
90          ArgUtils.notNull(processor, "processor");
91          
92          pocessorMap.put(annoClass, processor);
93          
94      }
95      
96  }
