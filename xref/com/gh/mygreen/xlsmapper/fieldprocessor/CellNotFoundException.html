3   import com.gh.mygreen.xlsmapper.XlsMapperException;
4   import com.gh.mygreen.xlsmapper.localization.MessageBuilder;
5   
6   
7   /**
8    * マッピング対象のCellが見つからない場合にスローする例外。
9    * 
10   * @version 2.0
11   * @author T.TSUCHIE
12   *
13   */
14  public class CellNotFoundException extends XlsMapperException {
15      
16      /** serialVersionUID */
17      private static final long serialVersionUID = -3913407241079675756L;
18      
19      /** 検索元のシート */
20      private final String sheetName;
21      
22      /** 検索対象のセルのラベル */
23      private final String label;
24      
25      /**
26       * 指定したラベルを持つセルが見つからない場合
27       * @param sheetName シート名
28       * @param label ラベル名
29       */
30      public CellNotFoundException(final String sheetName, final String label) {
31          super(MessageBuilder.create("cell.notNotFound.label")
32                  .var("sheetName", sheetName)
33                  .var("label", label)
34                  .format());
35          this.sheetName = sheetName;
36          this.label = label;
37          
38      }
39      
40      /**
41       * シート名を取得する
42       * @return シート名
43       */
44      public String getSheetName() {
45          return sheetName;
46      }
47      
48      /**
49       * ラベル名を取得する
50       * @return ラベル名
51       */
52      public String getLabel() {
53          return label;
54      }
55      
56  }
