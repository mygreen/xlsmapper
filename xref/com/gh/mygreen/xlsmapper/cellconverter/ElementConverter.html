5   /**
6    * {@link XlsArrayConverter}で指定する、配列やリストの要素をオブジェクトに変換するためのクラス。
7    * 
8    * @version 2.0
9    * @since 1.1
10   * @author T.TSUCHIE
11   * @param <T> 変換対象のクラスタイプ
12   *
13   */
14  public interface ElementConverter<T> {
15      
16      /**
17       * 文字列を指定したクラス型に変換する。
18       * 
19       * <p>文字列の値がnullまたは空文字の場合、プリミティブ型の場合ゼロなどのデフォルト値を返す。
20       *  <br>オブジェクト型の場合はnullを返す。
21       * </p>
22       * 
23       * @since 1.1
24       * @param text 変換対象の文字列
25       * @param targetClass 変換後のクラスタイプ
26       * @return 変換した値。
27       * @throws IllegalArgumentException targetClass is null.
28       * @throws ConversionException 変換に失敗した場合にスローされます。
29       *                             また、引数targetClassが対応していないクラス型の場合にすろーされます。
30       */
31      T convertToObject(String text, Class<T> targetClass) throws ConversionException;
32      
33      /**
34       * 任意のオブジェクトを文字列に変換する。
35       * 
36       * @since 1.1
37       * @param value 変換対象のオブジェクト。
38       * @return 変換した文字列
39       */
40      String convertToString(T value);
41  }
