29      /**
30       * 引数で指定したCellConverterに対して、トリムなどの共通の設定を行う。
31       * @param cellConverter 設定を行うCellConverter
32       * @param field フィールド情報
33       * @param config システム設定情報
34       */
35      protected void setupCellConverter(final BaseCellConverter<T> cellConverter, final FieldAccessor field, final Configuration config) {
36  
37          final TextFormatter<T> textFormatter = createTextFormatter(field, config);
38          cellConverter.setTextFormatter(textFormatter);
39  
40          // トリムの設定
41          final Optional<XlsTrim> trimAnno = field.getAnnotation(XlsTrim.class);
42          final boolean trimmed = trimAnno.map(anno -> true).orElse(false);
43  
44          cellConverter.setTrimmed(trimmed);
45  
46          // 初期値の設定
47          final Optional<XlsDefaultValue> defaultValueAnno = field.getAnnotation(XlsDefaultValue.class);
48          defaultValueAnno.ifPresent(anno -> {
49              String text = Utils.trim(anno.value(), trimmed);
50  
51              try {
52                  T defaultValue = textFormatter.parse(text);
53                  cellConverter.setDefaultValue(defaultValue, anno.cases());
54  
55              } catch(TextParseException e) {
56                  throw new AnnotationInvalidException(anno, MessageBuilder.create("anno.XlsDefaultValue.failParse")
57                          .var("property", field.getNameWithClass())
58                          .var("defaultValue", text)
59                          .varWithClass("type", field.getType())
60                          .format(), e);
61              }
62          });
63  
64          // セルの書式の設定
65          final Optional<XlsCellOption> cellOptionAnno = field.getAnnotation(XlsCellOption.class);
66          cellOptionAnno.ifPresent(anno -> {
67              cellConverter.setShrinktToFit(anno.shrinkToFit());
68              cellConverter.setWrapText(anno.wrapText());
69              cellConverter.setIndent(anno.indent());
70              cellConverter.setHorizontalAlignment(anno.horizontalAlign().poiAlignType());
71              cellConverter.setVerticalAlignment(anno.verticalAlign().poiAlignType());
72          });
73  
74          // 数式の設定
75          final Optional<XlsFormula> formulaAnno = field.getAnnotation(XlsFormula.class);
76          formulaAnno.ifPresent(anno -> {
77              CellFormulaHandler formulaHandler = createCellFormulaHandler(anno, field, config);
78              cellConverter.setFormulaHandler(formulaHandler);
79  
80          });
81  
82          // 各個別の設定
83          setupCustom(cellConverter, field, config);
84  
85      }
