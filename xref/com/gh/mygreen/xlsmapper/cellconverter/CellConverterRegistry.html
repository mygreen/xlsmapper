46  /**
47   * ExcelのCell {@literal <=>} Javaオブジェクト の相互変換をする{@link CellConverter}を管理するクラス。
48   * 独自の{@link CellConverter}を登録したりする場合は、このクラスを経由する。
49   * 
50   * @version 1.0
51   * @author T.TSUCHIE
52   *
53   */
54  public class CellConverterRegistry {
55      
56      /**
57       * {@link CellConverterFactory}のクラスのキャッシュ情報
58       * ・key = 変換対象のJavaのクラスタイプ
59       * ・value = Converterクラスのインスタンス。
60       */
61      private Map<Class<?>, CellConverterFactory<?>> converterFactoryMap;
62      
63      public CellConverterRegistry() {
64          init();
65      }
66      
67      /**
68       * 初期化を行います。
69       * <p>システム標準の{@link CellConverter}を登録などを行います。
70       */
71      protected void init() {
72          
73          if(converterFactoryMap == null) {
74              this.converterFactoryMap = new ConcurrentHashMap<>();
75          } else {
76              converterFactoryMap.clear();
77          }
78          
79          registerConverter(Boolean.class, new BooleanCellConverterFactory());
80          registerConverter(boolean.class, new BooleanCellConverterFactory());
81          
82          registerConverter(Character.class, new CharacterCellConverterFactory());
83          registerConverter(char.class, new CharacterCellConverterFactory());
84          
85          registerConverter(String.class, new StringCellConverterFactory());
86          
87          registerConverter(Short.class, new ShortCellConverterFactory());
88          registerConverter(short.class, new ShortCellConverterFactory());
89          
90          registerConverter(Byte.class, new ByteCellConverterFactory());
91          registerConverter(byte.class, new ByteCellConverterFactory());
92          
93          registerConverter(Integer.class, new IntegerCellConverterFactory());
94          registerConverter(int.class, new IntegerCellConverterFactory());
95          
96          registerConverter(Long.class, new LongCellConverterFactory());
97          registerConverter(long.class, new LongCellConverterFactory());
98          
99          registerConverter(Float.class, new FloatCellConverterFactory());
100         registerConverter(float.class, new FloatCellConverterFactory());
101         
102         registerConverter(Double.class, new DoubleCellConverterFactory());
103         registerConverter(double.class, new DoubleCellConverterFactory());
104         
105         registerConverter(BigDecimal.class, new BigDecimalCellConverterFactory());
106         registerConverter(BigInteger.class, new BigIntegerCellConverterFactory());
107         
108         registerConverter(Date.class, new DateCellConverterFactory());
109         registerConverter(java.sql.Date.class, new SqlDateCellConverterFactory());
110         registerConverter(Timestamp.class, new SqlTimestampCellConverterFactory());
111         registerConverter(Time.class, new SqlTimeCellConverterFactory());
112         
113         registerConverter(Calendar.class, new CalendarCellConverterFactory());
114         
115         registerConverter(Enum.class, new EnumCellConverterFactory());
116         
117         registerConverter(List.class, new ListCellConverterFactory());
118         registerConverter(Set.class, new SetCellConverterFactory());
119         registerConverter(Object[].class, new ArrayCellConverterFactory());
120         
121         registerConverter(URI.class, new URICellConverterFactory());
122         registerConverter(CellLink.class, new CellLinkCellConverterFactory());
123         
124         registerConverter(LocalDateTime.class, new LocalDateTimeCellConverterFactory());
125         registerConverter(LocalDate.class, new LocalDateCellConverterFactory());
126         registerConverter(LocalTime.class, new LocalTimeCellConverterFactory());
127         
128     }
129     
130     /**
131      * タイプに対する{@link CellConverterFactory}を取得する。
132      * @param clazz 取得対象の{@link CellConverterFactory}のクラス。
133      * @return 見つからない場合はnullを返す。
134      */
135     @SuppressWarnings("unchecked")
136     public <T> CellConverterFactory<T> getConverterFactory(final Class<T> clazz) {
137         ArgUtils.notNull(clazz, "clazz");
138         
139         CellConverterFactory<T> converterFactory = (CellConverterFactory<T>) converterFactoryMap.get(clazz);
140         if(converterFactory == null) {
141             // 特別に判定が必要なクラス
142             if(Enum.class.isAssignableFrom(clazz)) {
143                 converterFactory = (CellConverterFactory<T>) converterFactoryMap.get(Enum.class);
144                 
145             } else if(List.class.isAssignableFrom(clazz)) {
146                 converterFactory = (CellConverterFactory<T>) converterFactoryMap.get(List.class);
147                 
148             } else if(clazz.isArray()) {
149                 converterFactory = (CellConverterFactory<T>) converterFactoryMap.get(Object[].class);
150                 
151             }
152         }
153         
154         return converterFactory;
155     }
156     
157     /**
158      * タイプに対する{@link CellConverter}を登録する。
159      * @param clazz 変換対象のJavaのクラスタイプ。
160      * @param converterFactory 変換する{@link CellConverterFactory}のインスタンス。
161      */
162     public <T> void registerConverter(final Class<T> clazz, final CellConverterFactory<T> converterFactory) {
163         ArgUtils.notNull(clazz, "clazz");
164         ArgUtils.notNull(converterFactory, "converterFactory");
165         
166         converterFactoryMap.put(clazz, converterFactory);
167     }
