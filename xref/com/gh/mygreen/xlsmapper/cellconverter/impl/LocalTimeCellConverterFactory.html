3   import java.time.LocalDate;
4   import java.time.LocalDateTime;
5   import java.time.LocalTime;
6   import java.time.ZoneId;
7   import java.time.ZonedDateTime;
8   import java.time.format.DateTimeFormatter;
9   import java.time.format.DateTimeParseException;
10  import java.util.Date;
11  
12  import com.gh.mygreen.xlsmapper.Configuration;
13  import com.gh.mygreen.xlsmapper.cellconverter.CellConverter;
14  import com.gh.mygreen.xlsmapper.fieldaccessor.FieldAccessor;
15  
16  /**
17   * {@link LocalTime}を処理する{@link CellConverter}を作成するためのファクトリクラス。
18   *
19   * @since 2.0
20   * @author T.TSUCHIE
21   *
22   */
23  public class LocalTimeCellConverterFactory extends AbstractTemporalCellConverterFactory<LocalTime>{
24  
25      @Override
26      public LocalTimeCellConverter create(final FieldAccessor field, final Configuration config) {
27  
28          final LocalTimeCellConverter cellConverter = new LocalTimeCellConverter(field, config);
29          setupCellConverter(cellConverter, field, config);
30  
31          return cellConverter;
32  
33      }
34  
35      @Override
36      protected LocalTime parseTemporal(final String str, final DateTimeFormatter formatter) throws DateTimeParseException {
37          return LocalTime.parse(str, formatter);
38      }
39  
40      @Override
41      protected String getDefaultJavaPattern() {
42          return "HH:mm:ss";
43      }
44  
45      @Override
46      protected String getDefaultExcelPattern() {
47          return "hh:mm:ss";
48      }
49  
50      public class LocalTimeCellConverter extends AbstractTemporalCellConverter<LocalTime> {
51  
52          private LocalTimeCellConverter(final FieldAccessor field, final Configuration config) {
53              super(field, config);
54          }
55  
56          @Override
57          protected LocalTime convertFromDate(final Date date) {
58              LocalDateTime dateTime = LocalDateTime.ofInstant(date.toInstant(), ZoneId.systemDefault());
59              return dateTime.toLocalTime();
60          }
61  
62          @Override
63          protected Date convertToDate(final LocalTime value, final boolean dateStart1904) {
64  
65              final LocalDateTime dateTime;
66              if(dateStart1904) {
67                  dateTime = value.atDate(LocalDate.of(1904, 1, 1));
68              } else {
69                  dateTime = value.atDate(LocalDate.of(1899, 12, 31));
70              }
71  
72              ZonedDateTime zoneDateTime = ZonedDateTime.of(dateTime, ZoneId.systemDefault());
73  
74              return Date.from(zoneDateTime.toInstant());
75          }
76  
77      }
78  
79  }
