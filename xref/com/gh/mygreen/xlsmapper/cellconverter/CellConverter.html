1   package com.gh.mygreen.xlsmapper.cellconverter;
2   
3   import org.apache.poi.ss.usermodel.Cell;
4   import org.apache.poi.ss.usermodel.Sheet;
5   
6   import com.gh.mygreen.xlsmapper.XlsMapperException;
7   import com.gh.mygreen.xlsmapper.util.CellPosition;
8   
9   /**
10   * ExcelのCellの値とJavaオブジェクトを相互に変換するインタフェース。
11   *
12   * @version 2.0
13   * @author T.TSUCHIE
14   * @param <T> 変換対象のJavaのオブジェクトのタイプ
15   */
16  public interface CellConverter<T> {
17      
18      /**
19       * シート読み込み時のExcel Cell {@literal =>} Javaオブジェクトに変換する。
20       * @param cell 読み込み対象のセル
21       * @return 変換したJavaオブジェクト
22       * @throws XlsMapperException 変換に失敗した場合
23       */
24      T toObject(Cell cell) throws XlsMapperException;
25      
26      /**
27       * シート書き込み時のJavaオブジェクト {@literal =>} Excel Cellに変換する。
28       * @param targetValue 書き込み対象のオブジェクト。
29       * @param targetBean 書き込み対象のフィールドが設定されているJavaBeanオブジェクト。
30       * @param sheet 書き込み先のシート
31       * @param address 書き込み先のセルのアドレス
32       * @return 書き込んだセル
33       * @throws XlsMapperException 変換に失敗した場合
34       */
35      Cell toCell(T targetValue, Object targetBean, Sheet sheet, CellPosition address)
36              throws XlsMapperException;
37      
38  }
