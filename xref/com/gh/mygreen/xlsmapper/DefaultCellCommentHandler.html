60      @Override
61      public Optional<String> handleLoad(final Cell cell, Optional<XlsCommentOption> commentOption) {
62          
63          Comment comment = getMergedCellComment(cell);
64          if(comment == null) {
65              return Optional.empty();
66          }
67          
68          String commentText = comment.getString().getString();
69          return Optional.of(commentText);
70      }
71      
72      /**
73       * 結合を考慮したセルのコメントを取得する。
74       * @param cell 元となるセル。
75       * @return コメント。コメントが設定されていなければ、nullを返す。
76       */
77      private Comment getMergedCellComment(final Cell cell) {
78          Comment comment = cell.getCellComment();
79          if(comment != null) {
80              return comment;
81          }
82          
83          final Sheet sheet = cell.getSheet();
84          final int size = sheet.getNumMergedRegions();
85          
86          for(int i=0; i < size; i++) {
87              final CellRangeAddress range = sheet.getMergedRegion(i);
88              if(!range.isInRange(cell)) {
89                  continue;
90              }
91              
92              // nullでないセルを取得する。
93              for(int rowIdx=range.getFirstRow(); rowIdx <= range.getLastRow(); rowIdx++) {
94                  final Row row = sheet.getRow(rowIdx);
95                  if(row == null) {
96                      continue;
97                  }
98  
99                  for(int colIdx=range.getFirstColumn(); colIdx <= range.getLastColumn(); colIdx++) {
100                     final Cell valueCell = row.getCell(colIdx);
101                     if(valueCell == null) {
102                         continue;
103                     }
104 
105                     comment = valueCell.getCellComment();
106                     if(comment != null) {
107                         return comment;
108                     }
109                 }
110             }
111         }
112         
113         return null;
114         
115     }

117     @Override
118     public void handleSave(final Cell cell, final Optional<String> text, final Optional<XlsCommentOption> commentOption) {
119         
120         if(!text.isPresent()) {
121             // コメントが空のとき
122             commentOption.ifPresent(option -> {
123                 if(option.removeIfEmpty()) {
124                     // コメントが空のとき既存のコメントを削除する
125                     cell.removeCellComment();
126                 }
127             });
128             return;
129         }
130         
131         final Sheet sheet = cell.getSheet();
132         final CreationHelper helper = sheet.getWorkbook().getCreationHelper();
133         final Drawing<?> drawing = sheet.createDrawingPatriarch();
134         
135         final Comment comment;
136         RichTextString richText = helper.createRichTextString(text.get());
137         if(cell.getCellComment() == null) {
138             ClientAnchor anchor = createAnchor(drawing, text.get(), cell, commentOption);
139             comment = drawing.createCellComment(anchor);
140             applyCommentFormat(richText, cell);
141         } else {
142             // 既存のコメントが存在する場合は、書式やサイズをコピーして使用する。
143             comment = cell.getCellComment();
144             RichTextString orgText = comment.getString();
145             if(orgText.numFormattingRuns() > 0) {
146                 copyCommentFormat(richText, orgText);
147             } else {
148                 applyCommentFormat(richText, cell);
149             }
150         }
151         
152         comment.setString(richText);
153         
154         // コメントの表示状態の更新
155         commentOption.ifPresent(option -> comment.setVisible(option.visible()));
156         
157         cell.setCellComment(comment);
158         
159     }

161     /**
162      * コメントの位置、サイズを作成する。
163      * @param drawing
164      * @param text 書き込むコメント
165      * @param cell 書込み対象のセル
166      * @param commentOption コメントのオプション
167      * @return コメントの表示位置
168      */
169     protected ClientAnchor createAnchor(final Drawing<?> drawing, final String text, final Cell cell,
170             final Optional<XlsCommentOption> commentOption) {
171         final CellPosition address = CellPosition.of(cell);
172         
173         // コメントを開業で分割し、最長の行を取得する。
174         String[] split = text.split("\r\n|\r|\n");
175         int maxLength = Arrays.stream(split)
176                 .mapToInt(str -> str.getBytes(Charset.forName("Windows-31j")).length)
177                 .max().orElse(0);
178         
179         /*
180          * コメントの横サイズ。文字数（バイト数）をもとに決定。
181          * ・1セルの文字数を元に出す。
182          * ・columnWidthは、1文字の幅を1/256にしたものが単位となる。
183          * ・最大3列分とする。
184          */
185         int charPerColumn = cell.getSheet().getColumnWidth(cell.getColumnIndex())/256;
186         int commentColumnSize = (int)Math.ceil(maxLength*1.0 / charPerColumn);
187         
188         int columnSize = commentColumnSize;
189         int lineWrappingCount = 0;
190         if(commentColumnSize > maxHorizontalSize) {
191             columnSize = maxHorizontalSize;
192             // 行の折り返し回数を計算する
193             lineWrappingCount = commentColumnSize / maxHorizontalSize;
194         }
195         
196         if(commentOption.isPresent() && commentOption.get().horizontalSize() > 0) {
197             // 直接指定されている場合
198             columnSize = commentOption.get().horizontalSize();
199             // 行の折り返し回数を計算する
200             lineWrappingCount = columnSize / maxHorizontalSize;
201         }
202         
203         // コメントの縦サイズ。行数をもとに決定。
204         int rowSize = split.length + lineWrappingCount > maxVerticalSize ? maxVerticalSize : split.length + lineWrappingCount;
205         if(commentOption.isPresent() && commentOption.get().verticalSize() > 0) {
206             // 直接指定されている場合
207             rowSize = commentOption.get().verticalSize();
208         }
209         
210         return drawing.createAnchor(
211                 0, 0, 0, 0,
212                 address.getColumn() + horizontalPrefix, address.getRow() + vertialPrefix,
213                 address.getColumn() + horizontalPrefix + columnSize, address.getRow() + vertialPrefix + rowSize);
214     }

216     /**
217      * 新規にコメントの装飾を設定する。
218      * セルの装飾に合わせる。
219      * 
220      * @param toRichText 設定先のコメント
221      * @param cell コメントを設定する先のセル
222      */
223     protected void applyCommentFormat(final RichTextString toRichText, final Cell cell) {
224         
225         toRichText.applyFont(cell.getCellStyle().getFontIndex());
226     }
227     
228     /**
229      * 既にコメントが設定されているときのコメントの装飾を設定する。
230      * 既存のコメントの装飾をコピーするが、そのとき、１つ目のフォント設定のみとする。
231      * 
232      * @param toRichText コピー先
233      * @param fromrichText コピー元
234      */
235     protected void copyCommentFormat(final RichTextString toRichText, final RichTextString fromrichText) {
236         
237         if(toRichText instanceof XSSFRichTextString) {
238             toRichText.applyFont(((XSSFRichTextString)fromrichText).getFontOfFormattingRun(0));
239             
240         } else if(toRichText instanceof HSSFRichTextString) {
241             toRichText.applyFont(((HSSFRichTextString)fromrichText).getFontOfFormattingRun(0));
242             
243         } else {
244             logger.warn("not suuported exdcel format comment : {}", toRichText.getClass().getName());
245         }
246         
247     }
