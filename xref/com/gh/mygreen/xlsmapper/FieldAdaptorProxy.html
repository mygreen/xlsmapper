7   import com.gh.mygreen.xlsmapper.fieldprocessor.FieldAdaptor;

12  
13  /**
14   * フィールドに対するアノテーション、プロセッサーなどを保持するクラス。
15   * 
16   * @author T.TSUCHIE
17   *
18   */
19  public class FieldAdaptorProxy {
20      
21      private final Annotation annotation;
22      
23      private final FieldProcessor processor;
24      
25      private final FieldAdaptor adaptor;
26      
27      public FieldAdaptorProxy(final Annotation annotation, FieldProcessor processor, FieldAdaptor adaptor) {
28          this.annotation = annotation;
29          this.processor = processor;
30          this.adaptor = adaptor;
31      }
32      
33      public boolean equals(final Object obj) {
34          if(obj instanceof FieldAdaptor) {
35              return ((FieldAdaptor)obj).equals(obj);
36          }
37          return super.equals(obj);
38      }
39      
40      public Annotation getAnnotation() {
41          return annotation;
42      }
43      
44      public FieldProcessor getProcessor() {
45          return processor;
46      }
47      
48      public LoadingFieldProcessor getProcessorAsLoading() {
49          return (LoadingFieldProcessor) processor;
50      }
51      
52      public SavingFieldProcessor getProcessorAsSaving() {
53          return (SavingFieldProcessor) processor;
54      }
55      
56      public FieldAdaptor getAdaptor() {
57          return adaptor;
58      }
59      
60      public void loadProcess(final Sheet sheet, final Object beanObj, final XlsMapperConfig config, final LoadingWorkObject work) throws XlsMapperException {
61          getProcessorAsLoading().loadProcess(sheet, beanObj, getAnnotation(), adaptor, config, work);
62      }
63      
64      public void saveProcess(final Sheet sheet, final Object beanObj, final XlsMapperConfig config, final SavingWorkObject work) throws XlsMapperException {
65          getProcessorAsSaving().saveProcess(sheet, beanObj, getAnnotation(), adaptor, config, work);
66      }
67  }
