18  /**
19   * Javassitを利用して、{@link Annotation}のインスタンスを動的に作成するクラス。
20   * <p>独自のClassLoaderを設定することが可能。
21   * <p>このクラスはシングルトンです。
22   * 
23   * @version 0.5
24   * @author Naoki Takezoe
25   * @author T.TSUCHIE
26   *
27   */
28  public class DynamicAnnotationBuilder {
29      
30      /**
31       * シングルトンのインスタンス。
32       */
33      private static DynamicAnnotationBuilder INSTANCE = new DynamicAnnotationBuilder();
34      
35      private ClassLoader classLoader;
36      
37      private OgnlContext ognlContext;
38      
39      private DynamicAnnotationBuilder() {
40          this.ognlContext = new OgnlContext();
41          this.ognlContext.setMemberAccess(new DefaultMemberAccess(true));
42          
43      }
44      
45      /**
46       * インスタンスを取得する。
47       * @return
48       */
49      public static DynamicAnnotationBuilder getInstance() {
50          return INSTANCE;
51      }
52      
53      /**
54       * アノテーションが定義された{@link ClassLoader}を設定する。
55       * <p>nullの場合、標準のClassLoaderが使用される。
56       * @param classLoader
57       */
58      public static void init(final ClassLoader classLoader) {
59          getInstance().classLoader = classLoader;
60      }
61      
62      /**
63       * アノテーションが定義されたClassLoaderとJavaBeansを定義したClassLoaderを設定する。
64       * @param classLoader アノテーションを見つけるために使用するClassLoader
65       * @param propertyLoaders JavaBeansを見つけるために使用するClassLoader
66       */
67      public static void init(final ClassLoader classLoader, final ClassLoader[] propertyLoaders) {
68          getInstance().classLoader = classLoader;
69          
70          if(propertyLoaders != null && propertyLoaders.length != 0) {
71              Map<Integer, ClassLoader> loaderMap = new HashMap<>();
72              for(ClassLoader propertyLoader : propertyLoaders) {
73                  loaderMap.put(propertyLoader.hashCode(), propertyLoader);
74              }
75              
76              getInstance().ognlContext = new OgnlContext(loaderMap);
77              getInstance().ognlContext.setMemberAccess(new DefaultMemberAccess(true));
78              getInstance().ognlContext.setClassResolver(new MultipleLoaderClassResolver());
79              
80          }
81      }
82      
83      /**
84       * 指定したアノテーションのクラス情報から、アノテーションのインスタンスを組み立てる。
85       * @param annoClass アノテーションのクラス
86       * @param info アノテーションの情報
87       * @return アノテーションのインスタンス。
88       * @throws AnnotationReadException
89       */
90      public Annotation buildAnnotation(final Class<?> annoClass, final AnnotationInfo info) throws AnnotationReadException {
91          
92          final Map<String, Object> defaultValues = new HashMap<>();
93          for(Method method : annoClass.getMethods()) {
94              final Object defaultValue = method.getDefaultValue();
95              if(defaultValue != null) {
96                  defaultValues.put(method.getName(), defaultValue);
97              }
98          }
99          
100         final Map<String, Object> xmlValues = new HashMap<>();
101         for(String key : info.getAttributeKeys()) {
102             try {
103                 Object value = Ognl.getValue(info.getAttribute(key), ognlContext, new Object());
104                 xmlValues.put(key, value);
105             } catch(OgnlException e) {
106                 throw new AnnotationReadException(String.format("fail annotation attribute %s with ognl.", key), e);
107             }
108         }
109         
110         ClassLoader loader = classLoader;
111         if(loader == null){
112             loader = Thread.currentThread().getContextClassLoader();
113         }
114         
115         Object obj = Proxy.newProxyInstance(loader, new Class[]{annoClass},
116                 new InvocationHandler() {
117                     @Override
118                     public Object invoke(final Object proxy, final Method method, final Object[] args) throws Throwable {
119                         String name = method.getName();
120                         if (name.equals("annotationType")) {
121                             return annoClass;
122                         } else if(xmlValues.containsKey(name)){
123                             return xmlValues.get(name);
124                         } else {
125                             return defaultValues.get(name);
126                         }
127                     }
128         });
129         
130         return (Annotation) obj;
131     }
