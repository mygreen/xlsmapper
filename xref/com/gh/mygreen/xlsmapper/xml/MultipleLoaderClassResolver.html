3   import java.util.Collection;
4   import java.util.HashSet;
5   import java.util.Iterator;
6   import java.util.Map;
7   
8   import ognl.ClassResolver;
9   
10  /**
11   * ClassResolver loading from multiple ClassLoader in Ognl.
12   *
13   * @author Mitsuyoshi Hasegawa
14   */
15  public class MultipleLoaderClassResolver implements ClassResolver {
16  
17      /**
18       * Loads class from multiple ClassLoader.
19       *
20       * If this method can not load target class,
21       * it tries to add package java.lang(default package)
22       *  and load target class.
23       * Still, if it can not the class, throws ClassNotFoundException.
24       * (behavior is put together on DefaultClassResolver.)
25       *
26       * @param className class name --  that wants to load it.
27       * @param loaderMap map -- that has VALUES ClassLoader (KEYS are arbitary).
28       * @return loaded class from ClassLoader defined loaderMap.
29       */
30      @SuppressWarnings({"unchecked", "rawtypes"})
31      public Class classForName(String className, Map loaderMap) throws ClassNotFoundException {
32          Collection<ClassLoader> loaders = null;
33          // add context-classloader
34          loaders = new HashSet<ClassLoader>();
35          loaders.add(Thread.currentThread().getContextClassLoader());
36          if (loaderMap != null && !loaderMap.isEmpty()) {
37              loaders.addAll(loaderMap.values());
38          }
39          Class clazz = null;
40          ClassNotFoundException lastCause = null;
41          for (Iterator<ClassLoader> it = loaders.iterator(); it.hasNext();) {
42              ClassLoader loader = it.next();
43              try {
44                  clazz = loader.loadClass(className);
45                  return clazz;
46              } catch (ClassNotFoundException fqcnException) {
47                  lastCause = fqcnException;
48                  try {
49                      if (className.indexOf('.') == -1) {
50                          clazz = loader.loadClass("java.lang." + className);
51                          return clazz;
52                      }
53                  } catch (ClassNotFoundException defaultClassException) {
54                      lastCause = defaultClassException;
55                      // try next loader.
56                  }
57              }
58          }
59          
60          if(lastCause != null) {
61              // can't load class at end.
62              throw lastCause;
63          }
64          
65          return clazz;
66      }
67  }
