21  /**
22   * XMLのアノテーション情報を保持する
23   * 
24   * <pre>
25   *  XMLの使用：
26   *  <annotation name="net.java.amateras.xlsbeans.annotation.Sheet"> <- 属性 「name」を持ち必須。
27   *  <attribute name="name">'Users'</attribute> <- 
28   *  </annotation>
29   * 
30   * </pre>
31   * 
32   * @version 1.1
33   * @since 0.5
34   * @author T.TSUCHIE
35   *
36   */
37  public class AnnotationInfo implements Serializable {
38      
39      /** serialVersionUID */
40      private static final long serialVersionUID = 1L;
41      
42      /**
43       * アノテーションのクラス名
44       */
45      private String className;
46      
47      /**
48       * アノテーションの属性値のマップ。
49       * <p>キー：属性名。
50       * <p>値：属性値。
51       * 
52       */
53      private Map<String, String> attributes = new LinkedHashMap<>();
54      
55      /**
56       * ビルダクラスのインスタンスを取得する。
57       * @since 1.1
58       * @return
59       */
60      public static Builder builder() {
61          return new Builder();
62      }
63      
64      /**
65       * ビルダクラスのインスタンスを取得する。
66       * @since 1.1
67       * @param valueFormatter JavaオブジェクトをOGNL式に変換するためのクラス。
68       * @return
69       * @throws IllegalArgumentException valueFormatter is null.
70       */
71      public static Builder builder(final OgnlValueFormatter valueFormatter) {
72          ArgUtils.notNull(valueFormatter, "valueFormatter");
73          return new Builder(valueFormatter);
74      }
75      
76      public AnnotationInfo() {
77          
78      }
79      
80      private AnnotationInfo(Builder builder) {
81          this.className = builder.className;
82          setAttributeInfos(builder.attributeInfos);
83      }
84      
85      @Override
86      public String toString() {
87          
88          StringBuilder sb = new StringBuilder();
89          sb.append("AnnotationInfo:")
90              .append(String.format(" [name=%s]", getClassName()));
91          
92          for(Map.Entry<String, String> entry : attributes.entrySet()) {
93              sb.append(String.format(" [(attr)%s=%s]", entry.getKey(), entry.getValue()));
94          }
95          
96          return sb.toString();
97          
98      }
99      
100     /**
101      * アノテーションのクラス名を取得する。
102      * @return FQCN（完全限定クラス名）。
103      */
104     public String getClassName() {
105         return className;
106     }
107     
108     /**
109      * アノテーションのクラス名を設定する。
110      * @param className FQCN（完全限定クラス名）を指定します。
111      * @throws IllegalArgumentException className is empty.
112      */
113     @XmlAttribute(name="name", required=true)
114     public void setClassName(String className) {
115         ArgUtils.notEmpty(className, "className");
116         this.className = className;
117     }
118     
119     /**
120      * アノテーションの属性を追加する。
121      * @param name 属性名。必須です。
122      * @param value 値。
123      *              <a href="http://s2container.seasar.org/2.4/ja/ognl.html" target="_blank">OGNL形式</a>で指定します。
124      * @throws IllegalArgumentException name is empty.
125      */
126     public void addAttribute(final String name, final String value) {
127         ArgUtils.notEmpty(name, "name");
128         this.attributes.put(name, value);
129     }
130     
131     /**
132      * アノテーションの属性名の一覧を取得する。
133      * @return 属性名の一覧情報。
134      */
135     public String[] getAttributeKeys() {
136         return this.attributes.keySet().toArray(new String[attributes.size()]);
137     }
138     
139     /**
140      * アノテーションの属性名を指定して、アノテーションの値を取得する。
141      * @param name 属性名。
142      * @return 存在しない属性名の場合、nullを返します。
143      */
144     public String getAttribute(final String name) {
145         return this.attributes.get(name);
146     }
147     
148     /**
149      * 指定したアノテーションの属性情報を含むかどうか。
150      * @since 1.1
151      * @param name アノテーションの属性名。
152      * @return true: 指定したアノテーションの属性名が存在する場合。
153      */
154     public boolean containsAttribute(final String name) {
155         return this.attributes.containsKey(name);
156     }
157     
158     /**
159      * アノテーションの属性情報を保持するクラス。
160      * <p>JAXBによるXMLのマッピングのために使用する。
161      *
162      */
163     @XmlAccessorType(XmlAccessType.FIELD)
164     public static class AtttributeInfo {
165         
166         @XmlAttribute(name="name", required=true)
167         String name;
168         
169         @XmlValue
170         String value;
171         
172         public static AtttributeInfo create(final String name, final String value) {
173             
174             AtttributeInfo attr = new AtttributeInfo();
175             attr.name = name;
176             attr.value = value;
177             return attr;
178         }
179         
180     }
181     
182     /**
183      * JAXB用のアノテーションの属性情報を設定するメソッド。
184      * <p>XMLの読み込み時に呼ばれます。
185      * <p>既存の情報はクリアされます。
186      * @since 1.1
187      * @param attributeInfos アノテーションの属性情報。
188      */
189     @XmlElement(name="attribute")
190     public void setAttributeInfos(List<AtttributeInfo> attributeInfos) {
191         this.attributes.clear();
192         for(AtttributeInfo attr : attributeInfos) {
193             this.attributes.put(attr.name, attr.value);
194         }
195     }
196     
197     /**
198      * JAXB用のアノテーションの属性情報を取得するメソッド。
199      * <p>XMLの書き込み時に呼ばれます。
200      * @since 1.1
201      * @return
202      */
203     public List<AtttributeInfo> getAttributeInfos() {
204         List<AtttributeInfo> attrs = new ArrayList<>();
205         for(Map.Entry<String, String> entry : attributes.entrySet()) {
206             attrs.add(AtttributeInfo.create(entry.getKey(), entry.getValue()));
207         }
208         
209         return attrs;
210     }

212     /**
213      * {@link AnnotationInfo}を組み立てるためのクラス。
214      *
215      */
216     public static final class Builder {
217         
218         private static final OgnlValueFormatter DEFAULT_VALUE_FORMATTER = new OgnlValueFormatter();
219         
220         private OgnlValueFormatter valueFormatter;
221         
222         private String className;
223         
224         private List<AtttributeInfo> attributeInfos;
225         
226         private Builder() {
227             this(DEFAULT_VALUE_FORMATTER);
228         }
229         
230         private Builder(final OgnlValueFormatter valueFormatter) {
231             this.valueFormatter = valueFormatter;
232             this.attributeInfos = new ArrayList<>();
233             
234         }
235         
236         /**
237          * 組み立てた{@link AnnotationInfo}のインスタンスを取得する。
238          * @return
239          */
240         public AnnotationInfo buildAnnotation() {
241             
242             if(Utils.isEmpty(className)) {
243                 throw new IllegalStateException("class name is required.");
244             }
245             
246             return new AnnotationInfo(this);
247             
248         }
249         
250         /**
251          * アノテーションのクラス名を設定する。
252          * @param className アノテーションのクラス名。FQCN（完全限定クラス名）を指定します。
253          * @return
254          * @throws IllegalArgumentException className is empty.
255          */
256         public Builder name(final String className) {
257             ArgUtils.notEmpty(className, "className");
258             this.className = className;
259             return this;
260         }
261         
262         /**
263          * アノテーションのクラス名を設定する。
264          * @param clazz アノテーションのクラス。
265          * @return
266          * @throws IllegalArgumentException clazz is null.
267          */
268         public Builder name(final Class<? extends Annotation> clazz) {
269             ArgUtils.notNull(clazz, "clazz");
270             return name(clazz.getName());
271         }
272         
273         /**
274          * アノテーションの属性値を設定する。
275          * @param attrName 属性名
276          * @param attrValue 属性の値。
277          *              <a href="http://s2container.seasar.org/2.4/ja/ognl.html" target="_blank">OGNL形式</a>で指定します。
278          * @return
279          * @throws IllegalArgumentException attrName is empty.
280          */
281         public Builder attributeWithNative(final String attrName, final String attrValue) {
282             ArgUtils.notEmpty(attrName, "attrName");
283             this.attributeInfos.add(AtttributeInfo.create(attrName, attrValue));
284             return this;
285         }
286         
287         /**
288          * アノテーションの属性値を設定する。
289          * @param attrName 属性名
290          * @param attrValue 属性の値。値は自動的にOGNLの書式に変換されて設定されます。
291          * @return
292          * @throws IllegalArgumentException attrName is empty.
293          */
294         public Builder attribute(final String attrName, final Object attrValue) {
295             ArgUtils.notEmpty(attrName, "attrName");
296             
297             String ognlValue = valueFormatter.format(attrValue);
298             attributeWithNative(attrName, ognlValue);
299             
300             return this;
301         }
302         
303     }
