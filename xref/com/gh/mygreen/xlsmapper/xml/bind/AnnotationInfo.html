19  /**
20   * XMLのアノテーション情報を保持する。
21   * 
22   * XMLの使用例：
23   * 
24   * <pre class="highlight"><code class="xml">
25   * {@literal <!-- 属性 「name」を持ち必須。--> }
26   * {@literal <annotation name="net.java.amateras.xlsbeans.annotation.Sheet">}
27   *     {@literal <attribute name="name">'Users'</attribute>}
28   * {@literal </annotation>}
29   * </code></pre>
30   * 
31   * @version 1.4.1
32   * @since 0.5
33   * @author T.TSUCHIE
34   *
35   */
36  public class AnnotationInfo implements Serializable {
37      
38      /** serialVersionUID */
39      private static final long serialVersionUID = 1L;
40      
41      /**
42       * アノテーションのクラス名
43       */
44      private String className;
45      
46      private List<AttributeInfo> attributes = new ArrayList<>();
47      
48      /**
49       * ビルダクラスのインスタンスを取得する。
50       * @since 1.1
51       * @return
52       */
53      public static Builder builder() {
54          return new Builder();
55      }
56      
57      /**
58       * ビルダクラスのインスタンスを取得する。
59       * @since 1.1
60       * @param valueFormatter JavaオブジェクトをOGNL式に変換するためのクラス。
61       * @return
62       * @throws IllegalArgumentException valueFormatter is null.
63       */
64      public static Builder builder(final OgnlValueFormatter valueFormatter) {
65          ArgUtils.notNull(valueFormatter, "valueFormatter");
66          return new Builder(valueFormatter);
67      }
68      
69      public AnnotationInfo() {
70          
71      }
72      
73      private AnnotationInfo(final Builder builder) {
74          this.className = builder.className;
75          setAttributeInfos(builder.attributeInfos);
76      }
77      
78      @Override
79      public String toString() {
80          
81          StringBuilder sb = new StringBuilder();
82          sb.append("AnnotationInfo:")
83              .append(String.format(" [name=%s]", getClassName()));
84          
85          for(AttributeInfo entry : attributes) {
86              sb.append(String.format(" [(attr)%s=%s]", entry.name, entry.value));
87          }
88          
89          return sb.toString();
90          
91      }
92      
93      /**
94       * アノテーションのクラス名を取得する。
95       * @return FQCN（完全限定クラス名）。
96       */
97      public String getClassName() {
98          return className;
99      }
100     
101     /**
102      * アノテーションのクラス名を設定する。
103      * @param className FQCN（完全限定クラス名）を指定します。
104      * @throws IllegalArgumentException className is empty.
105      */
106     @XmlAttribute(name="name", required=true)
107     public void setClassName(String className) {
108         ArgUtils.notEmpty(className, "className");
109         this.className = className;
110     }
111     
112     /**
113      * アノテーションの属性を追加する。
114      * <p>ただし、既に同じ属性名が存在する場合は、それと入れ替えされます。</p>
115      * @param name 属性名。必須です。
116      * @param value 値。
117      *              <a href="http://s2container.seasar.org/2.4/ja/ognl.html" target="_blank">OGNL形式</a>で指定します。
118      * @throws IllegalArgumentException name is empty.
119      */
120     public void addAttribute(final String name, final String value) {
121         ArgUtils.notEmpty(name, "name");
122         removeAttribute(name);
123         this.attributes.add(AttributeInfo.create(name, value));
124     }
125     
126     /**
127      * アノテーションの属性名の一覧を取得する。
128      * @return 属性名の一覧情報。
129      */
130     public String[] getAttributeKeys() {
131         
132         final List<String> list = new ArrayList<>(attributes.size());
133         for(AttributeInfo item : attributes) {
134             list.add(item.name);
135         }
136         
137         return list.toArray(new String[list.size()]);
138     }
139     
140     /**
141      * アノテーションの属性名を指定して、アノテーションの値を取得する。
142      * @param name 属性名。
143      * @return 存在しない属性名の場合、nullを返します。
144      */
145     public String getAttribute(final String name) {
146         for(AttributeInfo item : attributes) {
147             if(item.name.equals(name)) {
148                 return item.value;
149             }
150         }
151         
152         return null;
153     }
154     
155     /**
156      * 指定したアノテーションの属性情報を含むかどうか。
157      * @since 1.1
158      * @param name アノテーションの属性名。
159      * @return true: 指定したアノテーションの属性名が存在する場合。
160      */
161     public boolean containsAttribute(final String name) {
162         return getAttribute(name) != null;
163     }
164     
165     /**
166      * 指定したアノテーションの属性情報を削除します。
167      * @since 1.4.1
168      * @param name アノテーションの属性名。
169      * @return true:指定したアノテーション属性名を含み、それが削除できた場合。
170      */
171     public boolean removeAttribute(final String name) {
172         
173         AttributeInfo existInfo = null;
174         for(AttributeInfo item : attributes) {
175             if(item.name.equals(name)) {
176                 existInfo = item;
177                 break;
178             }
179             
180         }
181         
182         if(existInfo != null) {
183             this.attributes.remove(existInfo);
184             return true;
185         }
186         
187         return false;
188         
189     }

191     /**
192      * アノテーションの属性情報を保持するクラス。
193      * <p>JAXBによるXMLのマッピングのために使用する。</p>
194      *
195      */
196     @XmlAccessorType(XmlAccessType.FIELD)
197     public static class AttributeInfo implements Serializable {
198         
199         /** serialVersionUID */
200         private static final long serialVersionUID = 5570368711168203217L;
201         
202         @XmlAttribute(name="name", required=true)
203         String name;
204         
205         @XmlValue
206         String value;
207         
208         public static AttributeInfo create(final String name, final String value) {
209             
210             AttributeInfo attr = new AttributeInfo();
211             attr.name = name;
212             attr.value = value;
213             return attr;
214         }
215         
216     }
217     
218     /**
219      * JAXB用のアノテーションの属性情報を設定するメソッド。
220      * <p>XMLの読み込み時に呼ばれます。
221      *  <br>ただし、Java8からはこのメソッドは呼ばれず、{@link #getAttributeInfos()} で取得したインスタンスに対して要素が追加されます。
222      * </p>
223      * <p>既存の情報はクリアされます。</p>
224      * @since 1.1
225      * @param attributeInfos アノテーションの属性情報。
226      */
227     @XmlElement(name="attribute")
228     public void setAttributeInfos(final List<AttributeInfo> attributeInfos) {
229         if(attributeInfos == this.attributes) {
230             // Java7の場合、getterで取得したインスタンスをそのまま設定するため、スキップする。
231             return;
232         }
233         
234         this.attributes.clear();
235         for(AttributeInfo attr : attributeInfos) {
236             addAttribute(attr.name, attr.value);
237         }
238     }
239     
240     /**
241      * JAXB用のアノテーションの属性情報を取得するメソッド。
242      * <p>XMLの書き込み時に呼ばれます。
243      *  <br>Java8から読み込み時に呼ばれるようになり、取得したインスタンスに対して、読み込んだ要素が呼ばれます。
244      * </p>
245      * @since 1.1
246      * @return
247      */
248     public List<AttributeInfo> getAttributeInfos() {
249         return attributes;
250     }

252     /**
253      * {@link AnnotationInfo}を組み立てるためのクラス。
254      *
255      */
256     public static final class Builder {
257         
258         private static final OgnlValueFormatterl#OgnlValueFormatter">OgnlValueFormatter DEFAULT_VALUE_FORMATTER = new OgnlValueFormatter();
259         
260         private OgnlValueFormatter valueFormatter;
261         
262         private String className;
263         
264         private List<AttributeInfo> attributeInfos;
265         
266         private Builder() {
267             this(DEFAULT_VALUE_FORMATTER);
268         }
269         
270         private Builder(final OgnlValueFormatter valueFormatter) {
271             this.valueFormatter = valueFormatter;
272             this.attributeInfos = new ArrayList<>();
273             
274         }
275         
276         /**
277          * 組み立てた{@link AnnotationInfo}のインスタンスを取得する。
278          * @return
279          */
280         public AnnotationInfo buildAnnotation() {
281             if(Utils.isEmpty(className)) {
282                 throw new IllegalStateException("class name is required.");
283             }
284             
285             return new AnnotationInfo(this);
286             
287         }
288         
289         /**
290          * アノテーションのクラス名を設定する。
291          * @param className アノテーションのクラス名。FQCN（完全限定クラス名）を指定します。
292          * @return
293          * @throws IllegalArgumentException className is empty.
294          */
295         public Builder name(final String className) {
296             ArgUtils.notEmpty(className, "className");
297             this.className = className;
298             return this;
299         }
300         
301         /**
302          * アノテーションのクラス名を設定する。
303          * @param clazz アノテーションのクラス。
304          * @return
305          * @throws IllegalArgumentException clazz is null.
306          */
307         public Builder name(final Class<? extends Annotation> clazz) {
308             ArgUtils.notNull(clazz, "clazz");
309             return name(clazz.getName());
310         }
311         
312         /**
313          * アノテーションの属性値を設定する。
314          * @param attrName 属性名
315          * @param attrValue 属性の値。
316          *              <a href="http://s2container.seasar.org/2.4/ja/ognl.html" target="_blank">OGNL形式</a>で指定します。
317          * @return
318          * @throws IllegalArgumentException attrName is empty.
319          */
320         public Builder attributeWithNative(final String attrName, final String attrValue) {
321             ArgUtils.notEmpty(attrName, "attrName");
322             this.attributeInfos.add(AttributeInfo.create(attrName, attrValue));
323             return this;
324         }
325         
326         /**
327          * アノテーションの属性値を設定する。
328          * @param attrName 属性名
329          * @param attrValue 属性の値。値は自動的にOGNLの書式に変換されて設定されます。
330          * @return
331          * @throws IllegalArgumentException attrName is empty.
332          */
333         public Builder attribute(final String attrName, final Object attrValue) {
334             ArgUtils.notEmpty(attrName, "attrName");
335             
336             String ognlValue = valueFormatter.format(attrValue);
337             attributeWithNative(attrName, ognlValue);
338             
339             return this;
340         }
341         
342     }
