102     public <P> SheetBindingErrors<P> loadDetail(final InputStream xlsIn, final Class<P> clazz)
103             throws XlsMapperException, IOException {
104 
105         ArgUtils.notNull(xlsIn, "xlsIn");
106         ArgUtils.notNull(clazz, "clazz");
107 
108         final AnnotationReaderder.html#AnnotationReader">AnnotationReader annoReader = new AnnotationReader(configuration.getAnnotationMapping().orElse(null));
109 
110         final XlsSheet sheetAnno = annoReader.getAnnotation(clazz, XlsSheet.class);
111         if(sheetAnno == null) {
112             throw new AnnotationInvalidException(sheetAnno, MessageBuilder.create("anno.notFound")
113                     .varWithClass("property", clazz)
114                     .varWithAnno("anno", XlsSheet.class)
115                     .format());
116         }
117 
118         Workbook book = null;
119         try {
120             book = WorkbookFactory.create(xlsIn);
121 
122         } catch (InvalidFormatException e) {
123             throw new XlsMapperException(MessageBuilder.create("file.failLoadExcel.notSupportType").format(), e);
124         } finally {
125             if(book != null) {
126                 book.close();
127             }
128         }
129 
130         try {
131             final Sheet[] xlsSheet = configuration.getSheetFinder().findForLoading(book, sheetAnno, annoReader, clazz);
132             return loadSheet(xlsSheet[0], clazz, annoReader);
133 
134         } catch(SheetNotFoundException e) {
135             if(configuration.isIgnoreSheetNotFound()){
136                 logger.warn(MessageBuilder.create("log.skipNotFoundSheet").format(), e);
137                 return null;
138 
139             } else {
140                 throw e;
141             }
142         }
143     }

173      * @return 複数のシートのマッピング結果。
174      *         {@link Configuration#isIgnoreSheetNotFound()}の値がtrueで、シートが見つからない場合、マッピング結果には含まれません。
175      * @throws IllegalArgumentException {@literal xlsIn == null or clazz == null}

179     public <P> MultipleSheetBindingErrors<P> loadMultipleDetail(final InputStream xlsIn, final Class<P> clazz)
180             throws XlsMapperException, IOException {
181 
182         ArgUtils.notNull(xlsIn, "xlsIn");
183         ArgUtils.notNull(clazz, "clazz");
184 
185         final AnnotationReaderder.html#AnnotationReader">AnnotationReader annoReader = new AnnotationReader(configuration.getAnnotationMapping().orElse(null));
186 
187         final XlsSheet sheetAnno = annoReader.getAnnotation(clazz, XlsSheet.class);
188         if(sheetAnno == null) {
189             throw new AnnotationInvalidException(sheetAnno, MessageBuilder.create("anno.notFound")
190                     .varWithClass("property", clazz)
191                     .varWithAnno("anno", XlsSheet.class)
192                     .format());
193         }
194 
195         final MultipleSheetBindingErrors<P> multipleResult = new MultipleSheetBindingErrors<>();
196 
197         Workbook book = null;
198         try {
199             book = WorkbookFactory.create(xlsIn);
200 
201         } catch (InvalidFormatException e) {
202             throw new XlsMapperException(MessageBuilder.create("file.failLoadExcel.notSupportType").format(), e);
203         } finally {
204             if(book != null) {
205                 book.close();
206             }
207         }
208 
209         if(sheetAnno.number() == -1 && sheetAnno.name().isEmpty() && sheetAnno.regex().isEmpty()) {
210             // 読み込むシートの条件が指定されていない場合、全て読み込む
211             int sheetNum = book.getNumberOfSheets();
212             for(int i=0; i < sheetNum; i++) {
213                 final Sheet sheet = book.getSheetAt(i);
214 
215                 multipleResult.addBindingErrors(loadSheet(sheet, clazz, annoReader));
216 
217             }
218 
219         } else {
220             // 読み込むシートの条件が指定されている場合
221             try {
222                 final Sheet[] xlsSheet = configuration.getSheetFinder().findForLoading(book, sheetAnno, annoReader, clazz);
223                 for(Sheet sheet : xlsSheet) {
224                     multipleResult.addBindingErrors(loadSheet(sheet, clazz, annoReader));
225 
226                 }
227 
228             } catch(SheetNotFoundException e) {
229                 if(configuration.isIgnoreSheetNotFound()){
230                     logger.warn(MessageBuilder.create("log.skipNotFoundSheet").format(), e);
231                 } else {
232                     throw e;
233                 }
234             }
235 
236         }
237 
238         return multipleResult;
239     }

326      * シートを読み込み、任意のクラスにマッピングする。
327      * @param sheet シート情報
328      * @param clazz マッピング先のクラスタイプ。

334     private <P> SheetBindingErrors<P> loadSheet(final Sheet sheet, final Class<P> clazz, final AnnotationReader annoReader)
335             throws XlsMapperException {
336 
337         // 値の読み込み対象のJavaBeanオブジェクトの作成
338         final P beanObj = configuration.createBean(clazz);
339 
340         final SheetBindingErrors<P> errors =  configuration.getBindingErrorsFactory().create(beanObj);
341         errors.setSheetName(sheet.getSheetName());
342         errors.setSheetIndex(sheet.getWorkbook().getSheetIndex(sheet));
343 
344         final LoadingWorkObjectbject.html#LoadingWorkObject">LoadingWorkObject work = new LoadingWorkObject();
345         work.setAnnoReader(annoReader);
346         work.setErrors(errors);
347 
348         // セルのキャッシュ情報の初期化
349         configuration.getCellFormatter().init(configuration.isCacheCellValueOnLoad());
350 
351         final FieldAccessorFactorysorFactory.html#FieldAccessorFactory">FieldAccessorFactory adpterFactory = new FieldAccessorFactory(annoReader);
352 
353         // リスナークラスの@PreLoad用メソッドの実行
354         final XlsListener listenerAnno = annoReader.getAnnotation(beanObj.getClass(), XlsListener.class);
355         if(listenerAnno != null) {
356             for(Class<?> listenerClass : listenerAnno.value()) {
357                 final Object listenerObj = configuration.createBean(listenerClass);
358 
359                 for(Method method : listenerObj.getClass().getMethods()) {
360                     if(annoReader.hasAnnotation(method, XlsPreLoad.class)) {
361                         Utils.invokeNeedProcessMethod(listenerObj, method, beanObj, sheet, configuration, work.getErrors(), ProcessCase.Load);
362                     }
363                 }
364             }
365 
366         }
367 
368         // @PreLoad用のメソッドの実行
369         for(Method method : clazz.getMethods()) {
370 
371             if(annoReader.hasAnnotation(method, XlsPreLoad.class)) {
372                 Utils.invokeNeedProcessMethod(beanObj, method, beanObj, sheet, configuration, work.getErrors(), ProcessCase.Load);
373             }
374         }
375 
376         final List<FieldAccessorProxy> accessorProxies = new ArrayList<>();
377 
378         // public メソッドの処理
379         for(Method method : clazz.getMethods()) {
380             method.setAccessible(true);
381             for(Annotation anno : annoReader.getAnnotations(method)) {
382                 final XlsFieldProcessor annoFieldProcessor = anno.annotationType().getAnnotation(XlsFieldProcessor.class);
383                 if(ClassUtils.isAccessorMethod(method) && annoFieldProcessor != null) {
384                     // 登録済みのFieldProcessorの取得
385                     FieldProcessor<?> processor = configuration.getFieldProcessorRegistry().getProcessor(anno.annotationType());
386 
387                     // アノテーションに指定されているFieldProcessorの場合
388                     if(processor == null && annoFieldProcessor.value().length > 0) {
389                         processor = configuration.createBean(annoFieldProcessor.value()[0]);
390 
391                     }
392 
393                     if(processor != null) {
394                         final FieldAccessor accessor = adpterFactory.create(method);
395                         final FieldAccessorProxyessorProxy.html#FieldAccessorProxy">FieldAccessorProxy accessorProxy = new FieldAccessorProxy(anno, processor, accessor);
396                         if(!accessorProxies.contains(accessorProxy)) {
397                             accessorProxies.add(accessorProxy);
398                         }
399 
400                     } else {
401                         // FieldProcessorが見つからない場合
402                         throw new AnnotationInvalidException(anno, MessageBuilder.create("anno.XlsFieldProcessor.notResolve")
403                                 .varWithAnno("anno", anno.annotationType())
404                                 .format());
405                     }
406 
407                 }
408 
409                 if(anno instanceof XlsPostLoad) {
410                     work.addNeedPostProcess(new NeedProcess(beanObj, beanObj, method));
411                 }
412             }
413 
414         }
415 
416         // フィールドの処理
417         for(Field field : clazz.getDeclaredFields()) {
418 
419             field.setAccessible(true);
420             for(Annotation anno : annoReader.getAnnotations(field)) {
421 
422                 final XlsFieldProcessor annoFieldProcessor = anno.annotationType().getAnnotation(XlsFieldProcessor.class);
423                 if(annoFieldProcessor != null) {
424 
425                     // 登録済みのFieldProcessorの取得
426                     FieldProcessor<?> processor = configuration.getFieldProcessorRegistry().getProcessor(anno.annotationType());
427 
428                     // アノテーションに指定されているFieldProcessorの場合
429                     if(processor == null && annoFieldProcessor.value().length > 0) {
430                         processor = configuration.createBean(annoFieldProcessor.value()[0]);
431 
432                     }
433 
434                     if(processor != null) {
435                         final FieldAccessor accessor = adpterFactory.create(field);
436                         final FieldAccessorProxyessorProxy.html#FieldAccessorProxy">FieldAccessorProxy accessorProxy = new FieldAccessorProxy(anno, processor, accessor);
437                         if(!accessorProxies.contains(accessorProxy)) {
438                             accessorProxies.add(accessorProxy);
439                         }
440 
441                     } else {
442                         // FieldProcessorが見つからない場合
443                         throw new AnnotationInvalidException(anno, MessageBuilder.create("anno.XlsFieldProcessor.notResolve")
444                                 .varWithAnno("anno", anno.annotationType())
445                                 .format());
446                     }
447 
448                 }
449 
450 
451             }
452         }
453 
454         // 順番を並び替えて保存処理を実行する
455         Collections.sort(accessorProxies, new FieldAccessorProxyComparator());
456         for(FieldAccessorProxy accessorProxy : accessorProxies) {
457             accessorProxy.loadProcess(sheet, beanObj, configuration, work);
458         }
459 
460         // リスナークラスの@PostLoadの取得
461         if(listenerAnno != null) {
462             for(Class<?> listenerClass : listenerAnno.value()) {
463                 Object listenerObj = configuration.createBean(listenerClass);
464                 for(Method method : listenerObj.getClass().getMethods()) {
465                     if(annoReader.hasAnnotation(method, XlsPostLoad.class)) {
466                         work.addNeedPostProcess(new NeedProcess(beanObj, listenerObj, method));
467                     }
468                 }
469             }
470 
471         }
472 
473         //@PostLoadが付与されているメソッドの実行
474         for(NeedProcess need : work.getNeedPostProcesses()) {
475             Utils.invokeNeedProcessMethod(need.getProcess(), need.getMethod(), need.getTarget(), sheet, configuration, work.getErrors(), ProcessCase.Load);
476         }
477 
478         // セルのキャッシュ情報の初期化
479         configuration.getCellFormatter().init(configuration.isCacheCellValueOnLoad());
480 
481         return errors;
482     }
