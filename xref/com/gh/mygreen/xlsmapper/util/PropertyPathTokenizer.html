1   package com.gh.mygreen.xlsmapper.util;
2   
3   import java.util.LinkedList;
4   
5   import com.gh.mygreen.xlsmapper.util.PropertyPath.Token;
6   
7   /**
8    * プロパティアクセス用の式をパースする。
9    * <p>{@link PropertyValueNavigator}で使用する。
10   * @since 1.0
11   * @author T.TSUCHIE
12   *
13   */
14  public class PropertyPathTokenizer {
15      
16      /**
17       * エスケープ文字（バックスペース\\）
18       */
19      private static final String STR_ESCAPE_CHAR = "\\";
20      
21      /**
22       * プロパティのアクセス用の書式をトークンに分割する。
23       * @param path
24       * @return
25       * @throws IllegalArgumentException path is null or empty.
26       */
27      public PropertyPath parse(final String path) {
28          
29          ArgUtils.notEmpty(path, "path");
30          
31          final PropertyPath tokenStore = new PropertyPath(path);
32          splitToken(tokenStore, path);
33          return tokenStore;
34      }
35      
36      /**
37       * パスを最小限の項目に分割する。
38       * 
39       * @param tokenStore 分割した結果
40       * @param path パス
41       * 
42       */
43      private void splitToken(final PropertyPath tokenStore, final String path) {
44          
45          // 解析時の途中の文字を一時的に補完しておく。
46          final LinkedList<String> stack = new LinkedList<String>();
47          
48          final int length = path.length();
49          for(int i=0; i < length; i++) {
50              final char c = path.charAt(i);
51              
52              if(StackUtils.equalsTopElement(stack, STR_ESCAPE_CHAR)) {
53                  // スタックの一番上がエスケープの文字の場合、通常の文字として扱う。
54                  stack.push(String.valueOf(c));
55                  
56              } else if (c == '.') {
57                  
58                  if(StackUtils.equalsBottomElement(stack, "[")) {
59                      // キーの囲み文字の途中の場合、文字列としてスタックに積む。
60                      stack.push(String.valueOf(c));
61                      
62                  } else if(!stack.isEmpty()) {
63                      // 既に文字が入っている場合は、既存のものを取り出し分割する。
64                      tokenStore.add(new Token.Name(StackUtils.popupAndConcat(stack).trim()));
65                      tokenStore.add(new Token.Separator());
66                      
67                  } else {
68                      tokenStore.add(new Token.Separator());
69                  }
70                  
71              } else if(c == '[') {
72                  
73                  if(!stack.isEmpty()) {
74                      // 既に文字が入っている場合は、既存のものを取り出し分割する。
75                      tokenStore.add(new Token.Name(StackUtils.popupAndConcat(stack).trim()));
76                  }
77                  
78                  stack.push(String.valueOf(c));
79                  
80              } else if(c == ']') {
81                  if(StackUtils.equalsBottomElement(stack, "[")) {
82                      // 条件の終わりの場合
83                      tokenStore.add(new Token.Key(StackUtils.popupAndConcat(stack) + c));
84                      
85                  } else {
86                      stack.push(String.valueOf(c));
87                  }
88                  
89              } else if(c == ' ') {
90                  // キーの囲み文字の中であれば、空白は無視しない。
91                  if(StackUtils.equalsBottomElement(stack, "[")) {
92                      stack.push(String.valueOf(c));
93                  }
94                  
95              } else {
96                  stack.push(String.valueOf(c));
97              }
98              
99          }
100         
101         if(!stack.isEmpty()) {
102             tokenStore.add(new Token.Name(StackUtils.popupAndConcat(stack).trim()));
103         }
104         
105     }
