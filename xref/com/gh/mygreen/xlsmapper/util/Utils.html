44      /**
45       * コレクションの要素を指定した区切り文字で繋げて1つの文字列とする。
46       * @param col 処理対象のコレクション。
47       * @param separator 区切り文字。
48       * @param ignoreEmptyElement 空、nullの要素を無視するかどうか。
49       * @param trim トリムをするかどうか。
50       * @param elementConverter 要素を変換するクラス。
51       * @return 結合した文字列
52       */
53      @SuppressWarnings("rawtypes")
54      public static String join(final Collection<?> col, final String separator,
55              final boolean ignoreEmptyElement, final boolean trim, final ElementConverter elementConverter) {
56  
57          final List<Object> list = new ArrayList<Object>();
58          for(Object element : col) {
59              if(element == null) {
60                  continue;
61              }
62  
63              Object value = element;
64  
65              if(element instanceof String) {
66                  String str = (String) element;
67                  if(ignoreEmptyElement && isEmpty(str)) {
68                      continue;
69  
70                  } else if(trim) {
71                      value = str.trim();
72                  }
73  
74              } else if(element instanceof Character && isEmpty(element.toString())) {
75                  String str = element.toString();
76                  if(ignoreEmptyElement && isEmpty(str)) {
77                      continue;
78  
79                  } else if(trim) {
80                      value = str.trim().charAt(0);
81                  }
82  
83              } else if(char.class.isAssignableFrom(element.getClass())) {
84                  String str = element.toString();
85                  if(ignoreEmptyElement && isEmpty(str)) {
86                      continue;
87  
88                  } else if(trim) {
89                      value = str.trim().charAt(0);
90                  }
91              }
92  
93              list.add(value);
94  
95          }
96  
97          return join(list, separator, elementConverter);
98  
99      }

115      * @param arrays 結合対象の配列
116      * @param separator 区切り文字
117      * @param elementConverter 要素を変換するクラス。
118      * @return 結合した文字列
119      */
120     @SuppressWarnings({"rawtypes", "unchecked"})
121     public static String join(final Object[] arrays, final String separator, final ElementConverter elementConverter) {
122 
123         if(arrays == null) {
124             return "";
125         }
126 
127         final int len = arrays.length;
128         if(len == 0) {
129             return "";
130         }
131 
132         StringBuilder sb = new StringBuilder();
133         for(int i=0; i < len; i++) {
134             final Object element = arrays[i];
135             sb.append(elementConverter.convertToString(element));
136 
137             if(separator != null && (i < len-1)) {
138                 sb.append(separator);
139             }
140         }
141 
142         return sb.toString();
143 
144     }

157      * Collectionの要素を指定した区切り文字で繋げて1つの文字列とする。
158      * @param col 結合対象のコレクション
159      * @param separator 区切り文字
160      * @param elementConverter 要素を変換するクラス。
161      * @return 結合した文字列
162      */
163     @SuppressWarnings({"rawtypes", "unchecked"})
164     public static String join(final Collection<?> col, final String separator, final ElementConverter elementConverter) {
165 
166         if(col == null) {
167             return "";
168         }
169 
170         final int size = col.size();
171         if(size == 0) {
172             return "";
173         }
174 
175         StringBuilder sb = new StringBuilder();
176         for(Iterator<?> itr = col.iterator(); itr.hasNext();) {
177             final Object element = itr.next();
178             String text = elementConverter.convertToString(element);
179             sb.append(text);
180 
181             if(separator != null && itr.hasNext()) {
182                 sb.append(separator);
183             }
184         }
185 
186         return sb.toString();
187 
188     }

249         } else {
250             return normalize(text1, config).equals(normalize(text2, config));
251 //            return normalize(text1, config).equals(text2);

446     /**
447      * PostProcessなどのメソッドを実行する。
448      * <p>メソッドの引数が既知のものであれば、インスタンスを設定する。
449      *
450      * @param processObj 実行対象の処理が埋め込まれているオブジェクト。
451      * @param method 実行対象のメソッド情報
452      * @param beanObj 処理対象のBeanオブジェクト。
453      * @param sheet シート情報
454      * @param config 共通設定
455      * @param errors エラー情報
456      * @param processCase 処理ケース
457      * @throws XlsMapperException
458      */
459     public static void invokeNeedProcessMethod(final Object processObj, final Method method, final Object beanObj,
460             final Sheet sheet, final Configuration config, final SheetBindingErrors<?> errors, final ProcessCase processCase)
461                     throws XlsMapperException {
462 
463         final Class<?>[] paramTypes = method.getParameterTypes();
464         final Object[] paramValues =  new Object[paramTypes.length];
465 
466         for(int i=0; i < paramTypes.length; i++) {
467             if(Sheet.class.isAssignableFrom(paramTypes[i])) {
468                 paramValues[i] = sheet;
469 
470             } else if(Configuration.class.isAssignableFrom(paramTypes[i])) {
471                 paramValues[i] = config;
472 
473             } else if(SheetBindingErrors.class.isAssignableFrom(paramTypes[i])) {
474                 paramValues[i] = errors;
475 
476             } else if(paramTypes[i].isAssignableFrom(beanObj.getClass())) {
477                 paramValues[i] = beanObj;
478 
479             } else if(ProcessCase.class.equals(paramTypes[i])) {
480                 paramValues[i] = processCase;
481 
482             } else if(paramTypes[i].equals(Object.class)) {
483                 paramValues[i] = beanObj;
484 
485             } else {
486                 paramValues[i] = null;
487             }
488         }
489 
490         try {
491             method.setAccessible(true);
492             method.invoke(processObj, paramValues);
493         } catch (IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
494             Throwable t = e.getCause() == null ? e : e.getCause();
495             throw new XlsMapperException(
496                     String.format("fail execute method '%s#%s'.", processObj.getClass().getName(), method.getName()),
497                     t);
498         }
499     }

507     public static Locale getLocale(final String str) {
508

511         }
512 
513         if(!str.contains("_")) {
514             return new Locale(str);
515         }
516 
517         final String[] split = str.split("_");
518         if(split.length == 2) {
519             return new Locale(split[0], split[1]);
520 
521         } else {
522             return new Locale(split[0], split[1], split[2]);
523         }

527     /**
528      * エスケープ文字を除去した文字列を取得する。
529      * @param str
530      * @param escapeChar
531      * @return
532      */
533     public static String removeEscapeChar(final String str, final char escapeChar) {
534 
535         if(str == null || str.isEmpty()) {
536             return str;
537         }
538 
539         final String escapeStr = String.valueOf(escapeChar);
540         StringBuilder sb = new StringBuilder();
541 
542         LinkedList<String> stack = new LinkedList<String>();
543 
544         final int length = str.length();
545         for(int i=0; i < length; i++) {
546             final char c = str.charAt(i);
547 
548             if(StackUtils.equalsTopElement(stack, escapeStr)) {
549                 // スタックの一番上がエスケープ文字の場合
550                 StackUtils.popup(stack);
551                 sb.append(c);
552 
553             } else if(c == escapeChar) {
554                 // スタックに積む
555                 stack.push(String.valueOf(c));
556 
557             } else {
558                 sb.append(c);
559             }
560 
561         }
562 
563         if(!stack.isEmpty()) {
564             sb.append(StackUtils.popupAndConcat(stack));
565         }
566 
567         return sb.toString();
568 
569     }

572      * Listのインスタンスを他のCollectionのインスタンスに変換する。
573      * <p>ただし、変換先のクラスタイプがインタフェースの場合は変換しない。
574      * <p>変換元のクラスと変換先のクラスが同じ場合は、変換しない。
575      *
576      * @since 1.0
577      * @param list 変換元のListのインスタンス
578      * @param toClass 変換先のCollectionのクラス
579      * @param beanFactory てインスタンスを生成するファクトリクラス。
580      * @return 変換したコレクションのインスタンス
581      */
582     @SuppressWarnings({"rawtypes", "unchecked"})
583     public static Collection convertListToCollection(final List list, final Class<Collection> toClass,
584             final BeanFactory<Class<?>, Object> beanFactory) {
585 
586         if(list.getClass().equals(toClass)) {
587             return list;
588         }
589 
590         if(toClass.isInterface()) {
591             if(List.class.isAssignableFrom(toClass)) {
592                 // 変換先がListの実態の場合はそのまま。
593                 return list;
594 
595             } else if(Set.class.isAssignableFrom(toClass)) {
596 
597                 Collection value = (Collection) beanFactory.create(LinkedHashSet.class);
598                 value.addAll(list);
599                 return value;
600 
601             } else if(Queue.class.isAssignableFrom(toClass)) {
602 
603                 Collection value = (Collection) beanFactory.create(LinkedList.class);
604                 value.addAll(list);
605                 return value;
606 
607             } else if(Collection.class.isAssignableFrom(toClass)) {
608                 Collection value = (Collection) beanFactory.create(ArrayList.class);
609                 value.addAll(list);
610                 return value;
611 
612             } else {
613                 throw new IllegalArgumentException("not support class type:" + toClass.getName());
614             }
615 
616         }
617 
618         Collection value = (Collection) beanFactory.create(toClass);
619         value.addAll(list);
620 
621         return value;
622 
623     }

642     /**
643      * リストに要素のインデックスを指定して追加します。
644      * <p>リストのサイズが足りない場合は、サイズを自動的に変更します。</p>
645      * @since 2.0
646      * @param list リスト
647      * @param element 追加する要素。値はnullでもよい。
648      * @param index 追加する要素のインデックス番号(0以上)
649      * @throws IllegalArgumentException {@literal list == null.}
650      * @throws IllegalArgumentException {@literal index < 0.}
651      */
652     public static <P> void addListWithIndex(final List<P> list, final P element, final int index) {
653         ArgUtils.notNull(list, "list");
654         ArgUtils.notMin(index, 0, "index");
655 
656         final int listSize = list.size();
657         if(listSize < index) {
658             // 足りない場合は、要素を追加する。
659             final int lackSize = index - listSize;
660             for(int i=0; i < lackSize; i++) {
661                 list.add(null);
662             }
663             list.add(element);
664 
665         } else if(listSize == index) {
666             // 最後の要素に追加する
667             list.add(element);
668 
669         } else {
670             // リストのサイズが足りている場合
671             list.set(index, element);
672         }
673 
674     }

885     /**
886      * 文字列配列の結合
887      * @param array1
888      * @param array2
889      * @return 結合した配列。引数のどちらからnullの場合は、cloneした配列を返します。
890      */
891     public static String[] concat(final String[] array1, final String[] array2) {
892 
893         if(array1 == null || array1.length == 0) {
894             return clone(array2);
895 
896         } else if(array2 == null || array2.length == 0) {
897             return clone(array1);
898         }
899 
900         final String[] joinedArray = new String[array1.length + array2.length];
901         System.arraycopy(array1, 0, joinedArray, 0, array1.length);
902         System.arraycopy(array2, 0, joinedArray, array1.length, array2.length);
903         return joinedArray;
904 
905     }
