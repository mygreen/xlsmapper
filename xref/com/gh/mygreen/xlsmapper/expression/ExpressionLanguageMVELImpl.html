12  /**
13   * 式言語「MVEL」の実装。
14   * <p>利用する際には、MVEL2.xのライブラリが必要です。
15   * 
16   * @author T.TSUCHIE
17   *
18   */
19  public class ExpressionLanguageMVELImpl implements ExpressionLanguage {
20      
21      private static final Logger logger = LoggerFactory.getLogger(ExpressionLanguageMVELImpl.class);
22      
23      private final ObjectCache<String, Object> expressionCache = new ObjectCache<String, Object>();
24      
25      @Override
26      public Object evaluate(final String expression, final Map<String, ?> values) {
27          
28          ArgUtils.notEmpty(expression, "expression");
29          ArgUtils.notEmpty(values, "values");
30          
31          try {
32              if(logger.isDebugEnabled()) {
33                  logger.debug("Evaluating MVEL expression: {}", expression);
34              }
35              
36              Object expr = expressionCache.get(expression);
37              if (expr == null) {
38                  expr = MVEL.compileExpression(expression);
39                  expressionCache.put(expression, expr);
40              }
41              return MVEL.executeExpression(expr, values);
42          } catch(Exception ex) {
43              throw new ExpressionEvaluationException(String.format("Evaluating [%s] script with MVEL failed.", expression), ex);
44          }
45      }
46  }
