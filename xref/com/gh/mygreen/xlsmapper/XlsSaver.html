85      /**
86       * JavaのオブジェクトをExeclファイルに出力する。
87       * <p>出力するファイルは、引数で指定した雛形となるテンプレート用のExcelファイルをもとに出力する。</p>
88       *
89       * @param <P> マッピング対象のクラスタイプ
90       * @param templateXlsIn 雛形となるExcelファイルの入力
91       * @param xlsOut 出力先のストリーム
92       * @param beanObj 書き込むBeanオブジェクト
93       * @return マッピング結果。
94       *         {@link Configuration#isIgnoreSheetNotFound()}の値がtrueで、シートが見つからない場合、nullを返します。
95       * @throws IllegalArgumentException {@literal templateXlsIn == null or xlsOut == null or beanObj == null}
96       * @throws XlsMapperException マッピングに失敗した場合
97       * @throws IOException テンプレｰトのファイルの読み込みやファイルの出力に失敗した場合
98       */
99      public <P> SheetBindingErrors<P> saveDetail(final InputStream templateXlsIn, final OutputStream xlsOut, final P beanObj)
100             throws XlsMapperException, IOException {
101 
102         ArgUtils.notNull(templateXlsIn, "templateXlsIn");
103         ArgUtils.notNull(xlsOut, "xlsOut");
104         ArgUtils.notNull(beanObj, "beanObj");
105 
106 
107         final AnnotationReaderder.html#AnnotationReader">AnnotationReader annoReader = new AnnotationReader(configuration.getAnnotationMapping().orElse(null));
108 
109         try(Workbook book = WorkbookFactory.create(templateXlsIn)) {
110 
111             final Class<?> clazz = beanObj.getClass();
112             final XlsSheet sheetAnno = clazz.getAnnotation(XlsSheet.class);
113             if(sheetAnno == null) {
114                 throw new AnnotationInvalidException(sheetAnno, MessageBuilder.create("anno.notFound")
115                         .varWithClass("property", clazz)
116                         .varWithAnno("anno", XlsSheet.class)
117                         .format());
118     
119             }
120     
121             final SheetBindingErrors<P> bindingResult;
122             try {
123                 final Sheet[] xlsSheet = configuration.getSheetFinder().findForSaving(book, sheetAnno, annoReader, beanObj);
124                 bindingResult = saveSheet(xlsSheet[0], beanObj, annoReader);
125     
126             } catch(SheetNotFoundException e) {
127                 if(configuration.isIgnoreSheetNotFound()){
128                     logger.warn(MessageBuilder.create("log.skipNotFoundSheet").format(), e);
129                     return null;
130     
131                 } else {
132                     throw e;
133                 }
134             }
135     
136             if(configuration.isFormulaRecalcurationOnSave()) {
137                 book.setForceFormulaRecalculation(true);
138             }
139     
140             book.write(xlsOut);
141     
142             return bindingResult;
143             
144         } catch (InvalidFormatException e) {
145             throw new XlsMapperException(MessageBuilder.create("file.faiiLoadTemplateExcel.notSupportType").format(), e);
146         }
147 
148     }

280 
281         // public メソッドの処理
282         for(Method method : clazz.getMethods()) {

309                         throw new AnnotationInvalidException(anno, MessageBuilder.create("anno.XlsFieldProcessor.notResolve")
310                                 .varWithAnno("anno", anno.annotationType())
311                                 .format());
312                     }
313 
314                 }
315 
316                 if(anno instanceof XlsPostSave) {
317                     work.addNeedPostProcess(new NeedProcess(beanObj, beanObj, method));
318                 }
319             }
320         }
321

323         for(Field field : clazz.getDeclaredFields()) {
324

326             final FieldAccessor accessor = adpterFactory.create(field);
327 
328             for(Annotation anno : annoReader.getAnnotations(field)) {

364         // リスナークラスの@PostSaveの取得
365         if(listenerAnno != null) {
366             for(Class<?> listenerClass : listenerAnno.value()) {
367                 final Object listenerObj = configuration.createBean(listenerClass);
368                 for(Method method : listenerObj.getClass().getMethods()) {
369                     if(annoReader.hasAnnotation(method, XlsPostSave.class)) {

375         }

377         //@PostSaveが付与されているメソッドの実行
378         for(NeedProcess need : work.getNeedPostProcesses()) {
379             Utils.invokeNeedProcessMethod(need.getProcess(), need.getMethod(), need.getTarget(), sheet, configuration, work.getErrors(), ProcessCase.Save);
380         }
381 
382         return errors;
