12  
13  /**
14   * 書き込み処理中で持ち回すオブジェクトを保持するクラス。
15   *
16   */
17  public class SavingWorkObject {
18      
19      private AnnotationReader annoReader;
20      
21      private final List<NeedProcess> needPostProcesses = new ArrayList<NeedProcess>();
22      
23      private SheetBindingErrors errors;
24      
25      public AnnotationReader getAnnoReader() {
26          return annoReader;
27      }
28      
29      public void setAnnoReader(AnnotationReader annoReader) {
30          this.annoReader = annoReader;
31      }
32      
33      public void addNeedPostProcess(NeedProcess needProcess) {
34          this.needPostProcesses.add(needProcess);
35      }
36      
37      public List<NeedProcess> getNeedPostProcesses() {
38          return needPostProcesses;
39      }
40      
41      public SheetBindingErrors getErrors() {
42          return errors;
43      }
44      
45      public void setErrors(SheetBindingErrors errors) {
46          this.errors = errors;
47      }
48      
49      public void addTypeBindError(final TypeBindException e, final Point position, final String fieldName, final String label) {
50          this.errors.rejectSheetTypeBind(fieldName, e.getTargetValue(), e.getBindClass(),
51                  position, label);
52      }
53      
54      public void addTypeBindError(final TypeBindException e, final Cell cell, final String fieldName, final String label) {
55          this.errors.rejectSheetTypeBind(fieldName, e.getTargetValue(), e.getBindClass(),
56                  new Point(cell.getColumnIndex(), cell.getRowIndex()), label);
57      }
58  }
