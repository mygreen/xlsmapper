77      public XlsMapperConfig() {
78          
79          // 数式をフォーマットする際のEL関数を登録する。
80          ExpressionLanguageJEXLImpl formulaEL = new ExpressionLanguageJEXLImpl();
81          Map<String, Object> funcs = new HashMap<>(); 
82          funcs.put("x", CustomFunctions.class);
83          formulaEL.getJexlEngine().setFunctions(funcs);
84          
85          formulaFormatter.setExpressionLanguage(formulaEL);
86      }

263      * <p>アノテーション{@literal @XlsHorizontalRecords}で行の追加などを行うときに補試します。</p>
264      * <p>POI-3.15以上の場合の<a href="https://bz.apache.org/bugzilla/show_bug.cgi?id=59740" target="_blank">BUg 59740</a>において、

266      *   自動的に解除されるのを補正します。
267      * </p>
268      * @since 1.6
269      * @return trueのとき、セルノ結合を修正します。
270      *         初期値は、POI-3.15以上のときtrueとな、POI 3.14以前はfalseになります。
