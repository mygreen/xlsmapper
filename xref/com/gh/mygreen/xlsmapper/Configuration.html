80      public Configuration() {
81  
82          // 数式をフォーマットする際のEL関数を登録する。
83          ExpressionLanguageJEXLImpl formulaEL = new ExpressionLanguageJEXLImpl();
84          Map<String, Object> funcs = new HashMap<>();
85          funcs.put("x", CustomFunctions.class);
86          formulaEL.getJexlEngine().setFunctions(funcs);
87  
88          formulaFormatter.setExpressionLanguage(formulaEL);
89      }
