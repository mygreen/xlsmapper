246      * 指定したパスで現在のパスを初期化します。

251         final String canonicalPath = normalizePath(nestedPath);

267     /**
268      * パスを正規化する。
269      * <ol>
270      *  <li>トリムする。</li>
271      *  <li>値がnullの場合は、空文字を返す。</li>
272      *  <li>最後に'.'がついている場合、除去する。</li>
273      * </ol>
274      * @param subPath
275      * @return
276      */
277     private String normalizePath(final String subPath) {
278         if(subPath == null) {
279             return "";
280         }
281         
282         String value = subPath.trim();
283         if(value.isEmpty()) {
284             return value;
285         }
286         
287         if(value.startsWith(PATH_SEPARATOR)) {
288             value = value.substring(1);
289         }
290         
291         if(value.endsWith(PATH_SEPARATOR)) {
292             value = value.substring(0, value.length()-1);
293         }
294         
295         return value;
296         
297     }

299     /**
300      * パスを１つ下位に移動します。
301      * @param subPath ネストするパス
302      * @throws IllegalArgumentException subPath is empty.
303      */
304     public void pushNestedPath(final String subPath) {
305         final String canonicalPath = normalizePath(subPath);
306         ArgUtils.notEmpty(canonicalPath, "canonicalPath");
307         
308         this.nestedPathStack.push(canonicalPath);
309         this.currentPath = buildPath();
310     }

312     /**
313      * 配列やリストなどのインデックス付きのパスを１つ下位に移動します。
314      * @param subPath ネストするパス
315      * @param index インデックス番号(0から始まります。)
316      * @throws IllegalArgumentException {@literal subPath is empty or index < 0}
317      */
318     public void pushNestedPath(final String subPath, final int index) {
319         final String canonicalPath = normalizePath(subPath);
320         ArgUtils.notEmpty(subPath, "subPath");
321         ArgUtils.notMin(index, -1, "index");
322         
323         pushNestedPath(String.format("%s[%d]", canonicalPath, index));
324     }

329      * @param key マップのキー

332     public void pushNestedPath(final String subPath, final String key) {
333         final String canonicalPath = normalizePath(subPath);

335         ArgUtils.notEmpty(key, "key");
336

340     /**
341      * パスを１つ上位に移動します。
342      * @return 現在のパスを返しまます。
343      * @throws IllegalStateException {@literal パスがこれ以上移動できない場合}
344      */
345     public String popNestedPath() {
346         
347         if(nestedPathStack.isEmpty()) {
348             throw new IllegalStateException("Cannot pop nested path: no nested path on stack");
349         }
350         
351         final String subPath = nestedPathStack.pop();
352         this.currentPath = buildPath();
353         return subPath;
354     }

556     /**
557      * 指定したパスがフィールドエラーのパスと一致するかチェックするかどうか。
558      * @param path パス
559      * @param fieldError フィールドエラー
560      * @return true: 一致する場合。
561      */
562     private boolean isMatchingFieldError(final String path, final FieldError fieldError) {
563         
564         if (fieldError.getField().equals(path)) {
565             return true;
566         }
567         
568         if(path.endsWith("*")) {
569             String subPath = path.substring(0, path.length()-1);
570             return fieldError.getField().startsWith(subPath);
571         }
572         
573         return false;
574     }

613     /**
614      * フィールドエラーのビルダーを作成します。
615      * @param field フィールドパス。
616      * @param errorCodes エラーコード。先頭の要素が優先されます。
617      * @return {@link FieldError}のインスタンスを組み立てるビルダクラス。
618      */
619     public InternalFieldErrorBuilder createFieldError(final String field, final String[] errorCodes) {
620         
621         final String fieldPath = buildFieldPath(field);
622         final Class<?> fieldType = getFieldType(field);
623         final Object fieldValue = getFieldValue(field);
624         
625         String[] codes = new String[0];
626         for(String errorCode : errorCodes) {
627             codes = Utils.concat(codes, generateMessageCodes(errorCode, fieldPath, fieldType));
628         }
629         
630         return new InternalFieldErrorBuilder(this, getObjectName(), fieldPath, codes)
631                 .sheetName(getSheetName())
632                 .rejectedValue(fieldValue);
633                 
634     }

638      * @param field フィールドパス。
639      * @param fieldType フィールドのクラスタイプ
640      * @param rejectedValue 型変換に失敗した値
641      * @return {@link FieldError}のインスタンスを組み立てるビルダクラス。

643     public InternalFieldErrorBuilder createFieldConversionError(final String field, final Class<?> fieldType, final Object rejectedValue) {
644         
645         final String fieldPath = buildFieldPath(field);
646         final String[] codes = messageCodeGenerator.generateTypeMismatchCodes(getObjectName(), fieldPath, fieldType);
647         
648         return new InternalFieldErrorBuilder(this, getObjectName(), fieldPath, codes)
649                 .sheetName(getSheetName())
650                 .rejectedValue(rejectedValue)
651                 .conversionFailure(true);

671      * @since 2.0
672      * @param field フィールド名
673      * @param fieldType フィールドのクラスタイプ
674      * @param formatter フォーマッタ
675      * @param strippedIndex 登録するときにフィールドパスから、インデックス情報を除去するかどうか。
676      */
677     @SuppressWarnings({"unchecked", "rawtypes"})
678     public void registerFieldFormatter(final String field, final Class<?> fieldType, final FieldFormatter<?> formatter,
679             final boolean strippedIndex) {
680         
681         String fieldPath = buildFieldPath(field);
682         
683         if(strippedIndex) {
684             // パスからインデックスやキーを削除する
685             List<String> strippedPaths = new ArrayList<>();
686             fieldFormatterRegistry.addStrippedPropertyPaths(strippedPaths, "", fieldPath);
687             
688             if(strippedPaths.size() > 0) {
689                 // 辞書順位並び変えて先頭に来るのが、インデックスを全て削除されたパス
690                 Collections.sort(strippedPaths);
691                 fieldPath = strippedPaths.get(0);
692             }
693         }
694         
695         fieldFormatterRegistry.registerFormatter(fieldPath, (Class)fieldType, (FieldFormatter)formatter);
696         
697     }

699     /**
700      * フィールドとクラスタイプを指定してフォーマッタを取得する。
701      * @since 2.0
702      * @param field フィールド名
703      * @param fieldType フィールドのクラスタイプ
704      * @return 見つからない場合は、nullを返す。
705      */
706     public <T> FieldFormatter<T> findFieldFormatter(final String field, final Class<T> fieldType) {
707         String fieldPath = buildFieldPath(field);
708         return fieldFormatterRegistry.findFormatter(fieldPath, fieldType);
709     }
