64      /**
65       * 指定したパスで現在のパスを初期化します。
66       * <p>nullまたは空文字を与えると、トップに移動します。
67       * @param nestedPath
68       */
69      public void setNestedPath(final String nestedPath) {
70          final String canonicalPath = getCanonicalPath(nestedPath);
71          this.nestedPathStack.clear();
72          if(canonicalPath.isEmpty()) {
73              this.currentPath = buildPath();
74          } else {
75              pushNestedPath(canonicalPath);
76          }
77      }

86      /**
87       * パスを標準化する。
88       * <ol>
89       *  <li>トリムする。
90       *  <li>値がnullの場合は、空文字を返す。
91       *  <li>最後に'.'がついている場合、除去する。
92       * @param subPath
93       * @return
94       */
95      private String getCanonicalPath(final String subPath) {
96          if(subPath == null) {
97              return "";
98          }
99          
100         String value = subPath.trim();
101         if(value.isEmpty()) {
102             return value;
103         }
104         
105         if(value.startsWith(PATH_SEPARATOR)) {
106             value = value.substring(1);
107         }
108         
109         if(value.endsWith(PATH_SEPARATOR)) {
110             value = value.substring(0, value.length()-1);
111         }
112         
113         return value;
114         
115     }

117     /**
118      * パスを１つ下位に移動します。
119      * @param subPath
120      * @throws IllegalArgumentException subPath is empty.
121      */
122     public void pushNestedPath(final String subPath) {
123         final String canonicalPath = getCanonicalPath(subPath);
124         if(canonicalPath.isEmpty()) {
125             throw new IllegalArgumentException(String.format("subPath is invalid path : '%s'", subPath));
126         }
127         this.nestedPathStack.push(canonicalPath);
128         this.currentPath = buildPath();
129     }

137     public void pushNestedPath(final String subPath, final int index) {
138         pushNestedPath(String.format("%s[%d]", subPath, index));

151     /**
152      * パスを１つ上位に移動します。
153      * @return 
154      * @throws IllegalStateException path stask is empty.
155      */
156     public String popNestedPath() {
157         
158         if(nestedPathStack.isEmpty()) {
159             throw new IllegalStateException("Cannot pop nested path: no nested path on stack");
160         }
161         
162         final String subPath = nestedPathStack.pop();
163         this.currentPath = buildPath();
164         return subPath;
165     }

532     /**
533      * 指定したパスがフィールドエラーのパスと一致するかチェックするかどうか。
534      * @param path 
535      * @param fieldError
536      * @return true: 一致する場合。
537      */
538     protected boolean isMatchingFieldError(final String path, final FieldError fieldError) {
539         
540         if (fieldError.getFieldPath().equals(path)) {
541             return true;
542         }
543         
544         if(path.endsWith("*")) {
545             String subPath = path.substring(0, path.length()-1);
546             return fieldError.getFieldPath().startsWith(subPath);
547         }
548         
549         return false;
550     }
