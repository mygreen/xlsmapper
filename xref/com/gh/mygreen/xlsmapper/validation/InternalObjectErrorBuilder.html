3   import java.util.Map;
4   import java.util.Optional;
5   
6   /**
7    * {@link SheetBindingErrors}経由で呼び出される{@link ObjectError}のビルダクラス。
8    * <p>{@link #buildAndAddError()}でチェインでより使い安くする。</p>
9    * 
10   * @since 2.0
11   * @author T.TSUCHIE
12   *
13   */
14  public class InternalObjectErrorBuilder extends ObjectErrorBuilder {
15      
16      private final SheetBindingErrors<?> errors;
17      
18      /**
19       * ビルダのインスタンスを作成します。
20       * @param errors エラー情報
21       * @param objectName オブジェクト名
22       * @param codes メッセージコード。複数指定可能で、先頭にあるものほど優先度が高い。
23       */
24      public InternalObjectErrorBuilder(final SheetBindingErrors<?> errors, final String objectName, final String[] codes) {
25          super(objectName, codes);
26          this.errors = errors;
27          
28      }
29      
30      @Override
31      public InternalObjectErrorBuilder variables(final Map<String, Object> variables) {
32          super.variables(variables);
33          return this;
34      }
35      
36      @Override
37      public InternalObjectErrorBuilder variables(final String key, final Object value) {
38          super.variables(key, value);
39          return this;
40      }
41      
42      @Override
43      public InternalObjectErrorBuilder defaultMessage(final String defaultMessage) {
44          super.defaultMessage(defaultMessage);
45          return this;
46      }
47      
48      @Override
49      public InternalObjectErrorBuilder sheetName(final String sheetName) {
50          super.sheetName(sheetName);
51          return this;
52      }
53      
54      @Override
55      public InternalObjectErrorBuilder label(final String label) {
56          super.label(label);
57          return this;
58      }
59      
60      @Override
61      public InternalObjectErrorBuilder label(final Optional<String> label) {
62          super.label(label);
63          return this;
64      }
65      
66      /**
67       * {@link ObjectError}のインスタンスを組み立て、{@link SheetBindingErrors}にエラーとして追加します。
68       * @return {@link SheetBindingErrors}のインスタンス
69       */
70      public SheetBindingErrors<?> buildAndAddError() {
71          
72          final ObjectError error = build();
73          
74          errors.addError(error);
75          
76          return errors;
77      }
78      
79  }
