19  /**
20   * 名前付き変数のメッセージをフォーマットするクラス。
21   * <p><code>{...}</code>の場合、変数を単純に置換する。
22   * <p><code>${...}</code>の場合、EL式を利用し処理する。
23   * <p>文字'$', '{', '}'は特殊文字のため、<code>\</code>でエスケープを行う。
24   * <p>ELのパーサは、{@link ExpressionLanguage}の実装クラスで切り替え可能。
25   * <p>{@link MessageResolver}を指定した場合、メッセージ中の変数<code>{...}</code>をメッセージ定義コードとして解決する。
26   *    ただし、メッセージ変数で指定されている変数が優先される。
27   * 
28   * @version 1.0
29   * @author T.TSUCHIE
30   *
31   */
32  public class MessageInterpolator {
33      
34      private static final Logger logger = LoggerFactory.getLogger(MessageInterpolator.class);
35      
36      private final Formatter formatter = new Formatter();
37      
38      private ExpressionLanguage expressionLanguage = new ExpressionLanguageELImpl();
39      
40      public MessageInterpolator() {
41      }
42      
43      /**
44       * 式言語の実装を指定するコンストラクタ
45       * @param expressionLanguage
46       */
47      public MessageInterpolator(final ExpressionLanguage expressionLanguage) {
48          ArgUtils.notNull(expressionLanguage, "expressionLanguage");
49          this.expressionLanguage = expressionLanguage;
50      }
51      
52      /**
53       * メッセージを引数varsで指定した変数で補完する。
54       * 
55       * @param message 対象のメッセージ。
56       * @param vars メッセージ中の変数に対する値のマップ。
57       * @return 補完したメッセージ。
58       */
59      public String interpolate(final String message, final Map<String, ?> vars) {
60          return interpolate(message, vars, false);
61      }
62      
63      /**
64       * メッセージを引数varsで指定した変数で補完する。
65       * 
66       * @param message 対象のメッセージ。
67       * @param vars メッセージ中の変数に対する値のマップ。
68       * @param recursive 変換したメッセージに対しても再帰的に処理するかどうか
69       * @return 補完したメッセージ。
70       */
71      public String interpolate(final String message, final Map<String, ?> vars, boolean recursive) {
72          return parse(message, vars, recursive, null);
73      }
74      
75      /**
76       * メッセージを引数varsで指定した変数で補完する。
77       * <p>{@link MessageResolver}を指定した場合、メッセージ中の変数をメッセージコードとして解決します。
78       * 
79       * @param message 対象のメッセージ。
80       * @param vars メッセージ中の変数に対する値のマップ。
81       * @param recursive 変換したメッセージに対しても再帰的に処理するかどうか
82       * @param messageResolver メッセージを解決するクラス。nullの場合、指定しないと同じ意味になります。
83       * @return 補完したメッセージ。
84       */
85      public String interpolate(final String message, final Map<String, ?> vars, boolean recursive,
86              final MessageResolver messageResolver) {
87          return parse(message, vars, recursive, messageResolver);
88      }
89      
90      /**
91       * メッセージをパースし、変数に値を差し込み、EL式を評価する。
92       * @param message 対象のメッセージ。
93       * @param vars メッセージ中の変数に対する値のマップ。
94       * @param messageResolver メッセージを解決するクラス。nullの場合、指定しないと同じ意味になります。
95       * @return 補完したメッセージ。
96       */
97      protected String parse(final String message, final Map<String, ?> vars, boolean recursive, final MessageResolver messageResolver) {
98          
99          // 評価したメッセージを格納するバッファ。
100         final StringBuilder sb = new StringBuilder(message.length());
101         
102         /*
103          * 変数とEL式を解析する際に使用する、スタック変数。
104          * 式の開始が現れたらスタックに積み、式の終了が現れたらスタックから全てを取り出す。
105          * スタックに積まれるのは、1つ文の変数またはEL式。
106          */
107         final LinkedList<String> stack = new LinkedList<String>();
108         
109         final int length = message.length();
110         
111         for(int i=0; i < length; i++) {
112             final char c = message.charAt(i);
113             
114             if(StackUtils.equalsTopElement(stack, "\\")) {
115                 // 直前の文字がエスケープ文字の場合、エスケープ文字として結合する。
116                 String escapedChar = StackUtils.popup(stack) + c;
117                 
118                 if(!stack.isEmpty()) {
119                     // 取り出した後もスタックがある場合は、式の途中であるため、再度スタックに積む。
120                     stack.push(escapedChar);
121                     
122                 } else {
123                     // 取り出した後にスタックがない場合は、エスケープを解除して通常の文字として積む。
124                     sb.append(c);
125                     
126                 }
127                 
128             } else if(c == '\\') {
129                 // エスケープ文字の場合はスタックに積む。
130                 stack.push(String.valueOf(c));
131                 
132             } else if(c == '$') {
133                 stack.push(String.valueOf(c));
134                 
135             } else if(c == '{') {
136                 
137                 if(!stack.isEmpty() && !StackUtils.equalsAnyBottomElement(stack, new String[]{"$", "{"})) {
138                     // スタックの先頭が式の開始形式でない場合
139                     throw new MessageParseException(message, "expression not start with '{' or '$'");
140                     
141                 } else {
142                     stack.push(String.valueOf(c));
143                 }
144                 
145                 
146             } else if(c == '}') {
147                 
148                 if(StackUtils.equalsAnyBottomElement(stack, new String[]{"{", "$"})) {
149                     // 式の終わりの場合は、式を取り出し評価する。
150                     String expression = StackUtils.popupAndConcat(stack) + c;
151                     
152                     // エスケープを解除する
153                     expression = Utils.removeEscapeChar(expression, '\\');
154                     
155                     String result = evaluate(expression, vars, recursive, messageResolver);
156                     sb.append(result);
157                     
158                 } else {
159                     sb.append(c);
160                     
161                 }
162                 
163             } else {
164                 
165                 if(stack.isEmpty()) {
166                     sb.append(c);
167                     
168                 } else {
169                     stack.push(String.valueOf(c));
170                 }
171                 
172             }
173             
174         }
175         
176         if(!stack.isEmpty()) {
177             String val = StackUtils.popupAndConcat(stack);
178             val = Utils.removeEscapeChar(val, '\\');
179             sb.append(val);
180         }
181         
182         return sb.toString();
183     }
184     
185     private String evaluate(final String expression, final Map<String, ?> values, final boolean recursive,
186             final MessageResolver messageResolver) {
187         
188         if(expression.startsWith("{")) {
189             // 変数の置換の場合
190             final String varName = expression.substring(1, expression.length()-1);
191             
192             if(values.containsKey(varName)) {
193                 // 該当するキーが存在する場合
194                 final Object value = values.get(varName);
195                 final String eval = (value == null) ? "" : value.toString();
196                 if(!eval.isEmpty() && recursive) {
197                     return parse(eval, values, recursive, messageResolver);
198                 } else {
199                     return eval;
200                 }
201                 
202             } else if(messageResolver != null) {
203                 // メッセージコードをとして解決をする。
204                 final String eval = messageResolver.getMessage(varName);
205                 if(eval == null) {
206                     // 該当するキーが存在しない場合は、値をそのまま返す。
207                     return String.format("{%s}", varName);
208                 }
209                 
210                 if(recursive) {
211                     return parse(eval, values, recursive, messageResolver);
212                 } else {
213                     return eval;
214                 }
215                 
216             } else {
217                 // 該当するキーが存在しない場合は、値をそのまま返す。
218                 return expression.toString();
219             }
220             
221         } else if(expression.startsWith("${")) {
222             // EL式で処理する
223             final String expr = expression.substring(2, expression.length()-1);
224             final String eval = evaluateExpression(expr, values);
225             if(recursive) {
226                 return parse(eval, values, recursive, messageResolver);
227             } else {
228                 return eval;
229             }
230             
231         }
232         
233         throw new MessageParseException(expression, "not support expression.");
234         
235     }

237     /**
238      * EL式を評価する。
239      * @param expression EL式
240      * @param values EL式中の変数。
241      * @return 評価した式。
242      * @throws ExpressionEvaluationException 
243      */
244     protected String evaluateExpression(final String expression, final Map<String, ?> values) throws ExpressionEvaluationException {
245         
246         final Map<String, Object> context = new LinkedHashMap<String, Object>();
247         context.putAll(values);
248         
249         // フォーマッターの追加
250         if(!context.containsKey("formatter")) {
251             context.put("formatter", formatter);
252         }
253         
254         final String value = expressionLanguage.evaluate(expression, context).toString();
255         if(logger.isTraceEnabled()) {
256             logger.trace("evaluate expression language: expression='{}' ===> value='{}'", expression, value);
257         }
258         
259         return value;
260     }
