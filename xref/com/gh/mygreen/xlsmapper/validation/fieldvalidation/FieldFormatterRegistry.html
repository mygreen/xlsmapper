11  /**
12   * {@link FieldFormatter}を管理するためのクラス。
13   * <p>フィールドの値を検証する際に、フィールドの値をフォーマットしてエラーメッセージに埋め込むために使用します。</p>
14   * 
15   * @since 2.0
16   * @author T.TSUCHIE
17   *
18   */
19  public class FieldFormatterRegistry {
20      
21      /**
22       * クラスタイプで関連づけられたフォーマッタ
23       */
24      private Map<Class<?>, FieldFormatter<?>> typeMap = new HashMap<>();
25      
26      /**
27       * フィールドパスで関連づけられたフォーマッタ
28       */
29      private Map<String, FormatterHolder> pathMap = new HashMap<>();
30      
31      /**
32       * 登録されているフォーマッタを初期化する。
33       */
34      public void init() {
35          this.typeMap.clear();
36          this.pathMap.clear();
37          
38          
39      }
40      
41      /**
42       * クラスタイプを指定してフォーマッタを登録する。
43       * @param <T> 処理対象のタイプ
44       * @param requiredType クラスタイプ
45       * @param formatter 登録対象のフォーマッタ
46       * @throws IllegalArgumentException {@literal requiredType == null or formatter == null}
47       */
48      public <T> void registerFormatter(final Class<T> requiredType, FieldFormatter<T> formatter) {
49          ArgUtils.notNull(requiredType, "requiredType");
50          ArgUtils.notNull(formatter, "formatter");
51          
52          this.typeMap.put(requiredType, formatter);
53          
54      }
55      
56      /**
57       * フィールドパスとクラスタイプを指定してフォーマッタを登録する。
58       * @param <T> 処理対象のタイプ
59       * @param fieldPath
60       * @param requiredType クラスタイプ
61       * @param formatter 登録対象のフォーマッタ
62       */
63      public <T> void registerFormatter(final String fieldPath, final Class<T> requiredType, FieldFormatter<T> formatter) {
64          ArgUtils.notEmpty(fieldPath, "fieldPath");
65          ArgUtils.notNull(requiredType, "requiredType");
66          ArgUtils.notNull(formatter, "formatter");
67          
68          this.pathMap.put(fieldPath, new FormatterHolder(requiredType, formatter));
69          
70      }
71      
72      /**
73       * 指定したクラスタイプに対する{@link FieldFormatter}を取得する。
74       * @param <T> クラスタイプ
75       * @param requiredType 取得したいクラスタイプ
76       * @return 見つからない場合は、nullを返す。
77       */
78      @SuppressWarnings("unchecked")
79      public <T> FieldFormatter<T> getFormatter(final Class<T> requiredType) {
80          return (FieldFormatter<T>)typeMap.get(requiredType);
81      }
82      
83      /**
84       * 指定したパスとクラスタイプに対する{@link FieldFormatter}を取得する。
85       * <p>引数「fieldPath」に完全に一致するフォーマッタを取得します。</p>
86       * @param fieldPath フィールドパス。
87       * @param requiredType 取得したいクラスタイプ
88       * @return 見つからない場合は、nullを返す。
89       */
90      public <T> FieldFormatter<T> getFormatter(final String fieldPath, final Class<T> requiredType) {
91          FormatterHolder holder = pathMap.get(fieldPath);
92          return holder != null ? holder.getFormatter(requiredType) : null;
93      }
94      
95      /**
96       * 指定したパスとクラスタイプに対する{@link FieldFormatter}を取得する。
97       * <p>ただし、リストのインデックスやキーが含まれている場合、引数「fieldPath」からそれらを取り除いた値で比較する。
98       *  <br>一致するフィールドに対するフォーマッタが見つからない場合は、タイプのみで比較する。
99       * </p>
100      * @param fieldPath フィールドパス。
101      * @param requiredType 取得したいクラスタイプ
102      * @return 見つからない場合は、nullを返す。
103      */
104     public <T> FieldFormatter<T> findFormatter(final String fieldPath, final Class<T> requiredType) {
105         
106         // 完全なパスで比較
107         FieldFormatter<T> formatter = getFormatter(fieldPath, requiredType);
108         if(formatter != null) {
109             return formatter;
110         }
111         
112         // インデックスを除去した形式で比較
113         final List<String> strippedPaths = new ArrayList<>();
114         addStrippedPropertyPaths(strippedPaths, "", fieldPath);
115         
116         for(String strippedPath : strippedPaths) {
117             formatter = getFormatter(strippedPath, requiredType);
118             if(formatter != null) {
119                 return formatter;
120             }
121             
122         }
123         
124         // 見つからない場合は、タイプのみで比較した物を取得する
125         return getFormatter(requiredType);
126         
127         
128     }
129     
130     /**
131      * パスからリストのインデックス([1])やマップのキー([key])を除去したものを構成する。
132      * <p>SpringFrameworkの「PropertyEditorRegistrySupport#addStrippedPropertyPaths(...)」の処理</p>
133      * @param strippedPaths 除去したパス
134      * @param nestedPath 現在のネストしたパス
135      * @param propertyPath 処理対象のパス
136      */
137     public void addStrippedPropertyPaths(List<String> strippedPaths, String nestedPath, String propertyPath) {
138         
139         final int startIndex = propertyPath.indexOf('[');
140         if (startIndex != -1) {
141             final int endIndex = propertyPath.indexOf(']');
142             if (endIndex != -1) {
143                 final String prefix = propertyPath.substring(0, startIndex);
144                 final String key = propertyPath.substring(startIndex, endIndex + 1);
145                 final String suffix = propertyPath.substring(endIndex + 1, propertyPath.length());
146                 
147                 // Strip the first key.
148                 strippedPaths.add(nestedPath + prefix + suffix);
149                 
150                 // Search for further keys to strip, with the first key stripped.
151                 addStrippedPropertyPaths(strippedPaths, nestedPath + prefix, suffix);
152                 
153                 // Search for further keys to strip, with the first key not stripped.
154                 addStrippedPropertyPaths(strippedPaths, nestedPath + prefix + key, suffix);
155             }
156         }
157     }
158     
159     private static class FormatterHolder {
160         
161         private final Class<?> registerdType;
162         
163         private final FieldFormatter<?> formatter;
164         
165         private FormatterHolder(Class<?> registerdType, FieldFormatter<?> formatter) {
166             this.registerdType = registerdType;
167             this.formatter = formatter;
168         }
169         
170         @SuppressWarnings("unchecked")
171         private <T> FieldFormatter<T> getFormatter(final Class<T> requiredType) {
172             if(registerdType.isAssignableFrom(requiredType)) {
173                 return (FieldFormatter<T>)formatter;
174             } else {
175                 return null;
176             }
177         }
178     }
