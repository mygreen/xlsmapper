1   package com.gh.mygreen.xlsmapper.validation.fieldvalidation;
2   
3   import java.util.LinkedHashMap;
4   
5   import com.gh.mygreen.xlsmapper.ArgUtils;
6   
7   /**
8    * 値が指定した値以上かどうかの最小値のチェックする。
9    * <ul>
10   *  <li>メッセージキーは、「cellFieldError.min」。
11   * 
12   * @version 1.0
13   * @author T.TSUCHIE
14   *
15   */
16  public class MinValidator<T extends Comparable<T>> extends AbstractFieldValidator<T> {
17      
18      private final T min;
19      
20      /** エラーメッセージ中のための値のフォーマッタ */
21      private FieldFormatter<T> formatter;
22      
23      /**
24       * 最小値を指定するコンストラクタ
25       * @param min 下限値となる最小値。
26       */
27      public MinValidator(final T min) {
28          this(min, new DefaultFieldFormatter<T>(null));
29      }
30      
31      /**
32       * 値のフォーマットするための書式を指定するコンストラクタ
33       * @param min 下限値となる最小値
34       * @param pattern メッセージ中に表示するための値をフォーマットする際の書式。
35       */
36      public MinValidator(final T min, final String pattern) {
37          this(min, new DefaultFieldFormatter<T>(pattern));
38      }
39      
40      /**
41       * 値のフォーマッタ指定するコンストラクタ
42       * 
43       * @since 1.0
44       * @param min 下限値となる最小値。
45       * @param formatter エラーメッセージ中のための値のフォーマッタ
46       * @throws IllegalArgumentException formatter is null.
47       */
48      public MinValidator(final T min, final FieldFormatter<T> formatter) {
49          super();
50          
51          ArgUtils.notNull(min, "min");
52          ArgUtils.notNull(formatter, "formatter");
53          
54          this.min = min;
55          this.formatter = formatter;
56      }
57      
58      @Override
59      public String getDefaultMessageKey() {
60          return "cellFieldError.min";
61      }
62      
63      @Override
64      protected boolean validate(final T value) {
65          if(isNullValue(value)) {
66              return true;
67          }
68          
69          if(value.compareTo(getMin()) >= 0) {
70              return true;
71          }
72          
73          return false;
74      }
75      
76      @Override
77      protected LinkedHashMap<String, Object> getMessageVars(final T value) {
78          final LinkedHashMap<String, Object> vars = new LinkedHashMap<>();
79          vars.put("validatedValue", value);
80          vars.put("formattedValidatedValue", formatter.format(value));
81          vars.put("min", getMin());
82          vars.put("formattedMin", formatter.format(getMin()));
83          return vars;
84      }
85      
86      /**
87       * Validatorの下限値の最小値を取得する。
88       * @return 最小値の値。
89       */
90      public T getMin() {
91          return min;
92      }
93      
94  }
