4   import java.util.Arrays;

8   import com.gh.mygreen.xlsmapper.localization.MessageBuilder;

13  /**
14   * 配列や{@link Collection}型に対する入力値チェックをするためのクラス。
15   *
16   * @since 2.0
17   * @author T.TSUCHIE
18   *
19   */
20  public class ArrayCellField<E> extends CellField<E> {
21  
22      /**
23       * 要素が必須かどうか
24       */
25      private boolean requiredElement;
26  
27      /**
28       * 配列のフィールドに対するValidator
29       */
30      private List<ArrayFieldValidator<E>> arrayValidators;
31  
32      /**
33       * 指定されたフィールドの名称に対応するオブジェクトを構築します。
34       *
35       * @param fieldName フィールド名
36       * @param errors エラー情報
37       */
38      public ArrayCellField(String fieldName, SheetBindingErrors<?> errors) {
39          super(fieldName, errors);
40  
41          this.requiredElement = false;
42          this.arrayValidators = new ArrayList<>();
43      }
44  
45      /**
46       * 要素の値が必須かの設定を行う。
47       * @param requiredElement 必須チェックを行いたい場合、「true」を設定する。
48       * @return 自身のインスタンス。メソッドチェーンで記述する。
49       */
50      public ArrayCellField<E> setRequiredElement(final boolean requiredElement) {
51          this.requiredElement = requiredElement;
52          return this;
53      }
54  
55      /**
56       * 要素の値が必須かチェックを行うかどうか。
57       * @return true: 必須入力チェックを行う。
58       *               初期値は、非必須（オプション）です。
59       */
60      public boolean isRequiredElement() {
61          return requiredElement;
62      }
63  
64      /**
65       * フィールドの値をリストに変換して取得する。
66       * @return
67       */
68      @SuppressWarnings({"unchecked", "rawtypes"})
69      public List<E> getValueAsList() {
70  
71          final Object fieldValue = getValue();
72          final Class<?> fieldType = getType();
73  
74          if(Collection.class.isAssignableFrom(fieldType)) {
75              final Collection<Object> value = (fieldValue == null ? new ArrayList<Object>() : (Collection<Object>) fieldValue);
76              final List<Object> list = Utils.convertCollectionToList(value);
77              return (List)list;
78  
79          } else if(fieldType.isArray()) {
80              Class<?> componentType = fieldType.getComponentType();
81              final List<Object> list = Utils.asList(fieldValue, componentType);
82              return (List)list;
83  
84          } else {
85              throw new IllegalStateException(MessageBuilder.create("validation.notSupportType")
86                      .var("property", getFieldPath())
87                      .varWithClass("type", fieldType)
88                      .format());
89          }
90  
91      }
92  
93      /**
94       * フィールドの値が空かどうか。
95       * <p>値がnullまたは、文字列の場合空文字のとき、空と判定する。
96       *  <br>配列や{@link Collection}型のときは、要素の数が0のとき空と判定する。
97       * </p>
98       * @return trueの場合、値は空。
99       */
100     @SuppressWarnings("rawtypes")
101     public boolean isInputEmpty() {
102 
103         final Object fieldValue = getValue();
104         final Class<?> fieldType = getType();
105 
106         if(fieldValue == null) {
107             return true;
108 
109         } else if(Collection.class.isAssignableFrom(fieldType) && ((Collection) fieldValue).isEmpty()) {
110             return true;
111 
112         } else if(fieldType.isArray()) {
113 
114             Class<?> componentType = fieldType.getComponentType();
115             int length = Utils.getArraySize(fieldValue, componentType);
116             if(length > 0) {
117                 return false;
118             }
119 
120             return true;
121 
122         }
123 
124         return super.isInputEmpty();
125     }
126 
127     @Override
128     public ArrayCellField<E> validate(final Class<?>... groups) {
129 
130         // 既に型変換エラーなどがある場合、値が設定されていないため処理を終了します。
131         if(hasErrors()) {
132             return this;
133         }
134 
135         // 必須チェック
136         if(!validateForRequired()) {
137             return this;
138         }
139 
140         final List<Class<?>> hints = Arrays.asList(groups);
141 
142         // 配列用のValidatorの実行
143         if(getArrayValidators() != null && !getArrayValidators().isEmpty()) {
144             for(ArrayFieldValidator<E> validator : getArrayValidators()) {
145                 if(!validator.validate(this, hints)) {
146                     return this;
147                 }
148             }
149         }
150 
151         // 各要素に対するValidatorの実行
152         int size = getValueAsList().size();
153         if(size > 0) {
154             for(int i=0; i < size; i++) {
155                 final String elementName = String.format("%s[%d]", getFieldName(), i);
156 
157                 CellField<E> elementField = new CellField<>(elementName, getBindingErrors());
158                 elementField.setRequired(isRequiredElement())
159                     .add(getValidators())
160                     .validate(groups);
161             }
162 
163         }
164 
165         return this;
166     }
167 
168     /**
169      * {@link ArrayFieldValidator} を追加する。
170      * @param validator validatorのインスタンス。
171      * @return 自身のインスタンス。
172      * @throws NullPointerException validator is null.
173      */
174     public ArrayCellField<E> add(final ArrayFieldValidator<E> validator) {
175         ArgUtils.notNull(validator, "validator");
176         this.arrayValidators.add(validator);
177 
178         return this;
179     }
180 
181     /**
182      * 現在の{@link ArrayFieldValidator}を取得する。
183      * @return 現在設定されている{@link ArrayFieldValidator}。
184      */
185     public List<ArrayFieldValidator<E>> getArrayValidators() {
186         return arrayValidators;
187     }
188 
189 }
