4   
5   import com.gh.mygreen.xlsmapper.util.ArgUtils;
6   import com.gh.mygreen.xlsmapper.validation.fieldvalidation.AbstractFieldValidator;
7   import com.gh.mygreen.xlsmapper.validation.fieldvalidation.CellField;

10  /**
11   * 値が指定した範囲内かどうかチェックする。
12   * <p>メッセージキーは、「cellFieldError.range」。</p>
13   * <p>メッセージ中で利用可能な変数は次の通り。</p>
14   * <ul>
15   *   <li>「validatedValue」：検証対象の値のオブジェクト。</li>
16   *   <li>「min」：下限値となる最小値。</li>
17   *   <li>「max」：上限値となる最大値。</li>
18   *   <li>「inclusive」：値を比較する際に指定した値を含むかどうか。</li>
19   * </ul>
20   * 
21   * @version 2.0
22   * @author T.TSUCHIE
23   *
24   */
25  public class RangeValidator<T extends Comparable<T>> extends AbstractFieldValidator<T> {
26      
27      private final T min;
28      
29      private final T max;
30      
31      private final boolean inclusive;
32      
33      /**
34       * コンストラクタ
35       * 
36       * @since 1.0
37       * @param min 下限値となる最小値
38       * @param max 上限値となる最大値
39       * @throws IllegalArgumentException {@literal min == null or max == null}
40       * @throws IllegalArgumentException {@literal min > max}
41       */
42      public RangeValidator(final T min, final T max) {
43          this(min, max, true);
44      }
45      
46      /**
47       * コンストラクタ
48       * 
49       * @since 1.0
50       * @param min 下限値となる最小値
51       * @param max 上限値となる最大値
52       * @param inclusive 値を比較する際に指定した値を含むかどうかを指定します。
53       * @throws IllegalArgumentException {@literal min == null or max == null}
54       * @throws IllegalArgumentException {@literal min > max}
55       */
56      public RangeValidator(final T min, final T max, final boolean inclusive) {
57          ArgUtils.notNull(min, "min");
58          ArgUtils.notNull(max, "max");
59          ArgUtils.notMax(min, max, "min");
60          
61          this.min = min;
62          this.max = max;
63          this.inclusive = inclusive;
64      }
65      
66      @Override
67      public RangeValidator<T> addGroup(final Class<?>... group) {
68          return (RangeValidator<T>)super.addGroup(group);
69      }
70      
71      @Override
72      public String getMessageKey() {
73          return "cellFieldError.range";
74      }
75      
76      @Override
77      protected Map<String, Object> getMessageVariables(final CellField<T> cellField) {
78          final Map<String, Object> vars = super.getMessageVariables(cellField);
79          vars.put("min", getMin());
80          vars.put("max", getMax());
81          vars.put("inclusive", isInclusive());
82          return vars;
83      }
84      
85      @Override
86      protected void onValidate(final CellField<T> cellField) {
87          final T value = cellField.getValue();
88          if(validate(value)) {
89              return;
90          }
91          
92          error(cellField);
93      }
94      
95      private boolean validate(final T value) {
96          final int comparedMin = value.compareTo(min);
97          final int comparedMax = value.compareTo(max);
98          
99          if(comparedMin > 0 && comparedMax < 0) {
100             return true;
101         }
102         
103         if(inclusive && (comparedMin == 0 || comparedMax == 0)) {
104             return true;
105         }
106         
107         return false;
108         
109     }
110     
111     /**
112      * Validatorの下限値となる最小値を取得する。
113      * @return 最小値。
114      */
115     public T getMin() {
116         return min;
117     }
118     
119     /**
120      * Validatorの上限値となる最大値を取得する。
121      * @return 最大値。
122      */
123     public T getMax() {
124         return max;
125     }
126     
127     /**
128      * 値を比較する際に指定した値を含むかどうかを取得します。
129      * @return 値を比較する際に指定した値を含むかどうか
130      */
131     public boolean isInclusive() {
132         return inclusive;
133     }
134     
135 }
