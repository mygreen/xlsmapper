3   import java.util.Comparator;
4   import java.util.Optional;

6   import com.gh.mygreen.xlsmapper.annotation.XlsOrder;
7   
8   
9   /**
10   * アノテーション{@link XlsOrder}に従いフィールドの順番を並び替えるComparator。
11   * <ul>
12   *  <li>{@code @XlsOrder}の属性valueの順に並び替えます。</li>
13   *  <li>{@code @XlsOrder}の属性valueの値が同じ場合は、第2並び順としてフィールド名の昇順を使用します。</li>
14   *  <li>{@code @XlsOrder}が付与されていないフィールドは、付与されているフィールドよりも後になります。</li>
15   *  <li>{@code @XlsOrder}が付与されていないフィールドは、フィールド名の昇順で並び替えます。</li>
16   * </ul>
17   *
18   * @author T.TSUCHIE
19   *
20   */
21  public class FieldAccessorProxyComparator implements Comparator<FieldAccessorProxy> {
22  
23      @Override
24      public int compare(final FieldAccessorProxyssor/FieldAccessorProxy.html#FieldAccessorProxy">FieldAccessorProxy o1, final FieldAccessorProxy o2) {
25  
26          final Optional<Integer> order1 = getOrder(o1);
27          final Optional<Integer> order2 = getOrder(o2);
28  
29          if(!order1.isPresent() && !order2.isPresent()) {
30              // 並び順がない場合は、フィールド名の昇順
31              return o1.getField().getName().compareTo(o2.getField().getName());
32          } else if(!order1.isPresent()) {
33              return -1;
34          } else if(!order2.isPresent()) {
35              return 1;
36          }
37  
38          final int value1 = order1.get();
39          final int value2 = order2.get();
40          if(value1 == value2) {
41              // 並び順が同じ場合は、フィールド名の昇順
42              return o1.getField().getName().compareTo(o2.getField().getName());
43          } else if(value1 > value2) {
44              return 1;
45          } else {
46              return -1;
47          }
48  
49      }
50  
51      private Optional<Integer> getOrder(final FieldAccessorProxy accessorProxy) {
52  
53          return accessorProxy.getField().getAnnotation(XlsOrder.class)
54                  .map(anno -> anno.value());
55  
56      }
57  }
