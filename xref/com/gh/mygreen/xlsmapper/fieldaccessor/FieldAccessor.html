112     /**
113      * フィールドとメソッドに同じアノテーションが付与されているときに重複を除外するための判定に使用する。
114      * そのため、{@link FieldAccessor#getNameWithClass()}が等しいかで判定します。
115      * 
116      */
117     @Override
118     public boolean equals(final Object obj) {
119         if(this == obj) {
120             return true;
121         }
122         
123         if(!(obj instanceof FieldAccessor)) {
124             return false;
125         }
126         
127         final FieldAccessor target = (FieldAccessor)obj;
128         
129         // クラス名#フィールドの比較
130         if(!getNameWithClass().equals(target.getNameWithClass())) {
131             return false;
132         }
133         
134         return true;
135     }

156     /**
157      * オブジェクトのフィールドの値を取得します。
158      * <p>getterが存在する場合は、getterメソッド経由で取得します。</p>
159      * @param targetObj オブジェクト（インスタンス）
160      * @return フィールドの値。
161      * @throws IllegalArgumentException {@literal targetObj == null.}
162      * @throws FieldAccessException {@literal 値の取得に失敗した場合。}
163      */
164     public Object getValue(final Object targetObj) {
165         
166         ArgUtils.notNull(targetObj, "targetObj");
167         
168         if(targetGetter.isPresent()) {
169             try {
170                 return targetGetter.get().invoke(targetObj);
171             } catch (IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
172                 throw new FieldAccessException(this, "fail getter value", e);
173             }
174             
175         } else if(targetField.isPresent()) {
176             try {
177                 return targetField.get().get(targetObj);
178             } catch (IllegalArgumentException | IllegalAccessException e) {
179                 throw new FieldAccessException(this, "fail get field value", e);
180             }
181             
182         } else {
183             throw new FieldAccessException(this, "not found getter method or field.");
184         }
185         
186     }
187     
188     /**
189      * オブジェクトのフィールドに値を設定する。
190      * <p>setterが存在する場合は、setterメソッド経由で値を設定します。</p>
191      * @param targetObj オブジェクト（インスタンス）
192      * @param value フィールドの値。
193      * @throws IllegalArgumentException {@literal targetObj == null.}
194      * @throws FieldAccessException {@literal 値の設定に失敗した場合。}
195      * 
196      */
197     public void setValue(final Object targetObj, final Object value) {
198         ArgUtils.notNull(targetObj, "targetObj");
199         
200         if(targetSetter.isPresent()) {
201             try {
202                 targetSetter.get().invoke(targetObj, value);
203             } catch (IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
204                 throw new FieldAccessException(this, "fail setter value", e);
205             }
206             
207         } else if(targetField.isPresent()) {
208             try {
209                 targetField.get().set(targetObj, value);
210                 
211             } catch (IllegalArgumentException | IllegalAccessException e) {
212                 throw new FieldAccessException(this, "fail setter field value", e);
213             }
214             
215         } else {
216             throw new FieldAccessException(this, "not found setter method or field.");
217         }
218     }

220     /**
221      * フィールドがマップ形式の場合に、キーを指定して値を取得する。
222      * 
223      * @param key マップキーの値
224      * @param targetObj オブジェクト（インスタンス）
225      * @return マップの値
226      * @throws IllegalArgumentException {@literal targetObj == null.}
227      * @throws IllegalStateException {@literal フィールドのタイプがMap出ない場合}
228      */
229     @SuppressWarnings("unchecked")
230     public Object getValueOfMap(final Object key, final Object targetObj) {
231         ArgUtils.notNull(targetObj, "targetObj");
232         
233         if(!Map.class.isAssignableFrom(targetType)) {
234             throw new IllegalStateException("this method cannot call Map. This target type is " + targetType.getName());
235         }
236         
237         final Map<Object, Object> map = (Map<Object, Object>) getValue(targetObj);
238         if(map == null) {
239             return null;
240         }
241         
242         return map.get(key);
243         
244     }
