22  /**
23   * 名前付き変数のメッセージをフォーマットするクラス。
24   * <p><code>{...}</code>の場合、変数を単純に置換する。</p>
25   * <p><code>${...}</code>の場合、EL式を利用し処理する。</p>
26   * <p>文字'$', '{', '}'は特殊文字のため、<code>\</code>でエスケープを行う。</p>
27   * <p>ELのパーサは、{@link ExpressionLanguage}の実装クラスで切り替え可能。</p>
28   * <p>{@link MessageResolver}を指定した場合、メッセージ中の変数<code>{...}</code>をメッセージ定義コードとして解決する。
29   *   <br>ただし、メッセージ変数で指定されている変数が優先される。
30   * </p>
31   * 
32   * @version 2.0
33   * @author T.TSUCHIE
34   *
35   */
36  public class MessageInterpolator {
37      
38      private static final Logger logger = LoggerFactory.getLogger(MessageInterpolator.class);
39      
40      private ExpressionLanguage expressionLanguage;
41      
42      public MessageInterpolator() {
43          
44          // EL式中で使用可能な関数の登録
45          ExpressionLanguageJEXLImpl el = new ExpressionLanguageJEXLImpl();
46          
47          Map<String, Object> funcs = new HashMap<>(); 
48          funcs.put("f", CustomFunctions.class);
49          el.getJexlEngine().setFunctions(funcs);
50          
51          setExpressionLanguage(el);
52          
53      }

55      /**
56       * 式言語の実装を指定するコンストラクタ
57       * @param expressionLanguage
58       */
59      public MessageInterpolator(final ExpressionLanguage expressionLanguage) {
60          ArgUtils.notNull(expressionLanguage, "expressionLanguage");
61          this.expressionLanguage = expressionLanguage;
62      }

64      /**
65       * メッセージを引数varsで指定した変数で補完する。
66       * 
67       * @param message 対象のメッセージ。
68       * @param vars メッセージ中の変数に対する値のマップ。
69       * @return 補完したメッセージ。
70       */
71      public String interpolate(final String message, final Map<String, ?> vars) {
72          return interpolate(message, vars, false);
73      }
74      
75      /**
76       * メッセージを引数varsで指定した変数で補完する。
77       * 
78       * @param message 対象のメッセージ。
79       * @param vars メッセージ中の変数に対する値のマップ。
80       * @param recursive 変換したメッセージに対しても再帰的に処理するかどうか
81       * @return 補完したメッセージ。
82       */
83      public String interpolate(final String message, final Map<String, ?> vars, boolean recursive) {
84          return parse(message, vars, recursive, null);
85      }

87      /**
88       * メッセージを引数varsで指定した変数で補完する。
89       * <p>{@link MessageResolver}を指定した場合、メッセージ中の変数をメッセージコードとして解決します。
90       * 
91       * @param message 対象のメッセージ。
92       * @param vars メッセージ中の変数に対する値のマップ。
93       * @param recursive 変換したメッセージに対しても再帰的に処理するかどうか
94       * @param messageResolver メッセージを解決するクラス。nullの場合、指定しないと同じ意味になります。
95       * @return 補完したメッセージ。
96       */
97      public String interpolate(final String message, final Map<String, ?> vars, boolean recursive,
98              final MessageResolver messageResolver) {
99          return parse(message, vars, recursive, messageResolver);
100     }

102     /**
103      * メッセージをパースし、変数に値を差し込み、EL式を評価する。
104      * @param message 対象のメッセージ。
105      * @param vars メッセージ中の変数に対する値のマップ。
106      * @param messageResolver メッセージを解決するクラス。nullの場合、指定しないと同じ意味になります。
107      * @return 補完したメッセージ。
108      */
109     protected String parse(final String message, final Map<String, ?> vars, boolean recursive, final MessageResolver messageResolver) {
110         
111         // 評価したメッセージを格納するバッファ。
112         final StringBuilder sb = new StringBuilder(message.length());
113         
114         /*
115          * 変数とEL式を解析する際に使用する、スタック変数。
116          * 式の開始が現れたらスタックに積み、式の終了が現れたらスタックから全てを取り出す。
117          * スタックに積まれるのは、1つ文の変数またはEL式。
118          */
119         final LinkedList<String> stack = new LinkedList<String>();
120         
121         final int length = message.length();
122         
123         for(int i=0; i < length; i++) {
124             final char c = message.charAt(i);
125             
126             if(StackUtils.equalsTopElement(stack, "\\")) {
127                 // 直前の文字がエスケープ文字の場合、エスケープ文字として結合する。
128                 String escapedChar = StackUtils.popup(stack) + c;
129                 
130                 if(!stack.isEmpty()) {
131                     // 取り出した後もスタックがある場合は、式の途中であるため、再度スタックに積む。
132                     stack.push(escapedChar);
133                     
134                 } else {
135                     // 取り出した後にスタックがない場合は、エスケープを解除して通常の文字として積む。
136                     sb.append(c);
137                     
138                 }
139                 
140             } else if(c == '\\') {
141                 // エスケープ文字の場合はスタックに積む。
142                 stack.push(String.valueOf(c));
143                 
144             } else if(c == '$') {
145                 stack.push(String.valueOf(c));
146                 
147             } else if(c == '{') {
148                 
149                 if(!stack.isEmpty() && !StackUtils.equalsAnyBottomElement(stack, new String[]{"$", "{"})) {
150                     // スタックの先頭が式の開始形式でない場合
151                     throw new MessageParseException(message, "expression not start with '{' or '$'");
152                     
153                 } else {
154                     stack.push(String.valueOf(c));
155                 }
156                 
157                 
158             } else if(c == '}') {
159                 
160                 if(StackUtils.equalsAnyBottomElement(stack, new String[]{"{", "$"})) {
161                     // 式の終わりの場合は、式を取り出し評価する。
162                     String expression = StackUtils.popupAndConcat(stack) + c;
163                     
164                     // エスケープを解除する
165                     expression = Utils.removeEscapeChar(expression, '\\');
166                     
167                     String result = evaluate(expression, vars, recursive, messageResolver);
168                     sb.append(result);
169                     
170                 } else {
171                     sb.append(c);
172                     
173                 }
174                 
175             } else {
176                 
177                 if(stack.isEmpty()) {
178                     sb.append(c);
179                     
180                 } else {
181                     stack.push(String.valueOf(c));
182                 }
183                 
184             }
185             
186         }
187         
188         if(!stack.isEmpty()) {
189             String val = StackUtils.popupAndConcat(stack);
190             val = Utils.removeEscapeChar(val, '\\');
191             sb.append(val);
192         }
193         
194         return sb.toString();
195     }

197     private String evaluate(final String expression, final Map<String, ?> values, final boolean recursive,
198             final MessageResolver messageResolver) {
199         
200         if(expression.startsWith("{")) {
201             // 変数の置換の場合
202             final String varName = expression.substring(1, expression.length()-1);
203             
204             if(values.containsKey(varName)) {
205                 // 該当するキーが存在する場合
206                 final Object value = values.get(varName);
207                 final String eval = (value == null) ? "" : value.toString();
208                 if(!eval.isEmpty() && recursive) {
209                     return parse(eval, values, recursive, messageResolver);
210                 } else {
211                     return eval;
212                 }
213                 
214             } else if(messageResolver != null) {
215                 // メッセージコードをとして解決をする。
216                 final Optional<String> eval = messageResolver.getMessage(varName);
217                 if(!eval.isPresent()) {
218                     // 該当するキーが存在しない場合は、値をそのまま返す。
219                     return String.format("{%s}", varName);
220                 }
221                 
222                 if(recursive) {
223                     return parse(eval.get(), values, recursive, messageResolver);
224                 } else {
225                     return eval.get();
226                 }
227                 
228             } else {
229                 // 該当するキーが存在しない場合は、値をそのまま返す。
230                 return expression.toString();
231             }
232             
233         } else if(expression.startsWith("${")) {
234             // EL式で処理する
235             final String expr = expression.substring(2, expression.length()-1);
236             final String eval = evaluateExpression(expr, values);
237             if(recursive) {
238                 return parse(eval, values, recursive, messageResolver);
239             } else {
240                 return eval;
241             }
242             
243         }
244         
245         throw new MessageParseException(expression, "not support expression.");
246         
247     }

249     /**
250      * EL式を評価する。
251      * @param expression EL式
252      * @param values EL式中の変数。
253      * @return 評価した式。
254      * @throws ExpressionEvaluationException 
255      */
256     protected String evaluateExpression(final String expression, final Map<String, ?> values) throws ExpressionEvaluationException {
257         
258         final Map<String, Object> context = new LinkedHashMap<String, Object>();
259         context.putAll(values);
260         
261         // フォーマッターの追加
262         context.computeIfAbsent("formatter", key -> new Formatter());
263         
264         /*
265          * JEXLで存在しない変数名の場合、nullが帰ってくるため、null判定を行う。
266          */
267         Object eval = expressionLanguage.evaluate(expression, context);
268         String value = eval == null ? "" : eval.toString();
269         
270         if(logger.isTraceEnabled()) {
271             logger.trace("evaluate expression language: expression='{}' ===> value='{}'", expression, value);
272         }
273         
274         return value;
275     }
