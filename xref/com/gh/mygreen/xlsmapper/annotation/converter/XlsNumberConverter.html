3   import java.lang.annotation.Documented;
4   import java.lang.annotation.ElementType;
5   import java.lang.annotation.Retention;
6   import java.lang.annotation.RetentionPolicy;
7   import java.lang.annotation.Target;
8   import java.text.DecimalFormat;
9   import java.util.Currency;
10  
11  
12  
13  /**
14   * 数値型に対するConverter。
15   * 
16   * @version 0.5
17   * @author T.TSUCHIE
18   *
19   */
20  @Target({ElementType.METHOD, ElementType.FIELD})
21  @Retention(RetentionPolicy.RUNTIME)
22  @Documented
23  public @interface XlsNumberConverter {
24      
25      /**
26       * 数値の書式のパターン値を指定します。{@link DecimalFormat}で指定可能な書式を指定します。
27       * @return
28       */
29      String pattern() default "";
30      
31      /**
32       * 通過を指定します。{@link Currency}で処理可能なコード(ISO 4217のコード)で指定します。
33       * <p>formatting {@link Currency} Code(ISO 4217 Code)
34       * @return
35       */
36      String currency() default "";
37      
38      /**
39       * ロケールの指定を行います。指定しない場合、デフォルトのロケールで処理されます。
40       * <p>例. 'ja_JP', 'ja'
41       */
42      String locale() default "";
43      
44      /**
45       * 有効桁数を指定します。
46       * <p>Excelの場合、有効桁数は15桁であるため、デフォルト値は15です。
47       * <p>0以下の値を設定すると、桁数の指定を省略したことになります。
48       * @since 0.5
49       * @return
50       */
51      int precision() default 15;    
52  }
