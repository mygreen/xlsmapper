3   import java.lang.annotation.Documented;
4   import java.lang.annotation.ElementType;
5   import java.lang.annotation.Retention;
6   import java.lang.annotation.RetentionPolicy;
7   import java.lang.annotation.Target;
8   import java.util.TimeZone;
9   
10  
11  /**
12   * 日時に対する変換規則を指定するアノテーションです。
13   *
14   * <p>対応するJavaのクラスタイプと、アノテーションを付与していないときに適用されるの標準の書式は以下の通りです。</p>
15   *
16   * <table class="description">
17   *  <caption>対応する日時のクラスタイプと標準の書式</caption>
18   *  <thead>
19   *  <tr>
20   *   <th>対応するクラスタイプ</th>
21   *   <th>標準の書式</th>
22   *  </tr>
23   *  </thead>
24   *  <tbody>
25   *  <tr>
26   *   <td>java.util.Date</td>
27   *   <td>{@literal yyyy-MM-dd HH:mm:ss}</td>
28   *  </tr>
29   *  <tr>
30   *   <td>java.util.Calendar</td>
31   *   <td>{@literal yyyy-MM-dd HH:mm:ss}</td>
32   *  </tr>
33   *  <tr>
34   *   <td>java.sql.Date</td>
35   *   <td>{@literal yyyy-MM-dd}</td>
36   *  </tr>
37   *  <tr>
38   *   <td>java.sql.Time</td>
39   *   <td>{@literal HH:mm:ss}</td>
40   *  </tr>
41   *  <tr>
42   *   <td>java.sql.Timestamp</td>
43   *   <td>{@literal yyyy-MM-dd HH:mm:ss.SSS}</td>
44   *  </tr>
45   *  <tr>
46   *   <td>java.time.LocalDateTime</td>
47   *   <td>{@literal uuuu-MM-dd HH:mm:ss}</td>
48   *  </tr>
49   *  <tr>
50   *   <td>java.time.LocalDate</td>
51   *   <td>{@literal uuuu-MM-dd}</td>
52   *  </tr>
53   *  <tr>
54   *   <td>java.time.LocalTime</td>
55   *   <td>{@literal HH:mm:ss}</td>
56   *  </tr>
57   *  </tbody>
58   * </table>
59   *
60   * <h3 class="description">書き込み時の注意事項</h3>
61   *
62   * <p>テンプレートファイルのセルの書式を「標準」に設定している場合に書き込むと、
63   *     書式が「標準」設定の全てのセルの書式が書き換わってしまいます。
64   *   <br>そのため、日付や数値などの書式が必要な場合は、テンプレートファイルで予め書式を設定しておくか、
65   *     アノテーションの属性{@link #excelPattern()}で書式を指定しておいてください。
66   * </p>
67   *
68   * @version 2.0
69   * @author T.TSUCHIE
70   *
71   */
72  @Target({ElementType.METHOD, ElementType.FIELD})
73  @Retention(RetentionPolicy.RUNTIME)
74  @Documented
75  public @interface XlsDateTimeConverter {
76  
77      /**
78       * 読み込み時の日時の書式を指定します。
79       * <p>クラスタイプごとに、指定可能な書式は異なります。</p>
80       * <p>{@literal java.util.Date/java.util.Calendar/java.sql.Date/java.sql.Time/java.sql.Timestamp}の場合、
81       *   {@link java.text.SimpleDateFormat}で解釈可能な書式を指定します。
82       * </p>
83       * <p>{@literal java.time.LocalDateTime/java.time.LocalDate/java.time.LocalTime/java.time.ZonedDateTime/java.time.OffsetDateTime/java.time.OffsetTime/java.time.Year/java.time,YearMonth/java.time/MonthDay}の場合、
84       *   {@link java.time.format.DateTimeFormatter}で解釈可能な書式を指定します。
85       * </p>
86       *
87       * <p>読み込み時にセルの種類が日付、時刻ではない場合、
88       *    文字列として値を取得し、その値を属性{@link #javaPattern()}で指定した書式に従いパースし、Javaの日時型に変換します。
89       * </p>
90       * <p>属性{@link #javaPattern()}を指定するとき、他に、属性として、{@link #lenient()}、{@link #locale()}、{@link #timezone()}が指定可能です。 </p>
91       * <p>書式に合わない値をパースした場合、例外{@link com.gh.mygreen.xlsmapper.cellconverter.TypeBindException}が発生します。</p>
92       *
93       * <pre class="highlight"><code class="java">
94       * public class SampleRecord {
95       *
96       *     {@literal @XlsColumn(columnName="有効期限")}
97       *     {@literal @XlsDateTimeConverter(javaPattern="yyyy年MM月dd日 HH時mm分ss秒", locale="ja_JP", lenient=true)}
98       *     private Date expired;
99       *
100      * }
101      * </code></pre>
102      *
103      * @since 1.1
104      * @return
105      */
106     String javaPattern() default "";
107 
108     /**
109      * 日付／時刻の解析を厳密に行うか指定します。
110      * <p>属性{@link #javaPattern()}を設定した場合に有効になります。</p>
111      * @return trueのとき、厳密に解析を行います。
112      */
113     boolean lenient() default false;
114 
115     /**
116      * ロケールの指定を行います。
117      * <p>国コード、バリアントを指定したい場合は、アンダーバー '_' で区切ります。</p>
118      * <p>例. 'ja', 'ja_JP', 'ja_JP_JP'</p>
119      * <p>属性{@link #javaPattern()}を設定した場合に有効になります。</p>
120      * @return 指定しない場合、デフォルトのロケールで処理されます。
121      */
122     String locale() default "";
123 
124     /**
125      * タイムゾーンを指定します。
126      * <p>{@link TimeZone#getTimeZone(String)}で解釈可能な値を指定する必要があります。</p>
127      * <p>{@literal Asia/Tokyo, GMT, GMT+09:00}などの値を指定します。</p>
128      * <p>ただし、オフセットを持たないクラスタイプ{@literal LocalDateTime, LocalDate, LocalTime}の時は、指定しても意味がありません。</p>
129      * @since 2.0
130      * @return 省略した場合、システム標準の値を使用します。
131      */
132     String timezone() default "";
133 
134     /**
135      * 書き込み時にセルの書式を直接設定したい場合に指定します。
136      * <p>値を指定しない場合、テンプレートファイルに設定してある書式が適用されます。</p>
137      *
138      * <ul>
139      *   <li>属性{@link #excelPattern()}で書式を指定します。
140      *       <br>Excelの書式を指定する場合は、
141      *           <a href="http://mygreen.github.io/excel-cellformatter/sphinx/format_basic.html" target="_blank">ユーザ定義</a>
142      *           の形式で指定します。
143      *   </li>
144      * </ul>
145      *
146      * <pre class="highlight"><code class="java">
147      * public class SampleRecord {
148      *
149      *     {@literal @XlsColumn(columnName="有効期限")}
150      *     {@literal @XlsDateTimeConverter(excelPattern="[$-411]yyyy\"年\"mm\"月\"dd\"日\" hh\"時\"mm\"分\"ss\"秒\"")}
151      *     private Date expired;
152      *
153      * }
154      * </code></pre>
155      *
156      * @since 1.1
157      */
158     String excelPattern() default "";
159 
160 }
