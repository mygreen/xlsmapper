3   import java.lang.annotation.Documented;
4   import java.lang.annotation.ElementType;
5   import java.lang.annotation.Retention;
6   import java.lang.annotation.RetentionPolicy;
7   import java.lang.annotation.Target;
8   
9   /**
10   * マッピング用のアノテーションの処理順序を定義します。
11   *
12   * <p>セルに対するマッピング情報を指定したアノテーションが付与された、
13   *    フィールドまたはメソッドの処理順序は、Javaの実行環境に依存するため、
14   *    処理順序を一定にしたいときに付与します。
15   * </p>
16   * <p>そのため、書き込み時に、アノテーション{@link XlsHorizontalRecords}などでマッピングし、
17   *    行の追加、削除を伴う操作が行われるときには、アノテーションで指定したセルの位置がずれ正しく処理できない場合があります。
18   *    <br>また、フィールド{@literal Map<String, CellPosition> positions}などで保持するセルの位置情報もずれます。
19   * </p>
20   *
21   * <p>このような時は、アノテーション{@link XlsOrder}を付与し、処理順序は属性{@link #value()}で指定します。
22   *  <br>属性 value はJavaの仕様により省略が可能です。
23   * </p>
24   * <p>{@link XlsOrder} を付与しないフィールドは、付与しているフィールドよりも後から処理が実行されます。
25   *    <br>属性{@link #value()}が同じ値を設定されているときは、 フィールド名の昇順で優先度を決めて処理されます。
26   * </p>
27   *
28   * <pre class="highlight"><code class="java">
29   * {@literal @XlsSheet(name="Users")}
30   * public class SampleSheet {
31   *
32   *     // セルの位置情報
33   *     private {@literal Map<String, CellPosition>} positions;
34   *
35   *     {@literal @XlsOrder(1)}
36   *     {@literal @XlsHorizontalRecords(tableLabel="ユーザ一覧", terminal=RecordTerminal.Border)}
37   *     {@literal @XlsRecordOption(overOperation=OverOperation.Insert, remainedOperation=RemainedOperation.Delete)}
38   *     private {@literal List<UserRecord>} records;
39   *
40   *     // 属性valueは省略が可能
41   *     {@literal @XlsOrder(2)}
42   *     {@literal @XlsLabelledCell(label="更新日", type=LabelledCellType.Right)}
43   *     private Date updateTime;
44   *
45   * }
46   * </code></pre>
47   *
48   * <div class="picture">
49   *    <img src="doc-files/Hint.png" alt="">
50   *    <p>書き込み時にセルの位置がずれる場合</p>
51   * </div>
52   *
53   * <ul>
54   *  <li>同じ値を設定した場合は、第2処理順としてフィールド名の昇順を使用します。</li>
55   *  <li>このアノテーションの付与、または属性の値が-1以下の場合、付与しているいフィールドよりも処理順序は後になります。</li>
56   * </ul>
57   *
58   * @version 2.0
59   * @author T.TSUCHIE
60   *
61   */
62  @Target({ElementType.METHOD, ElementType.FIELD})
63  @Retention(RetentionPolicy.RUNTIME)
64  @Documented
65  public @interface XlsOrder {
66  
67      /**
68       * オブジェクト内でのフィールドの処理順序を定義します。
69       *
70       * @return 値が大きいほど後から実行されます。
71       */
72      int value();
73  }
