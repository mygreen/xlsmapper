11  /**
12   * ライフサイクルコールバック用のリスナークラスを指定するためのアノテーション。
13   * <p>アノテーション {@link XlsPreLoad}、{@link XlsPostLoad}、{@link XlsPreSave}、{@link XlsPostSave} を使用し、処理を実装します。
14   *
15   * <p>戻り値なしのpublicメソッドに付与する必要があります。</p>
16   * <p>引数は、次の任意の値が指定可能です。定義順は関係ありません。
17   *    <br>引数を取らないことも可能です。
18   * </p>
19   * <ul>
20   *   <li>処理対象のシートオブジェクト {@link org.apache.poi.ss.usermodel.Sheet}</li>
21   *   <li>XlsMapperの設定オブジェクト {@link com.gh.mygreen.xlsmapper.Configuration}</li>
22   *   <li>シートのエラー情報を格納するオブジェクト {@link com.gh.mygreen.xlsmapper.validation.SheetBindingErrors}</li>
23   *   <li>処理対象のBeanオブジェクト</li>
24   * </ul>
25   *
26   * <p>クラスにアノテーション{@link XlsListener#value()} で処理が実装されたクラスを指定します。 </p>
27   * <p>インスタンスは、システム設定{@link Configuration#getBeanFactory()}経由で作成されるため、
28   *   SpringFrameworkのコンテナからインスタンスを取得することもできます。
29   * </p>
30   *
31   *
32   * <pre class="highlight"><code class="java">
33   * // シートクラス
34   * {@literal @XlsSheet(name="Users")}
35   * {@literal @XlsListener(SampleSheetListener.class)}
36   * public class SampleSheet {
37   *
38   *     {@literal @XlsHorizontalRecords(tableLabel="ユーザ一覧")}
39   *     private {@literal List<UserRecord>} records;
40   * }
41   *
42   * // SampleSheetクラスのリスナー
43   * public static class SampleSheetListener {
44   *
45   *     {@literal @XlsPostLoad}
46   *     public void onPostLoad(SampleSheet targetObj) {
47   *         // 読み込み後に実行される処理
48   *     }
49   * }
50   *
51   * // レコードクラス
52   * {@literal @XlsListener(UserRecordListener.class)}
53   * public class UserRecord {
54   *
55   *     {@literal @XlsColumn(columnName="ID")}
56   *     private int id;
57   *
58   *     {@literal @XlsColumn(columnName="名前")}
59   *     private String name;
60   *
61   * }
62   *
63   * // UserRecordクラスのリスナー
64   * public static class UserRecordListener {
65   *
66   *     {@literal @XlsPostLoad}
67   *     public void onPostLoad(UserRecord targetObj, Sheet sheet, Configuration config, SheetBindingErrors errors) {
68   *         // 読み込み後に実行される処理
69   *         // 入力値チェックなどを行う
70   *     }
71   * }
72   * </code></pre>
73   *
74   * @version 2.0
75   * @since 1.3
76   * @author T.TSUCHIE
77   *
78   */
79  @Target({ElementType.TYPE})
80  @Retention(RetentionPolicy.RUNTIME)
81  @Documented
82  public @interface XlsListener {
83  
84      /**
85       * リスナークラスの指定。
86       * @return 任意の実装クラスを指定します。
87       */
88      Class<?>[] value();
89  
90  }
