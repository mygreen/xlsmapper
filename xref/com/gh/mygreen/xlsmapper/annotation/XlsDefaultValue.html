12  /**
13   * 初期値を指定するためのアノテーションです。
14   * 読み込み時、または書き込み時に値がnullの時に、代わりにとなる値を指定します。
15   *
16   * <ul>
17   *   <li>属性{@link #value()}で、読み込み時/書き込み時のデフォルト値を指定します。</li>
18   *   <li>属性{@link #cases()}で、 </li>
19   * </ul>
20   *
21   * <ul>
22   *   <li>日付などの書式がある場合、専用のアノテーションで指定した書式{@link XlsDateTimeConverter#javaPattern()}を元に、
23   *       文字列をそのオブジェクトに変換し処理します。
24   *   </li>
25   *   <li>指定したデフォルト値がマッピング先の型として不正な場合は、
26   *       通常の型変換エラーと同様に、例外 {@link TypeBindException} がスローされます。
27   *   </li>
28   *   <li>読み込みの際には、プリミティブ型に対して{@link XlsDefaultValue}を設定していても、その型の初期値が設定さるため注意してください。
29   *       例えば、int型は0、doubleは0.0、booleanはfalse。char型は、'{@literal \}u0000'。
30   *       <br>プリミティブ型のラッパークラスや参照型の場合は、nullが設定されます。
31   *   </li>
32   *   <li>char型にマッピングする場合、デフォルト値が2文字以上でも、先頭の位置文字がマッピングされます。</li>
33   * </ul>
34   *
35   * <pre class="highlight"><code class="java">
36   * public class SampleRecord {
37   *
38   *     {@literal @XlsColumn(columnName="ID")}
39   *     {@literal @XlsDefaultValue("-1")}
40   *     private int id;
41   *
42   *     {@literal @XlsColumn(columnName="更新日時")}
43   *     {@literal @XlsDefaultValue("2010/01/01"} // 属性javaPatternで指定した書式に沿った値を指定します。
44   *     {@literal @XlsDateTimeConverter(javaPattern="yyyy/MM/dd")}
45   *     private Date updateTime;
46   *
47   * }
48   * </code></pre>
49   *
50   * @since 2.0
51   * @author T.TSUCHIE
52   *
53   */
54  @Target({ElementType.METHOD, ElementType.FIELD})
55  @Retention(RetentionPolicy.RUNTIME)
56  @Documented
57  public @interface XlsDefaultValue {
58  
59      /**
60       * 初期値を指定します。
61       * @return 設定した初期値。
62       */
63      String value();
64  
65      /**
66       * 適用するケースを指定します。
67       * @return 何も指定しない場合は全てのケースに適用されます。
68       */
69      ProcessCase[] cases() default {};
70  
71  }
