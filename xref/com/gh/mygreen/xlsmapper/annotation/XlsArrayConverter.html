14  /**
15   * {@link java.util.Collection}({@link java.util.List}, {@link java.util.Set})または配列に対する変換規則を指定するアノテーション。
16   *
17   * <h3 class="description">基本的な使い方</h3>
18   * <p>{@link java.util.Collection}型のインタフェースを指定している場合、読み込み時のインスタンスは次のクラスが指定されます。</p>
19   * <ul>
20   *   <li>{@link java.util.List}の場合、{@link java.util.ArrayList}がインスタンスのクラスとなります。
21   *   <li>{@link java.util.Set} の場合、{@link java.util.LinkedHashSet}がインスタンスのクラスとなります。
22   * </ul>
23   *
24   * <p>配列、またはCollection型の要素で指定可能なクラスタイプは、次の通りです。</p>
25   * <ul>
26   *   <li>{@link String}型</li>
27   *   <li>プリミティブ型「boolean/char/byte/short/int/long/float/double」と、そのラッパークラス。</li>
28   *   <li>{@link java.math.BigDecimal}/{@link java.math.BigInteger}</li>
29   * </ul>
30   *
31   * <p>文字列のセルに対して、任意の区切り文字を指定し、配列やListに対してマッピングします。</p>
32   * <ul>
33   *   <li>属性{@link #separator()}で区切り文字を指定します。
34   *     <br>区切り文字の初期値は、半角カンマ(,)です。
35   *   </li>
36   *   <li>{@link XlsTrim}を設定すると、区切った項目にもトリム処理が適用されます。</li>
37   * </ul>
38   *
39   * <pre class="highlight"><code class="java">
40   * public class SampleRecord {
41   *
42   *     {@literal @XlsColumn(columnName="リスト")}
43   *     {@literal @XlsArrayConverter(separator="\n")}  // 区切り文字を改行コード({@literal \\}n)として指定します。
44   *     private {@literal List<String>} list;
45   *
46   *     {@literal @XlsColumn(columnName="配列")}
47   *     {@literal @XlsTrim}    // 区切った配列の要素にもトリムが適用されます。
48   *     {@literal @XlsArrayConverter(separator=",")}
49   *     private int[] array;
50   *
51   * }
52   * </code></pre>
53   *
54   *
55   * @version 2.0
56   * @author T.TSUCHIE
57   *
58   */
59  @Target({ElementType.METHOD, ElementType.FIELD})
60  @Retention(RetentionPolicy.RUNTIME)
61  @Documented
62  public @interface XlsArrayConverter {
63  
64      /**
65       * 区切り文字を指定します。
66       * @return
67       */
68      String separator() default ",";
69  
70      /**
71       * 区切った項目の値が空文字または{@literal null}の場合、無視するか指定します。
72       *
73       * <p>例えば、区切り文字が「,」のとき、セルの値が「{@code a,,b}」の場合、
74       *    trueを設定すると、空の項目は覗き、{@code ["a", "b"]}として読み込みます。
75       *   <br>書き込み時も同様に、値が空またはnullの項目を無視します。
76       * </p>
77       *
78       * <pre class="highlight"><code class="java">
79       * public class SampleRecord {
80       *
81       *     {@literal @XlsColumn(columnName="集合")}
82       *     {@literal @XlsArrayConverter(ignoreEmptyElement=true)}
83       *     private {@literal Set<String>} set;
84       *
85       * }
86       * </code></pre>
87       *
88       * @return trueのとき、区切った項目の値が空文字または{@literal null}の場合、無視します。
89       */
90      boolean ignoreEmptyElement() default false;
91  
92      /**
93       * 配列やリストの要素のクラス型を指定します。
94       * <p>プリミティブ型とそのラッパークラスのみ指定できます。</p>
95       * <p>省略した場合、Genericsから自動的に判断してます。</p>
96       */
97      Class<?> elementClass() default Object.class;
98  
99      /**
100      * 配列やリストの要素に対する変換用のクラスを指定します。
101      * <p>変換するクラスは、インタフェース{@link ElementConverter}を実装している必要があります。
102      *   <br>標準では、{@link DefaultElementConverter} が使用され、基本的な型のみサポートしています。
103      * </p>
104      * <p>インスタンスは、システム設定{@link Configuration#getBeanFactory()}経由で作成されるため、
105      *   SpringFrameworkのコンテナからインスタンスを取得することもできます。
106      * </p>
107      *
108      * <pre class="highlight"><code class="java">
109      * // 変換用クラス
110      * public class CustomElementConverter implements {@literal ElementConverter<User>} {
111      *
112      *     {@literal @Override}
113      *     public User convertToObject(final String str, final {@literal Class<User>} targetClass) throws ConversionException {
114      *         //TODO: 文字列 {@literal =>} オブジェクトに変換する処理
115      *     }
116      *
117      *     {@literal @Override}
118      *     public String convertToString(final User value) {
119      *         //TODO: オブジェクト {@literal =>} 文字列に変換する処理
120      *     }
121      *
122      * }
123      *
124      * // レコード用クラス
125      * public class SampleRecord {
126      *
127      *     // 任意のクラス型の要素の値を変換するConverterを指定します。
128      *     {@literal @XlsColumn(columnName="リスト")}
129      *     {@literal @XlsArrayConverter(elementConverter=CustomElementConverter.class)}
130      *     private {@literal List<User>} list;
131      *
132      * }
133      * </code></pre>
134      *
135      *
136      * @since 1.1
137      * @return {@link ElementConverter}の実装クラス。
138      */
139     @SuppressWarnings("rawtypes")
140     Class<? extends ElementConverter> elementConverter() default DefaultElementConverter.class;
141 
142 }
