1   package com.gh.mygreen.xlsmapper.annotation;
2   
3   import java.lang.annotation.Documented;
4   import java.lang.annotation.ElementType;
5   import java.lang.annotation.Retention;
6   import java.lang.annotation.RetentionPolicy;
7   import java.lang.annotation.Target;
8   
9   
10  /**
11   * 列挙型に対する変換規則を指定するアノテーションです。
12   *
13   * <h3 class="description">基本的な使い方</h3>
14   *
15   * <p>セルの値と列挙型の要素の値をマッピングさせます。
16   *    <br>要素の値とは、 {@link Enum#name()} の値です。
17   * </p>
18   *
19   * <ul>
20   *   <li>属性{@link #ignoreCase()}の値をtrueにすると、読み込み時に大文字/小文字の区別なく変換します。
21   * </ul>
22   *
23   * <pre class="highlight"><code class="java">
24   * public class SampleRecord {
25   *
26   *     {@literal @XlsColumn(columnName="権限")}
27   *     {@literal @XlsEnumConverter(ignoreCase=true)}
28   *     private RoleType role;
29   *
30   * }
31   *
32   * // 列挙型の定義
33   * public enum RoleType {
34   *     Normal, Admin;
35   * }
36   * </code></pre>
37   *
38   * @version 2.0
39   * @author T.TSUCHIE
40   *
41   */
42  @Target({ElementType.METHOD, ElementType.FIELD})
43  @Retention(RetentionPolicy.RUNTIME)
44  @Documented
45  public @interface XlsEnumConverter {
46  
47      /**
48       * 読み込み時に大文字／小文字を無視して比較するか指定します。
49       * @return trueの場合、大文字・小文字の区別は行いません。
50       */
51      boolean ignoreCase() default false;
52  
53      /**
54       * 列挙型のname()メソッド以外から取得し、別名でマッピングするときに指定します。
55       * <p>例). RoleType.localeName()のlocaleName()メソッドを指定するときには、'localeName'と指定します。</p>
56       *
57       * <pre class="highlight"><code class="java">
58       * public class SampleRecord {
59       *
60       *     {@literal @XlsColumn(columnName="権限")}
61       *     {@literal @XlsEnumConverter(aliasMethod="localeName")}
62       *     private RoleType role;
63       *
64       * }
65       *
66       * // 列挙型の定義
67       * public enum RoleType {
68       *     Normal("一般権限"), Admin("管理者権限");
69       *
70       *     // 別名の設定
71       *     private String localeName;
72       *
73       *     private RoleType(String localeName) {
74       *         this.localeName = localeName;
75       *     }
76       *
77       *     // 別名の取得用メソッド
78       *     public String localeName() {
79       *         return this.localeName;
80       *     }
81       * }
82       * </code></pre>
83       *
84       * @return 指定しない場合は、name()メソッドが使用されます。
85       */
86      String aliasMethod() default "";
87  }
