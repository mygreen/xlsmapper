1   package com.gh.mygreen.xlsmapper.annotation;
2   
3   import java.lang.annotation.Documented;
4   import java.lang.annotation.ElementType;
5   import java.lang.annotation.Retention;
6   import java.lang.annotation.RetentionPolicy;
7   import java.lang.annotation.Target;
8   
9   /**
10   * 数値型に対する変換規則を指定するアノテーションです。
11   * <p>対応するJavaのクラスタイプは次の通り。</p>
12   * <ul>
13   *   <li>byte/short/int/long/float/doubleのプリミティブ型とそのラッパークラス。</li>
14   *   <li>{@link java.math.BigDecimal}/{@link java.math.BigInteger}</li>
15   * </ul>
16   *
17   *
18   * <h3 class="description">書き込み時の注意事項</h3>
19   *
20   * <p>テンプレートファイルのセルの書式を「標準」に設定している場合に書き込むと、
21   *     書式が「標準」設定の全てのセルの書式が書き換わってしまいます。
22   *   <br>そのため、日付や数値などの書式が必要な場合は、テンプレートファイルで予め書式を設定しておくか、
23   *     アノテーションの属性{@link #excelPattern()}で書式を指定しておいてください。
24   * </p>
25   *
26   * @version 1.1
27   * @author T.TSUCHIE
28   *
29   */
30  @Target({ElementType.METHOD, ElementType.FIELD})
31  @Retention(RetentionPolicy.RUNTIME)
32  @Documented
33  public @interface XlsNumberConverter {
34  
35      /**
36       * 数値の書式のパターン値を指定します。
37       * <p>{@link java.text.DecimalFormat}で指定可能な書式を指定します。</p>
38       *  <p>読み込み時にセルの種類が数値（通貨、会計、パーセンテージ、分数、指数）ではない場合、
39       *    文字列として値を取得し、その値を属性{@link #javaPattern()}で指定した書式に従いパースし、Javaの数値型に変換します。
40       * </p>
41       *
42       * <p>属性{@link #javaPattern()}を指定するとき、他に、属性として、{@link #locale()}、{@link #currency()}}が指定可能です。 </p>
43       * <p>書式に合わない値をパースした場合、例外{@link com.gh.mygreen.xlsmapper.cellconverter.TypeBindException}が発生します。</p>
44       *
45       * <pre class="highlight"><code class="java">
46       * public class SampleRecord {
47       *
48       *     {@literal @XlsColumn(columnName="給料")}
49       *     {@literal @XlsNumberConverter(pattern="\u00A4\u00A4 #,##0.0000", locale="ja_JP", currency="USD")}
50       *     private double salary;
51       *
52       * }
53       * </code></pre>
54       *
55       * @return
56       */
57      String javaPattern() default "";
58  
59      /**
60       * 通貨を指定します。
61       * <p>{@link java.util.Currency}で処理可能なコード(ISO 4217のコード)で指定します。</p>
62       * <p>属性{@link #javaPattern()}を設定した場合に有効になります。
63       * @return
64       */
65      String currency() default "";
66  
67      /**
68       * ロケールの指定を行います。
69       * <p>指定しない場合、デフォルトのロケールで処理されます。</p>
70       * <p>国コード、バリアントを指定したい場合は、アンダーバー '_' で区切ります。</p>
71       * <p>例. 'ja', 'ja_JP', 'ja_JP_JP'</p>
72       * <p>属性{@link #javaPattern()}を設定した場合に有効になります。</p>
73       */
74      String locale() default "";
75  
76      /**
77       * 有効桁数を指定します。
78       * <p>Excel内部とJavaの数値は、表現可能な有効桁数が異なります。
79       *    <br>そのため、特に読み込み時などExcelの仕様に合わせてJavaのクラスに指定することが可能です。
80       * </p>
81       * <p>Excelの場合、有効桁数は15桁であるため、デフォルト値は15です。</p>
82       * <p>0以下の値を設定すると、桁数の指定を省略したことになり、無制限になります。</p>
83       * <p>Excelでは有効桁数が15桁であるため、Javaのlong型など15桁を超える表現が可能な数値を書き込んだ場合、
84       *    数値が丸められるため注意してください。
85       * </p>
86       * <ul>
87       *   <li>例えば、long型の19桁の数値 {@code 1234567890123456789} を書き込んだ場合、
88       *       16桁以降の値が丸められ {@code 1234567890123450000} として書き込まれます。
89       *   <li>Excelの仕様については、
90       *       <a href="https://support.office.com/ja-jp/article/Excel-%E3%81%AE%E4%BB%95%E6%A7%98%E3%81%A8%E5%88%B6%E9%99%90-1672b34d-7043-467e-8e27-269d656771c3?ui=ja-JP&rs=ja-JP&ad=JP" target="_blank">Excel の仕様と制限</a>
91       *       を参照してください。
92       * </ul>
93       *
94       * @since 0.5
95       * @return
96       */
97      int precision() default 15;
98  
99      /**
100      * Excelの書式のパターン。
101      * <p>書き込み時に、セルの書式を直接設定したい場合に指定します。</p>
102      * <p>値を指定しない場合、テンプレートファイルに設定してある書式が適用されます。</p>
103      *
104      * <ul>
105      *   <li>属性{@link #excelPattern()}で書式を指定します。
106      *       <br>Excelの書式を指定する場合は、
107      *           <a href="http://mygreen.github.io/excel-cellformatter/sphinx/format_basic.html" target="_blank">ユーザ定義</a>
108      *           の形式で指定します。
109      *   </li>
110      * </ul>
111      *
112      * <pre class="highlight"><code class="java">
113      * public class SampleRecord {
114      *
115      *     {@literal @XlsColumn(columnName="給与")}
116      *     {@literal @XlsNumberConverter(javaPattern="[$-411]\"￥\"#,##0.0000")}
117      *     private double salary;
118      *
119      * }
120      * </code></pre>
121      *
122      * @since 1.1
123      */
124     String excelPattern() default "";
125 
126 }
