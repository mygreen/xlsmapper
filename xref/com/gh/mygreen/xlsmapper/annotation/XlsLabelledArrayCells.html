13  /**
14   * セルの見出し用のラベルセルを指定し、左右もしくは下側に連続し隣接するセルをCollection(List, Set)または配列にマッピングします。
15   * <p>アノテーション {@link XlsArrayCells} と {@link XlsLabelledCell} を融合したアノテーションとなります。</p>
16   *
17   * <h3 class="description">基本的な使い方</h3>
18   * <p>属性{@link #label()} で、見出しとなるセルの値を指定します。</p>
19   * <p>属性{@link #type()}で見出しとなるセルから見て、設定されている位置を指定します。</p>
20   * <p>属性{@link #direction()}で、連続する隣接するセルの方向を指定します。</p>
21   * <p>属性{@link #size()}で、マッピングするセルの個数を指定します。</p>
22   * <p>セルが見つからない場合はエラーとなりますが、属性{@link #optional()}を'true'とすることで無視して処理を続行します。</p>
23   * <p>配列または、{@link java.util.Collection}({@link java.util.List}/{@link java.util.Set})にマッピングします。</p>
24   * <p>{@link java.util.Collection}型のインタフェースを指定している場合、読み込み時のインスタンスは次のクラスが指定されます。</p>
25   * <ul>
26   *   <li>{@link java.util.List}の場合、{@link java.util.ArrayList}がインスタンスのクラスとなります。
27   *   <li>{@link java.util.Set} の場合、{@link java.util.LinkedHashSet}がインスタンスのクラスとなります。
28   * </ul>
29   *
30   * <pre class="highlight"><code class="java">
31   * {@literal @XlsSheet(name="Users")}
32   * public class SampleSheet {
33   *
34   *     // ラベルの右側 + 横方向の隣接するセル
35   *     // 属性directionを省略した場合は、ArrayDirection.Horizonを指定したと同じ意味。
36   *     {@literal @XlsLabelledArrayCells(label="ラベル1", type=LabelledCellType.Right, size=6)}
37   *     private {@literal List<String>} nameKanas1;
38   *
39   *     // ラベルの下側 + 横方向の隣接するセル
40   *     // 属性optional=trueと設定すると、ラベルセルが見つからなくても処理を続行する
41   *     {@literal @XlsLabelledArrayCells(label="ラベル2", type=LabelledCellType.Bottom,}
42   *              direction=ArrayDirection.Horizon, size=6, optional=true)
43   *     private String[] nameKanas2;
44   *
45   * }
46   * </code></pre>
47   *
48   * <div class="picture">
49   *    <img src="doc-files/LabelledArrayCells.png" alt="">
50   *    <p>基本的な使い方</p>
51   * </div>
52   *
53   *
54   * <h3 class="description">書き込み時に配列・リストのサイズが不足、または余分である場合</h3>
55   * アノテーション {@link XlsArrayOption} を指定することで、書き込み時のセルの制御を指定することができます。
56   *
57   * <p>属性 {@link XlsArrayOption#overOpration()} で、書き込み時にJavaオブジェクトの配列・リストのサイズに対して、属性 {@link #size()} の値が小さく、足りない場合の操作を指定します。
58   * <p>属性 {@link XlsArrayOption#remainedOperation()} で、書き込み時にJavaオブジェクトの配列・リストのサイズに対して、属性 {@link #size()} の値が大きく、余っている場合の操作を指定します。
59   *
60   * <pre class="highlight"><code class="java">
61   * {@literal @XlsSheet(name="Users")}
62   * public class SampleSheet {
63   *
64   *     {@literal @XlsLabelledArrayCells(label="ふりがな", type=LabelledCellType.Right, size=6)}
65   *     {@literal @XlsArrayOption(overOperation=OverOperation.Error, remainedOperation=RemainedOperation.Clear)}
66   *     private {@literal List<String>} nameKana;
67   *
68   * }
69   * </code></pre>
70   *
71   * <div class="picture">
72   *    <img src="doc-files/LabelledArrayCells_ArrayOption.png" alt="">
73   *    <p>書き込み時の制御を行う場合</p>
74   * </div>
75   *
76   * <h3 class="description">位置情報／見出し情報を取得する際の注意事項</h3>
77   * <p>マッピング対象のセルのアドレスを取得する際に、フィールド{@literal Map<String, CellPosition> positions}を定義しておけば、
78   *    自動的にアドレスがマッピングされます。
79   *    <br>通常は、キーにはプロパティ名が記述（フィールドの場合はフィールド名）が入ります。
80   *    <br>アノテーション{@link XlsLabelledArrayCells}でマッピングしたセルのキーは、{@literal <プロパティ名>[<インデックス>]}の形式になります。
81   * </p>
82   * <p>同様に、マッピング対象の見出しを取得する、フィールド{@literal Map<String, String> labels}へのアクセスも、
83   *    キーは{@literal <プロパティ名>[<インデックス>]}の形式になります。
84   *    <br>ただし、見出し情報の場合は、全ての要素が同じ値になるため、従来通りの {@literal <プロパティ名>} でも取得できます。
85   * </p>
86   *
87   * <pre class="highlight"><code class="java">
88   * public class SampleRecord {
89   *
90   *     // 位置情報
91   *     private {@literal Map<String, CellPosition>} positions;
92   *
93   *     // 見出し情報
94   *     private {@literal Map<String, String>} labels;
95   *
96   *     {@literal @XlsLabelledArrayCells(label="ふりがな", type=LabelledCellType.Right, size=6)}
97   *     private {@literal List<String>} nameKana;
98   *
99   * }
100  *
101  *
102  * // 位置情報・見出し情報へのアクセス
103  * SampleRecord record = ...;
104  *
105  * CellPosition position = record.positions.get("nameKana[2]");
106  *
107  * String label = recrod.labeles.get("nameKana[2]");
108  *
109  * // 見出し情報の場合、従来通りのインデックスなしでも取得できる
110  * String label = recrod.labeles.get("nameKana");
111  * </code></pre>
112  *
113  *
114  * <div class="picture">
115  *    <img src="doc-files/LabelledArrayCells_positions.png" alt="">
116  *    <p>位置情報・見出し情報の取得</p>
117  * </div>
118  *
119  * <h3 class="description">ラベルセルを正規表現、正規化して指定する場合</h3>
120  *
121  * <p>シートの構造は同じだが、ラベルのセルが微妙に異なる場合、ラベルセルを正規表現による指定が可能です。
122  *   <br>また、空白や改行を除去してラベルセルを比較するように設定することも可能です。</p>
123  *
124  * <p>正規表現で指定する場合、アノテーションの属性の値を {@code /正規表現/} のように、スラッシュで囲みます。</p>
125  * <ul>
126  *   <li>スラッシュで囲まない場合、通常の文字列として処理されます。</li>
127  *   <li>正規表現の指定機能を有効にするには、システム設定のプロパティ {@link Configuration#setRegexLabelText(boolean)} の値を trueに設定します。</li>
128  * </ul>
129  *
130  * <p>ラベセルの値に改行が空白が入っている場合、それらを除去し正規化してアノテーションの属性値と比較することが可能です。</p>
131  * <ul>
132  *   <li>正規化とは、空白、改行、タブを除去することを指します。</li>
133  *   <li>ラベルを正規化する機能を有効にするには、、システム設定のプロパティ {@link Configuration#setNormalizeLabelText(boolean)} の値を trueに設定します。</li>
134  * </ul>
135  *
136  * <p>これらの指定が可能な属性は、{@link #label()}、{@link #headerLabel()}です。</p>
137  *
138  * <pre class="highlight"><code class="java">
139  * // システム設定
140  * XlsMapper xlsMapper = new XlsMapper();
141  * xlsMapper.getConfiguration()
142  *         .setRegexLabelText(true)        // ラベルを正規表現で指定可能にする機能を有効にする。
143  *         .setNormalizeLabelText(true);   // ラベルを正規化して比較する機能を有効にする。
144  *
145  * // シート用クラス
146  * {@literal @XlsSheet(name="Users")}
147  * public class SampleSheet {
148  *
149  *     // 正規表現による指定
150  *     {@literal @XlsLabelledArrayCells(label="/名前.+/", type=LabelledCellType.Right, size=10)}
151  *     private {@literal List<String>} names;
152  *
153  *     {@literal @XlsLabelledArrayCells(label="コメント（オプション）", type=LabelledCellType.Right, size=5)}
154  *     private String[] comments;
155  *
156  * }
157  * </code></pre>
158  *
159  *
160  * @since 2.0
161  * @author T.TSUCHIE
162  *
163  */
164 @Target({ElementType.METHOD, ElementType.FIELD})
165 @Retention(RetentionPolicy.RUNTIME)
166 @Documented
167 @XlsFieldProcessor(value={})
168 public @interface XlsLabelledArrayCells {
169 
170     /**
171      * 連続するセルの個数を指定します。
172      * @return 1以上の値を指定します。
173      */
174     int size();
175 
176     /**
177      * 値のセルが結合している場合、それを考慮するかどうか指定します。
178      * この値により、属性{@link #size()}の指定方法が変わります。
179      * <p>セル結合されている場合は、結合後の個数を指定します。</p>
180      * <ul>
181      *   <li>trueの場合は、結合されているセルを1つのセルとしてマッピングします。</li>
182      *   <li>falseの場合は、結合されていても解除した状態と同じマッピング結果となります。
183      *     <br>ただし、書き込む際には、結合が解除されます。
184      *   </li>
185      * </ul>
186      *
187      * <pre class="highlight"><code class="java">
188      * {@literal @XlsSheet(name="Users")}
189      * public class SampleSheet {
190      *
191      *     // elementMerged=trueは初期値なので、省略可
192      *     {@literal @XlsLabelledArrayCells(label="ラベル1", type=LabelledCellType.Right,size=3, elementMerged=true)}
193      *     private {@literal List<String>} words;
194      *
195      * }
196      * </code></pre>
197      *
198      * <div class="picture">
199      *    <img src="doc-files/LabelledArrayCells_elementMerged.png" alt="">
200      *    <p>結合したセルをマッピングする場合</p>
201      * </div>
202      *
203      * @return trueの場合、値のセルが結合されていることを考慮します。
204      */
205     boolean elementMerged() default true;
206 
207     /**
208      * 連続し隣接するセルの方向を指定します。
209      * <p>セルの位置{@link #type()}を{@link LabelledCellType#Left}のとき、
210      *   セルの方向{@link #direction()}を{@link ArrayDirection#Horizon} は、設定できません。
211      * </p>
212      *
213      * <pre class="highlight"><code class="java">
214      * {@literal @XlsSheet(name="Users")}
215      * public class SampleSheet {
216      *
217      *     // ラベルの右側 + 横方向の隣接するセル
218      *     // 属性direction=ArrayDirection.Verticalを指定すると、縦方向にマッピングします。
219      *     {@literal @XlsLabelledArrayCells(label="ラベル3", type=LabelledCellType.Right,}
220      *             direction=ArrayDirection.Vertical, size=4)
221      *     private {@literal List<String>} nameKanas3;
222      *
223      *     // ラベルの下側 + 横方向の隣接するセル
224      *     {@literal @XlsLabelledArrayCells(label="ラベル4", type=LabelledCellType.Bottom,}
225      *              direction=ArrayDirection.Vertical, size=4)
226      *     private String[] nameKanas4;
227      *
228      * }
229      * </code></pre>
230      *
231      * <div class="picture">
232      *    <img src="doc-files/LabelledArrayCells_direction.png" alt="">
233      *    <p>属性directionの概要</p>
234      * </div>
235      *
236      *
237      * @return セルの方向を指定します。
238      */
239     ArrayDirection direction() default ArrayDirection.Horizon;
240 
241     /**
242      * 配列またはリスト要素の値のクラスを指定します。
243      * <p>省略した場合、定義されたたGenericsの情報から取得します。
244      */
245     Class<?> elementClass() default Object.class;
246 
247     /**
248      * セルが見つからなかった場合はエラーとなりますが、optional属性にtrueを指定しておくと、無視して処理を続行します。
249      * <p>falseを指定し、セルが見つからない場合は、例外{@link CellNotFoundException}がスローされます。</p>
250      */
251     boolean optional() default false;
252 
253     /**
254      * 見出しセルから見て値が設定されているセルの位置を指定します。
255      */
256     LabelledCellType type();
257 
258     /**
259      * 見出しとなるセルの値を指定します。
260      * <p>システム設定により、正規表現による指定や正規化（改行、空白、タブの削除）による比較の対象となります。</p>
261      */
262     String label() default "";
263 
264     /**
265      * 見出しとなるセルの行番号を指定します。
266      * <p>{@link #labelColumn()}属性とセットで指定します。</p>
267      * <p>この属性は、{@link XlsIterateTables}の中で指定したときに、処理内部で使用されるため、通常は設定しません。</p>
268      *
269      * @return 値は0から始まります。-1以下の負の値は無視されます。
270      */
271     int labelRow() default -1;
272 
273     /**
274      * 見出しとなるセルの列番号を指定します。
275      * <p>{@link #labelRow()}属性とセットで指定します。</p>
276      * <p>この属性は、{@link XlsIterateTables}の中で指定したときに、処理内部で使用されるため、通常は設定しません。</p>
277      *
278      * @return 値は0から始まります。-1以下の負の値は無視されます。
279      */
280     int labelColumn() default -1;
281 
282     /**
283      * 同じラベルのセルが複数ある場合は、区別するため見出しを属性{@link #headerLabel()} で指定します。
284      * <p>属性{@link #headerLabel()}で指定したたセルから、属性{@link #label()}で指定したセルを下方向に検索し、
285      *    最初に見つかったセルをラベルセルとして使用します。
286      * </p>
287      * <p>システム設定により、正規表現による指定や正規化（改行、空白、タブの削除）による比較の対象となります。</p>
288      *
289      * <pre class="highlight"><code class="java">
290      * {@literal @XlsSheet(name="Users")}
291      * public class SampleSheet {
292      *
293      *     {@literal @XlsLabelledCell(label="ふりがな", type=LabelledCellType.Right, size=10, headerLabel="氏名")}
294      *     private {@literal List<String>} nameRuby;
295      *
296      *     {@literal @XlsLabelledCell(label="ふりがな", type=LabelledCellType.Right, size=10, headerLabel="住所")}
297      *     private {@literal List<String>} addressRuby;
298      *
299      * }
300      * </code></pre>
301      *
302      * <div class="picture">
303      *    <img src="doc-files/LabelledArrayCells_headerLabel.png" alt="">
304      *    <p>属性headerLabelの概要</p>
305      * </div>
306      *
307      * @return 見出しとなるセルを指定します。指定しない場合は空文字を指定します。
308      */
309     String headerLabel() default "";
310 
311     /**
312      * 属性{@link #type()}の位置に向かって指定したセル数分を検索し、最初に発見した空白以外のセルを開始位置としてマッピングします。
313      * <p>属性{@link #range()}と{@link #skip()}を同時に指定した場合、まず、skip分セルを読み飛ばし、そこからrangeの範囲で空白以外のセルを検索します。</p>
314      * <p>この属性 は、 <strong>読み込み時のみ有効</strong> です。書き込み時に指定しても無視されます。
315      *   <br>ただし、データセルが偶然空白のときは、マッピング対象のセルがずれるため、この属性を使用する場合は注意が必要です。
316      * </p>
317      *
318      * <pre class="highlight"><code class="java">
319      * {@literal @XlsSheet(name="Users")}
320      * public class SampleSheet {
321      *
322      *     {@literal @XlsLabelledArrayCells(label="ラベル1", type=LabelledCellType.Right, range=4, size=4)}
323      *     private {@literal List<String>} words1;
324      *
325      *     {@literal @XlsLabelledArrayCells(label="ラベル2", type=LabelledCellType.Bottom, range=5, size=3)}
326      *     private {@literal List<String>} words2;
327      *
328      * }
329      * </code></pre>
330      *
331      * <div class="picture">
332      *    <img src="doc-files/LabelledArrayCells_range.png" alt="">
333      *    <p>属性rangeの概要</p>
334      * </div>
335      *
336      * @return 値は1から始まります。指定しない場合は1を指定します。
337      */
338     int range() default 1;
339 
340     /**
341      * ラベルセルから指定したセル数分離れたセルを開始位置としてマッピングする際に指定します。
342      * <p>属性{@link #type()}の方向に向かってラベルセルから指定したセル数分離れたセルを開始としてマッピングすることができます。</p>
343      * <p>属性{@link #range()}と{@link #skip()}を同時に指定した場合、まず、skip分セルを読み飛ばし、そこからrangeの範囲で空白以外のセルを検索します。</p>
344      *
345      * <pre class="highlight"><code class="java">
346      * {@literal @XlsSheet(name="Users")}
347      * public class SampleSheet {
348      *
349      *     {@literal @XlsLabelledCell(label="ラベル1", type=LabelledCellType.Right, size=3, skip=2)}
350      *     private {@literal List<String>} wrods2;
351      *
352      *     {@literal @XlsLabelledCell(label="ラベル2", type=LabelledCellType.Bottom, size=3, skip=3)}
353      *     private {@literal List<String>} wrods2;
354      *
355      * }
356      * </code></pre>
357      *
358      * <div class="picture">
359      *    <img src="doc-files/LabelledArrayCells_skip.png" alt="">
360      *    <p>属性skipの概要</p>
361      * </div>
362      *
363      *
364      * @return 値は0から始まります。0以下の値は無視されます。
365      */
366     int skip() default 0;
367 
368     /**
369      * ラベルセルが結合している場合を考慮するかどうか指定します。
370      * <ul>
371      *  <li>値がtrueのとき、結合されているセルを1つのラベルセルとしてマッピングします。</li>
372      *  <li>値がfalseの場合は、結合されていても解除した状態と同じマッピング結果となります。</li>
373      * </ul>
374      * <p>初期値はtrueであるため、結合されているかどうかは特に意識はする必要はありません。</p>
375      * <p>{@link #labelMerged()}の値が falseのとき、ラベルセルが結合されていると、
376      *    値が設定されているデータセルまでの距離が変わるため、属性{@link #skip()}を併用します。
377      * </p>
378      *
379      * <pre class="highlight"><code class="java">
380      * {@literal @XlsSheet(name="Users")}
381      * public class SampleSheet {
382      *
383      *     // labelMerged=trueは初期値なので、省略可
384      *     {@literal @@XlsLabelledArrayCells(label="ラベル1", type=LabelledCellType.Right, size=6)}
385      *     private {@literal List<String>} name1;
386      *
387      *     // labelMerged=falseで、ラベルが結合しているときは、skip属性を併用します。
388      *     {@literal  @XlsLabelledCell(label="ラベル1", type=LabelledCellType.Right, size=6, labelMerged=false, skip=2)}
389      *     private {@literal List<String>} name2;
390      *
391      * }
392      * </code></pre>
393      *
394      * <div class="picture">
395      *    <img src="doc-files/LabelledArrayCells_labelMerged.png" alt="">
396      *    <p>属性labelMergedの概要</p>
397      * </div>
398      *
399      * @return trueの場合、ラベルがセルが結合されていることを考慮します。
400      */
401     boolean labelMerged() default true;
402 
403     /**
404      * 適用するケースを指定します。
405      * @since 2.0
406      * @return 何も指定しない場合は全てのケースに適用されます。
407      */
408     ProcessCase[] cases() default {};
409 }
