1   package com.gh.mygreen.xlsmapper.annotation;
2   
3   import java.lang.annotation.Documented;
4   import java.lang.annotation.ElementType;
5   import java.lang.annotation.Retention;
6   import java.lang.annotation.RetentionPolicy;
7   import java.lang.annotation.Target;
8   
9   
10  /**
11   * シートのマッピング対象のルートクラス（JavaBeans）に付与するアノテーション。
12   * <p>マッピング対象のシートをシート番号、シート名、シート名に対する正規表現のいずれかで指定します。</p>
13   *
14   *
15   * @author Naoki Takezoe
16   * @author T.TSUCHIE
17   *
18   */
19  @Target({ElementType.TYPE})
20  @Retention(RetentionPolicy.RUNTIME)
21  @Documented
22  public @interface XlsSheet {
23  
24      /**
25       * シート名を指定します。
26       *
27       * <pre class="highlight"><code class="java">
28       * {@literal @XlsSheet(name="Users")}
29       * public class SampleSheet {
30       *   ...
31       * }
32       * </code></pre>
33       *
34       * @return シート名
35       */
36      String name() default "";
37  
38      /**
39       * シート名を正規表現で指定します。
40       * <p>同じ形式の複数のシートを同時にマッピングする際に指定します。</p>
41       *
42       * <pre class="highlight"><code class="java">
43       * {@literal @XlsSheet(regex="Sheet_[0-9]+")}
44       * public class SampleSheet {
45       *    ・・・
46       * }
47       * </code></pre>
48       *
49       * <p>書き込み時に正規表現によるマッピングを行う際には、アノテーション{@link XlsSheetName}を利用して、
50       *    一意に関連づける必要があります。</p>
51       * <pre class="highlight"><code class="java">
52       * // マッピング用クラスの定義
53       * {@literal @XlsSheet(regex="Sheet_[0-9]+")}
54       * public class SampleSheet {
55       *
56       *    // シート名をマッピングするフィールド
57       *    {@literal @XlsSheetName}
58       *    private String sheetName;
59       *    ・・・
60       *
61       * }
62       *
63       * // 書き込み時に、シート名を設定して、一意に関連づけます。
64       * SampleSheet sheet1 = new SampleSheet();
65       * sheet1.sheetName = "Sheet_1"; // シート名の設定
66       *
67       * SampleSheet sheet2 = new SampleSheet();
68       * sheet2.sheetName = "Sheet_2"; // シート名の設定
69       *
70       * SampleSheet sheet3 = new SampleSheet();
71       * sheet3.sheetName = "Sheet_3"; // シート名の設定
72       *
73       * // 複数のシートの書き込み
74       * XlsMapper xlsMapper = new XlsMapper();
75       * xlsMapper.saveMultiple(new FileInputStream("template.xls"),
76       *   new FileOutputStream("out.xls"),
77       *   new Object[]{sheet1, sheet2, sheet3}
78       * );
79       * </code></pre>
80       *
81       * @return シート名の正規表現
82       */
83      String regex() default "";
84  
85      /**
86       * マッピング対象のシートを番号で指定します。
87       * <p>シート番号は0から始まります。</p>
88       *
89       * <pre class="highlight"><code class="java">
90       * {@literal @XlsSheet(number=0)}
91       * public class SampleSheet {
92       *   ...
93       * }
94       * </code></pre>
95       *
96       * @return シート番号
97       */
98      int number() default -1;
99  
100 }
