1   package com.gh.mygreen.xlsmapper.annotation;
2   
3   import java.lang.annotation.Documented;
4   import java.lang.annotation.ElementType;
5   import java.lang.annotation.Retention;
6   import java.lang.annotation.RetentionPolicy;
7   import java.lang.annotation.Target;
8   
9   import com.gh.mygreen.xlsmapper.CellCommentHandler;
10  
11  /**
12   * {@link XlsComment}/{@link XlsLabelledComment}/値セルのフィールド（{@code Map<String, String> comments}) の書き込み時のセルのコメントのサイズなどの制御を指定するためのアノテーションです。
13   *
14   * @since 2.1
15   * @author T.TSUCHIE
16   *
17   */
18  @Target({ElementType.METHOD, ElementType.FIELD})
19  @Retention(RetentionPolicy.RUNTIME)
20  @Documented
21  public @interface XlsCommentOption {
22      
23      /**
24       * コメントを表示状態にするかどうか指定します。
25       * <p>既に設定されているコメントの設定よりも、アノテーションの設定が優先されます。</p>
26       * @return trueのとき表示します。
27       */
28      boolean visible() default false;
29      
30      /**
31       * 書込み対象のコメントが空のとき、既に設定されているコメントを削除するかどうか指定します。
32       * @return trueのとき削除します。
33       */
34      boolean removeIfEmpty() default false;
35      
36      /**
37       * コメントの縦サイズを指定します。
38       * <p>単位は、行数で表現します。</p>
39       * <p>既にコメントが設定されている場合は無視されます。</p>
40       * @return 1以上のとき有効となります。
41       */
42      int verticalSize() default 0;
43      
44      /**
45       * コメントの横サイズを指定します。
46       * <p>単位は、列数で表現します。</p>
47       * <p>既にコメントが設定されている場合は無視されます。</p>
48       * @return 1以上の時有効とする。
49       */
50      int horizontalSize() default 0;
51      
52      /**
53       * 書き込み時にセルのコメント設定を独自に処理したいときに指定します。
54       * @return {@link CellCommentHandler}の実装クラス。
55       */
56      Class<? extends CellCommentHandler>[] handler() default {};
57      
58  }
