17  /**
18   * 書き込み時にセルの数式を定義するためのアノテーションです。
19   *
20   * <h3 class="description">式言語処理のカスタマイズ</h3>
21   *
22   * <p>数式を直接指定する場合は、EL式の1つの実装である <a href="http://commons.apache.org/proper/commons-jexl/" target="_blnak">JEXL</a>
23   *    が利用できますが、実装を切り替えたり、デフォルトの関数を登録したりとカスタマイズができます。
24   * </p>
25   * <p>設定を変更したい場合は、{@link Configuration#getFormulaFormatter()} の値を変更します。</p>
26   *
27   * <pre class="highlight"><code class="java">
28   * // 数式をフォーマットする際のEL関数を登録する。
29   * ExpressionLanguageJEXLImpl formulaEL = new ExpressionLanguageJEXLImpl();
30   * {@literal Map<String, Object>} funcs = new {@literal HashMap<>}();
31   * funcs.put("x", CustomFunctions.class);
32   * formulaEL.getJexlEngine().setFunctions(funcs);
33   *
34   * // 数式をフォーマットするEL式の実装を変更する
35   * XlsMapper mapper = new XlsMapper();
36   * mapper.getConig().getFormulaFormatter().setExpressionLanguage(formulaEL);
37   * </code></pre>
38   *
39   * @since 1.5
40   * @author T.TSUCHIE
41   *
42   */
43  @Target({ElementType.METHOD, ElementType.FIELD})
44  @Retention(RetentionPolicy.RUNTIME)
45  @Documented
46  public @interface XlsFormula {
47  
48      /**
49       * 数式を直接指定します。
50       * <p>数式を指定する際に、メッセージファイルと同様に、変数やEL式が利用可能です。</p>
51       * <ul>
52       *   <li>変数は {@literal {変数名}} で定義します。</li>
53       *   <li>EL式は {@literal ${EL式}} で定義します。
54       *     <ul>
55       *        <li>EL式は、<a href="http://commons.apache.org/proper/commons-jexl/" target="_blank">JEXL(Java Expression Language)</a>の形式で指定します。</li>
56       *        <li>JEXLの仕様は、<a href="http://commons.apache.org/proper/commons-jexl/reference/syntax.html" target="_blank">JEXL Reference</a> を参照してください。</li>
57       *     </ul>
58       *   </li>
59       * </ul>
60       *
61       * <p>EL式の中では、予め次の変数が登録されており、セルの値ごとに変わります。</p>
62       * <ul>
63       *   <li>rowIndex : 処理対象のセルの行のインデックス。0から始まります。</li>
64       *   <li>columnIndex : 処理対象のセルの列のインデックス。0から始まります。</li>
65       *   <li>rowNumber : 処理対象のセルの行番号。1から始まります。</li>
66       *   <li>columnNumber : 処理対象のセルの列番号。1から始まります。</li>
67       *   <li>columnAlpha : 処理対象のセルの列の名前。Aから始まります。</li>
68       *   <li>address : 処理対象のセルのアドレス。 A1 の形式です。</li>
69       *   <li>targetBean : 処理対象のプロパティが定義されているJavaBeanのオブジェクトです。</li>
70       *   <li>cell : 処理対象のセルのオブジェクトです。POIのクラス{@link Cell}のオブジェクトです。</li>
71       * </ul>
72       *
73       * <p>さらに、よく使う関数が登録されており、呼び出すことができます。
74       *   <br>関数の実態は、 {@link CustomFunctions}です。
75       *   <br>名前空間{@code x}で登録されており、{@literal x:colTlAlpha(...)}のように呼び出すことができます。
76       * </p>
77       *
78       * <pre class="highlight"><code class="java">
79       * {@literal @XlsSheet(name="サンプル")}
80       * public class SampleSheet {
81       *
82       *     // 数式の指定
83       *     {@literal @XlsOrder(1)}
84       *     {@literal @XlsLabelledCell(label="更新日付", type=LabelledCellType.Right)}
85       *     {@literal @XlsFormula("TODAY()")}
86       *     private Date date;
87       *
88       *     {@literal @XlsOrder(2)}
89       *     {@literal @XlsHorizontalRecords(tableLabel="レコード", terminal=RecordTerminal.Border)}
90       *     {@literal @XlsRecordOption(overOperation=OverOperation.Insert}
91       *     private {@literal List<SampleRecord>} records;
92       * }
93       *
94       * public class SampleRecord {
95       *
96       *     // マッピングした位置情報
97       *     private {@literal Map<String, CellPosition>} positions;
98       *
99       *     {@literal @XlsColumn(columnName="名前")}
100      *     private String name;
101      *
102      *     {@literal @XlsColumn(columnName="国語")}
103      *     private int kokugo;
104      *
105      *     {@literal @XlsColumn(columnName="算数")}
106      *     private int sansu;
107      *
108      *     // 数式の指定（変数、EL式を使用して指定）
109      *     {@literal @XlsColumn(columnName="合計")}
110      *     {@literal @XlsFormula(value="SUM(${x:colToAlpha(targetBean.kokugoColNum)}{rowNumber}:${x:colToAlpha(targetBean.sansuColNum)}{rowNumber})", primary=true)}
111      *     private int sum;
112      *
113      *     // プロパティ「kokugo」の列番号を返す。
114      *     public String getKokugoColNum() {
115      *         CellPosition position = positions.get("kokugo");
116      *         return position.addRow(1);
117      *
118      *     }
119      *
120      *     // プロパティ「sansu」の列番号を返す。
121      *     public String getSansuColNum() {
122      *         CellPosition position = positions.get("sansu");
123      *         return position.addRow(1);
124      *     }
125      *
126      * }
127      * </code></pre>
128      *
129      * @return Excelの数式を返す。(例. {@literal SUM(A{rowNumber},B{rowNumber})})
130      */
131     String value() default "";
132 
133     /**
134      * 数式を別メソッドで組み立てる場合に、そのメソッド名を指定します。
135      * <p>条件により数式を変更するような場合や、複雑な数式を組み立てる場合、数式を組み立てるメソッドを指定することができます。</p>
136      * <p>メソッドの条件は次のようになります。</p>
137      * <ul>
138      *    <li>定義位置は、プロパティが定義してあるJavaBeanのクラスと同じ箇所。</li>
139      *    <li>修飾子は、public/private/protected などなんでもよい。</li>
140      *    <li>引数は、指定しないか、または次の値が指定可能。順番は任意。
141      *      <ul>
142      *        <li>セルのオブジェクト : {@link Cell}</li>
143      *        <li>シートのオブジェクト : {@link Sheet}</li>
144      *        <li>セルの座標 : {@link Point}。0から始まります。</li>
145      *        <li>セルの座標 : {@link CellPosition}。</li>
146      *        <li>セルの座標 : {@link org.apache.poi.ss.util.CellAddress}。</li>
147      *        <li>システム設定 : {@link Configuration}</li>
148      *      </ul>
149      *    </li>
150      *    <li>戻り値は、String型。
151      *      <ul>
152      *        <li>nullまたは空文字を返すと、ブランクセルとして出力されます。</li>
153      *      </ul>
154      *    </li>
155      * </ul>
156      *
157      * <pre class="highlight"><code class="java">
158      * {@literal @XlsSheet(name="サンプル")}
159      * public class SampleSheet {
160      *
161      *     // 数式のメソッドの指定
162      *     {@literal @XlsOrder(1)}
163      *     {@literal @XlsLabelledCell(label="更新日付", type=LabelledCellType.Right)}
164      *     {@literal @XlsFormula(methodName="getDateFormula")}
165      *     private Date date;
166      *
167      *     {@literal @XlsOrder(2)}
168      *     {@literal @XlsHorizontalRecords(tableLabel="レコード", terminal=RecordTerminal.Border)}
169      *     {@literal @XlsRecordOption(overOperation=OverOperation.Insert}
170      *     private {@literal List<SampleRecord>} records;
171      *
172      *     // 数式を組み立てるメソッド
173      *     public String getDateFormula() {
174      *         return "TODAY()"
175      *     }
176      * }
177      *
178      * public class SampleRecord {
179      *
180      *     // マッピングした位置情報
181      *     private {@literal Map<String, CellPosition>} positions;
182      *
183      *     {@literal @XlsColumn(columnName="名前")}
184      *     private String name;
185      *
186      *     {@literal @XlsColumn(columnName="国語")}
187      *     private int kokugo;
188      *
189      *     {@literal @XlsColumn(columnName="算数")}
190      *     private int sansu;
191      *
192      *     // 数式の指定（メソッドを指定）
193      *     {@literal @XlsColumn(columnName="合計")}
194      *     {@literal @XlsFormula(methodName="getSumFormula", primary=true)}
195      *     private int sum;
196      *
197      *     // 数式を組み立てるメソッド
198      *     private String getSumFormula(CellPosition position) {
199      *
200      *         int rowNumber = CellPosition.addRow(1);
201      *         String colKokugo = CellReference.convertNumToColString(positions.get("kokugo").y);
202      *         String colSansu = CellReference.convertNumToColString(positions.get("sansu").y);
203      *
204      *         return String.format("SUM(%s%d:%s%d)", colKokugo, rowNumber, colSansu, rowNumber);
205      *     }
206      *
207      * }
208      * </code></pre>
209      *
210      * @return JavaBeanに定義されているメソッド名を返す。
211      */
212     String methodName() default "";
213 
214     /**
215      * 出力する際に、値が設定されていても、数式を優先するかどうかを指定します。
216      * <p>出力するオブジェクトのプロパティに値が設定されている場合、アノテーション {@link XlsFormula} を指定していても、デフォルトでは値が設定されます。</p>
217      * <p>属性 {@literal primary=true} を指定すると数式が優先されます。
218      *   <br>特に、プリミティブ型など初期値が入っている場合や、アノテーション {@literal @XlsConverter(defaultValue="<初期値>")} で初期値を指定している場合には、注意が必要です。
219      * </p>
220      *
221      * <pre class="highlight"><code class="java">
222      * public class SampleRecord {
223      *
224      *     // マッピングした位置情報
225      *     private {@literal Map<String, CellPosition>} positions;
226      *
227      *     {@literal @XlsColumn(columnName="名前")}
228      *     private String name;
229      *
230      *     {@literal @XlsColumn(columnName="国語")}
231      *     private int kokugo;
232      *
233      *     {@literal @XlsColumn(columnName="算数")}
234      *     private int sansu;
235      *
236      *     // 数式の指定（数式を優先する場合）
237      *     {@literal @XlsColumn(columnName="合計")}
238      *     {@literal @XlsFormula(value="SUM(B{rowNumber}:C{rowNumber})", primary=true)
239      *     private int sum;
240      *
241      * }
242      * </code></pre>
243      *
244      * @return 'true'の場合、数式の設定を優先します。
245      *         'false'の場合、値が設定されていると、その値が出力されます。
246      */
247     boolean primary() default false;
248 
249 }
