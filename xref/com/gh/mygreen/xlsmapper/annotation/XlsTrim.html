9   /**
10   * 読み込み時、または書き込み時に値をトリミング行うかどうか指定するためのアノテーション。
11   * 文字列型の時にのみ適用されます。
12   *
13   * <ul>
14   *   <li>シート上のセルのタイプ（分類）が数値などの文字列以外の場合は、トリム処理は行われません。
15   *       <br>ただし、シートのセルタイプが文字列型で、Javaの型がString型以外の数値型やDate型などの場合は、
16   *          変換する前にトリム処理を行います。
17   *   </li>
18   *   <li>値が空のセルをString型に読み込む場合、アノテーション{@link XlsTrim}を付与していないとはnullが設定されますが、
19   *       付与していると空文字が設定されます。
20   *   </li>
21   * </ul>
22   *
23   * <pre class="highlight"><code class="java">
24   * public class SampleRecord {
25   *
26   *     {@literal @XlsColumn(columnName="ID")}
27   *     {@literal @XlsTrim}
28   *     {@literal @XlsDefaultValue("  123   ")} // 初期値もトリム対象となる。
29   *     private Integer id;
30   *
31   *     {@literal @XlsColumn(columnName="名前")}
32   *     {@literal @XlsTrim}   // 空のセルを読み込むと空文字が設定される。
33   *     private String name;
34   *
35   * }
36   * </code></pre>
37   *
38   * @since 2.0
39   * @author T.TSUCHIE
40   *
41   */
42  @Target({ElementType.METHOD, ElementType.FIELD})
43  @Retention(RetentionPolicy.RUNTIME)
44  @Documented
45  public @interface XlsTrim {
46  
47  }
