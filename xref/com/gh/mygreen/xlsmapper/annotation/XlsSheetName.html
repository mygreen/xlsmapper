11  /**
12   * 処理対象のシート名をマッピングするメソッド、またはフィールドに付与します。
13   * <p>付与するフィートのクラスタイプは、{@link String}型にする必要があります。
14   *
15   * <pre class="highlight"><code class="java">
16   * {@literal @XlsSheet(number=1)}
17   * public class SampleSheet {
18   *
19   *    // シート名をマッピングするフィールド
20   *    {@literal @XlsSheetName}
21   *    private String sheetName;
22   *    ・・・
23   *
24   * }
25   * </code></pre>
26   *
27   * @version 2.0
28   * @author Naoki Takezoe
29   * @author T.TSUCHIE
30   */
31  @Target({ElementType.METHOD, ElementType.FIELD})
32  @Retention(RetentionPolicy.RUNTIME)
33  @Documented
34  @XlsFieldProcessor(value={})
35  public @interface XlsSheetName {
36  
37      /**
38       * 適用するケースを指定します。
39       * @since 2.0
40       * @return 何も指定しない場合は全てのケースに適用されます。
41       */
42      ProcessCase[] cases() default {};
43  }
