29  /**
30   * 列挙型のConverter。
31   * <p>一度読み込んだ列挙型の情報はキャッシュする。列挙型はstaticであるため、動的に変更できないため。
32   * 
33   * @author T.TSUCHIE
34   *
35   */
36  @SuppressWarnings("rawtypes")
37  public class EnumCellConverter extends AbstractCellConverter<Enum> {
38      
39      /**
40       * 列挙科型のクラスとその値とのマップ。(キャッシュデータ)
41       * <p>key=列挙型のクラスタイプ、value=列挙型の各項目の文字列形式とオブジェクト形式の値のマップ。
42       */
43      private Map<Class<?>, Map<String, Enum>> cacheData;
44      
45      public EnumCellConverter() {
46          this.cacheData =  new ConcurrentHashMap<Class<?>, Map<String,Enum>>();
47      }
48      
49      @SuppressWarnings({"unchecked"})
50      @Override
51      public Enum<?> toObject(final Cell cell, final FieldAdaptor adaptor, final XlsMapperConfig config) throws XlsMapperException {
52          final XlsConverter converterAnno = adaptor.getLoadingAnnotation(XlsConverter.class);
53          final XlsEnumConverter anno = getLoadingAnnotation(adaptor);
54          
55          String cellValue = POIUtils.getCellContents(cell, config.getCellFormatter());
56          cellValue = Utils.trim(cellValue, converterAnno);
57          if(Utils.isEmpty(cellValue) && Utils.hasNotDefaultValue(converterAnno)) {
58              return null;
59          }
60          cellValue = Utils.getDefaultValueIfEmpty(cellValue, converterAnno);
61          
62          final Class<Enum> taretClass = (Class<Enum>) adaptor.getTargetClass();
63          Enum<?> resultValue = convertToObject(cellValue, taretClass, anno);
64          if(resultValue == null && Utils.isNotEmpty(cellValue)) {
65              // 値があり変換できない場合
66              throw newTypeBindException(cell, adaptor, cellValue)
67                  .addAllMessageVars(createTypeErrorMessageVars(taretClass, anno));
68          }
69          
70          return resultValue;
71      }
72      
73      /**
74       * 型変換エラー時のメッセージ変数の作成
75       * @throws ConversionException 
76       */
77      private Map<String, Object> createTypeErrorMessageVars(final Class<Enum> taretClass, final XlsEnumConverter anno) throws ConversionException {
78          
79          final Map<String, Object> vars = new LinkedHashMap<>();
80          vars.put("candidateValues", Utils.join(getLoadingAvailableValue(taretClass, anno), ", "));
81          vars.put("ignoreCase", anno.ignoreCase());
82          
83          return vars;
84      }

86      private XlsEnumConverter getDefaultEnumConverterAnnotation() {
87          return new XlsEnumConverter() {
88              
89              @Override
90              public Class<? extends Annotation> annotationType() {
91                  return XlsEnumConverter.class;
92              }
93              
94              @Override
95              public boolean ignoreCase() {
96                  return false;
97              }
98              
99              @Override
100             public String valueMethodName() {
101                 return "";
102             }
103             
104         };
105     }

107     private XlsEnumConverter getLoadingAnnotation(final FieldAdaptor adaptor) {
108         XlsEnumConverter anno = adaptor.getLoadingAnnotation(XlsEnumConverter.class);
109         if(anno == null) {
110             anno = getDefaultEnumConverterAnnotation();
111         }
112         return anno;
113     }
114     
115     private XlsEnumConverter getSavingAnnotation(final FieldAdaptor adaptor) {
116         XlsEnumConverter anno = adaptor.getSavingAnnotation(XlsEnumConverter.class);
117         if(anno == null) {
118             anno = getDefaultEnumConverterAnnotation();
119         }
120         return anno;
121     }

123     /**
124      * 読み込み時の入力の候補となる値を取得する
125      * @param clazz 列挙型の値
126      * @param anno
127      * @return
128      * @throws ConversionException 
129      */
130     private Collection<String> getLoadingAvailableValue(final Class<Enum> clazz, final XlsEnumConverter anno) throws ConversionException {
131         
132         final Set<String> values = new LinkedHashSet<String>();
133         
134         final Map<String, Enum> map = getEnumValueMapFromCache(clazz);
135         for(Map.Entry<String, Enum> entry : map.entrySet()) {
136             if(anno.valueMethodName().isEmpty()) {
137                 values.add(entry.getKey());
138             } else {
139                 try {
140                     final Method method = clazz.getMethod(anno.valueMethodName(), new Class[]{});
141                     method.setAccessible(true);
142                     final String value = method.invoke(entry.getValue(), new Object[]{}).toString();
143                     values.add(value);
144                 } catch(Exception e) {
145                     throw new ConversionException(
146                             String.format("Not found Enum method '%s#%s()'.", clazz.getName(), anno.valueMethodName()),
147                             e, clazz);
148                 }
149             }
150         }
151         
152         return values;
153         
154     }
155     
156     /**
157      * 列挙型のキーと値のマップをキャッシュから取得する。
158      * <p>key=列挙型のname()の値。value=列挙型の各項目のオブジェクト。
159      * <p>キャッシュ上に存在しなければ、新しく情報を作成しキャッシュに追加する。
160      * @param clazz
161      * @return
162      */
163     private synchronized Map<String, Enum> getEnumValueMapFromCache(final Class<Enum> clazz) {
164         if(cacheData.containsKey(clazz)) {
165             return cacheData.get(clazz);
166         }
167         
168         final Map<String, Enum> map = createEnumValueMap(clazz);
169         cacheData.put(clazz, map);
170         return map;
171     }
172     
173     /**
174      * 列挙型のマップを作成する。
175      * 
176      * @param clazz 列挙型のクラス。
177      * @return key=項目名（{@link Enum#name()}メソッドの値）、value=列挙型の項目オブジェクト。
178      */
179     @SuppressWarnings({"unchecked"})
180     private Map<String, Enum> createEnumValueMap(final Class<Enum> clazz) {
181         
182         final Map<String, Enum> map = new LinkedHashMap<>();
183         final EnumSet set = EnumSet.allOf(clazz);
184         final Iterator<Enum> it = set.iterator();
185         while(it.hasNext()) {
186             final Enum e = it.next();
187             map.put(e.name(), e);
188         }
189         
190         return Collections.unmodifiableMap(map);
191     }
192     
193     private Enum<?> convertToObject(final String value, final Class<Enum> clazz, final XlsEnumConverter anno) throws ConversionException {
194         
195         final Map<String, Enum> map = getEnumValueMapFromCache(clazz);
196         for(Map.Entry<String, Enum> entry : map.entrySet()) {
197             
198             final String key;
199             if(anno.valueMethodName().isEmpty()) {
200                 key = entry.getKey();
201             } else {
202                 try {
203                     final Method method = clazz.getMethod(anno.valueMethodName(), new Class[]{});
204                     method.setAccessible(true);
205                     key = method.invoke(entry.getValue(), new Object[]{}).toString();
206                 } catch(Exception e) {
207                     throw new ConversionException(
208                             String.format("Not found Enum method '%s#%s()'.", clazz.getName(), anno.valueMethodName()),
209                             e, clazz);
210                 }
211             }
212             
213             if(anno.ignoreCase() && value.equalsIgnoreCase(key)) {
214                 return entry.getValue();
215                 
216             } else if(value.equals(key)) {
217                 return entry.getValue();
218             }
219         }
220         
221         return null;
222         
223     }
224     
225     private String convertToString(final Enum<?> value, final Class<Enum> clazz, final XlsEnumConverter anno) throws ConversionException {
226         
227         final Map<String, Enum> map = getEnumValueMapFromCache(clazz);
228         for(Map.Entry<String, Enum> entry : map.entrySet()) {
229             
230             if(entry.getValue() != value) {
231                 continue;
232             }
233             
234             if(anno.valueMethodName().isEmpty()) {
235                 return value.name();
236             } else {
237                 try {
238                     final Method method = clazz.getMethod(anno.valueMethodName(), new Class[]{});
239                     method.setAccessible(true);
240                     return method.invoke(entry.getValue(), new Object[]{}).toString();
241                 } catch(Exception e) {
242                     throw new ConversionException(
243                             String.format("Not found Enum method '%s#%s()'.", clazz.getName(), anno.valueMethodName()),
244                             e, clazz);
245                 }
246             }
247         }
248         
249         return null;
250         
251     }

267     @SuppressWarnings("unchecked")
268     private Cell toCell(final FieldAdaptor adaptor, final Object targetObj, final Sheet sheet,
269             final int column, final int row, final XlsMapperConfig config, final String mapKey) throws XlsMapperException {
270         
271         final XlsConverter converterAnno = adaptor.getLoadingAnnotation(XlsConverter.class);
272         final XlsEnumConverter anno = getSavingAnnotation(adaptor);
273         
274         final Cell cell = POIUtils.getCell(sheet, column, row);
275         
276         // セルの書式設定
277         if(converterAnno != null) {
278             POIUtils.wrapCellText(cell, converterAnno.forceWrapText());
279             POIUtils.shrinkToFit(cell, converterAnno.forceShrinkToFit());
280         }
281         
282         Enum value;
283         if(mapKey == null) {
284             value = (Enum) adaptor.getValue(targetObj);
285         } else {
286             value = (Enum) adaptor.getValueOfMap(mapKey, targetObj);
287         }
288         
289         // デフォルト値から値を設定する
290         if(value == null && Utils.hasDefaultValue(converterAnno)) {
291             value = convertToObject(Utils.getDefaultValue(converterAnno), (Class<Enum>) adaptor.getTargetClass(), anno);
292         }
293         
294         if(value != null) {
295             final String cellValue = convertToString(value, (Class<Enum>) adaptor.getTargetClass(), anno);
296             cell.setCellValue(cellValue);
297         } else {
298             cell.setCellType(Cell.CELL_TYPE_BLANK);
299         }
300         
301         return cell;
302     }
