14  /**
15   * char/Charcter型を処理するConverter.
16   *
17   * @author T.TSUCHIE
18   *
19   */
20  public class CharacterCellConverter extends AbstractCellConverter<Character> {
21  
22      @Override
23      public Character toObject(Cell cell, FieldAdaptor adaptor, XlsMapperConfig config) {
24          
25          final XlsConverter converterAnno = adaptor.getLoadingAnnotation(XlsConverter.class);
26          
27          String cellValue = POIUtils.getCellContents(cell, config.getCellFormatter());
28          cellValue = Utils.trim(cellValue, converterAnno);
29          
30          Character resultValue = null;
31          if(Utils.isEmpty(cellValue)) {
32              if(Utils.hasDefaultValue(converterAnno)) {
33                  resultValue = converterAnno.defaultValue().charAt(0);
34              }
35              
36          } else {
37              resultValue = cellValue.charAt(0);
38          }
39          
40          if(resultValue == null && adaptor.getTargetClass().isPrimitive()) {
41              resultValue = '\u0000';
42          }
43          
44          return resultValue;
45      }
46  
47      @Override
48      public Cell toCell(final FieldAdaptor adaptor, final Character targetValue, final Sheet sheet, final int column, final int row,
49              final XlsMapperConfig config) {
50          
51          final XlsConverter converterAnno = adaptor.getSavingAnnotation(XlsConverter.class);
52          
53          final Cell cell = POIUtils.getCell(sheet, column, row);
54          
55          // セルの書式設定
56          if(converterAnno != null) {
57              POIUtils.wrapCellText(cell, converterAnno.forceWrapText());
58              POIUtils.shrinkToFit(cell, converterAnno.forceShrinkToFit());
59          }
60          
61          Character value = targetValue;
62          
63          String cellValue = Utils.trim(value, converterAnno);
64          cellValue = Utils.getDefaultValueIfEmpty(cellValue, converterAnno);
65          if(cellValue.length() >= 1) {
66              cellValue = cellValue.substring(0, 1);
67          }
68          
69          if(Utils.isNotEmpty(cellValue)) {
70              cell.setCellValue(cellValue);
71          } else {
72              cell.setCellType(Cell.CELL_TYPE_BLANK);
73          }
74          
75          return cell;
76      }
77  }
