11  /**
12   * Converterの抽象クラス。
13   * 通常は、このクラスを継承してConverterを作成する。
14   * 
15   * @author T.TSUCHIE
16   *
17   */
18  public abstract class AbstractCellConverter<T> implements CellConverter<T> {
19      
20      /**
21       * 型変換失敗したときの例外{@link TypeBindException}をスローします。
22       * @param cell
23       * @param adaptor
24       * @param targetValue
25       * @return
26       */
27      public TypeBindException newTypeBindException(final Cell cell, final FieldAdaptor adaptor, final Object targetValue) {
28          
29          final String message = new StringBuilder()
30              .append(String.format("Fail conversion field value '%s' => type '%s'.",
31                      Utils.convertToString(targetValue), adaptor.getTargetClass()))
32              .append(String.format(" Cell '%s' map to '%s#%s'.", 
33                      Utils.formatCellAddress(cell), adaptor.getDeclaringClass().getName(), adaptor.getName()))
34              .toString();
35          
36          return new TypeBindException(message, adaptor.getTargetClass(), targetValue);
37          
38      }

40  }
