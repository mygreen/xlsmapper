5   
6   import com.gh.mygreen.xlsmapper.XlsMapperConfig;
7   import com.gh.mygreen.xlsmapper.XlsMapperException;
8   import com.gh.mygreen.xlsmapper.fieldprocessor.FieldAdaptor;

10  
11  /**
12   * Cellの値を変換するインタフェース。
13   *
14   * @version 1.5
15   * @author T.TSUCHIE
16   * @param <T> 変換対象のJavaのオブジェクト
17   */
18  public interface CellConverter<T> {
19      
20      /**
21       * シート読み込み時のExcel Cell => Javaオブジェクトに変換する。
22       * @param cell 読み込み対象のセル
23       * @param adaptor マッピング対象のフィールド情報。
24       * @param config 設定情報
25       * @return 変換したJavaオブジェクト
26       * @throws XlsMapperException
27       */
28      T toObject(Cell cell, FieldAdaptor adaptor, XlsMapperConfig config) throws XlsMapperException;
29      
30      
31      /**
32       * シート書き込み時のJavaオブジェクト => Excel Cellに変換する。
33       * @param adaptor  マッピング対象のフィールド情報。
34       * @param targetValue 書き込み対象のオブジェクト。
35       * @param targetBean 書き込み対象のフィールドが設定されているJavaBeanオブジェクト。
36       * @param sheet 書き込み先のシート
37       * @param column 書き込み先のセルの列番号。0から始まる。
38       * @param row 書き込み先のセルの行番号。0から始まる。
39       * @param config 設定情報
40       * @return 書き込んだセル
41       * @throws XlsMapperException 
42       */
43      Cell toCell(FieldAdaptor adaptor, T targetValue, Object targetBean, Sheet sheet, int column, int row, XlsMapperConfig config) throws XlsMapperException;
44      
45  }
