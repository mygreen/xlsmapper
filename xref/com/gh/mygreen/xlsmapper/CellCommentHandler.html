5   import org.apache.poi.ss.usermodel.Cell;

7   import com.gh.mygreen.xlsmapper.annotation.XlsCommentOption;

9   /**
10   * セルのコメントを操作する
11   * <p>シートを書き込む時に、コメントの位置やサイズを任意で指定したい時に独自の実装を指定します。</p>
12   *
13   * @since 2.1
14   * @author T.TSUCHIE
15   *
16   */
17  public interface CellCommentHandler {
18      
19      /**
20       * シート読み込み時にセルのコメントを取得する。
21       * @param cell コメント取得対象のセル
22       * @param commentOption コメントのオプション
23       * @return セルのコメント。コメントがない場合は、空を返す。
24       */
25      Optional<String> handleLoad(Cell cell, Optional<XlsCommentOption> commentOption);
26      
27      /**
28       * シート書き込み時にセルのコメントを設定する。
29       * @param cell 書込み対象のセル
30       * @param text コメント。コメントがない場合は、空が設定されます。
31       * @param commentOption コメントのオプション
32       */
33      void handleSave(Cell cell, Optional<String> text, Optional<XlsCommentOption> commentOption);
34      
35  }
