68      private Validator getBeanValidator() {
69          
70          ValidatorFactory validatorFactory = Validation.buildDefaultValidatorFactory();
71          Validator validator = validatorFactory.usingContext()
72                  .messageInterpolator(new MessageInterpolatorAdapter(
73                          new ResourceBundleMessageResolver(), new MessageInterpolator()))
74                  .getValidator();
75          
76          return validator;
77      }

380     private void printErrors(SheetBindingErrors<?> errors) {
381         
382         for(ObjectError error : errors.getAllErrors()) {
383             String message = errorFormatter.format(error);
384             System.out.println(message);

389     @XlsSheet(name="単純なBean")
390     private static class SimpleBeanSheet {
391         
392         private Map<String, Point> positions;
393         
394         private Map<String, String> labels;
395         
396         @Length(max=10)
397         @XlsLabelledCell(label="説明", type=LabelledCellType.Bottom)
398         private String description;
399         
400         @Range(min=0, max=100)
401         @XlsLabelledCell(label="年齢", type=LabelledCellType.Right)
402         private Integer age;
403         
404         @NotBlank
405         @Email
406         @XlsLabelledCell(label="e-mail(必須)", type=LabelledCellType.Right)
407         private String email;
408         
409         @DecimalMin(value="0.0", inclusive=true)
410         @XlsLabelledCell(label="得点（平均）", type=LabelledCellType.Right)
411         private Double average;
412         
413         @NotNull
414         @Future
415         @XlsDateTimeConverter(lenient=true, javaPattern="yyyy年M月d日")
416         @XlsLabelledCell(label="更新日時", type=LabelledCellType.Right)
417         private Date updateTime;
418         
419     }

438     private static class PersonRecord {
439         
440         private Map<String, Point> positions;
441         
442         private Map<String, String> labels;
443         
444         @XlsColumn(columnName="No.")
445         private int no;
446         
447         @Length(max=10)
448         @XlsColumn(columnName="氏名")
449         private String name;
450         
451         @NotBlank
452         @Email
453         @XlsColumn(columnName="メールアドレス")
454         private String email;
455         
456         @NotNull
457         @Past
458         @XlsDateTimeConverter(lenient=true, javaPattern="yyyy年M月d日")
459         @XlsColumn(columnName="生年月日")
460         private Date birthday;
461         
462         @XlsIgnorable
