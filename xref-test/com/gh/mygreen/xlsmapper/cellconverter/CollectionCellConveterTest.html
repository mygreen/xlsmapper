1074         private static class DateElementConverter implements ElementConverter<Date> {
1075 
1076             @Override
1077             public Date convertToObject(final String text, final Class<Date> targetClass) throws ConversionException {
1078                 SimpleDateFormat formatter = new SimpleDateFormat("yyyy年MM月dd日");
1079                 try {
1080                     return formatter.parse(text);
1081                 } catch (ParseException e) {
1082                     String message = String.format("fail parse '%s' to java.util.Date", text);
1083                     throw new ConversionException(message, e, targetClass);
1084                 }
1085             }

1089 
1090                 if(value == null) {

1092                 }
1093

1095                 return formatter.format(value);
