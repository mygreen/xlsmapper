470     @XlsSheet(name="日時型")
471     private static class DateTimeSheet {
472 
473         @XlsOrder(1)
474         @XlsHorizontalRecords(tableLabel="日時型（アノテーションなし）", terminal=RecordTerminal.Border)
475         @XlsRecordOption(overOperation=OverOperation.Insert)
476         private List<SimpleRecord> simpleRecords;
477 
478         @XlsOrder(2)
479         @XlsHorizontalRecords(tableLabel="日付型（初期値、書式）", terminal=RecordTerminal.Border)
480         @XlsRecordOption(overOperation=OverOperation.Insert)
481         private List<FormattedRecord> formattedRecords;
482 
483         @XlsOrder(3)
484         @XlsHorizontalRecords(tableLabel="日時型（数式）", terminal=RecordTerminal.Border)
485         @XlsRecordOption(overOperation=OverOperation.Insert)
486         private List<FormulaRecord> formulaRecords;
487 
488         /**
489          * レコードを追加する。noを自動的に付与する。
490          * @param record
491          * @return
492          */
493         public DateTimeSheet add(SimpleRecord record) {
494             if(simpleRecords == null) {
495                 this.simpleRecords = new ArrayList<>();
496             }
497             this.simpleRecords.add(record);
498             record.no(simpleRecords.size());
499             return this;
500         }
501 
502         /**
503          * レコードを追加する。noを自動的に付与する。
504          * @param record
505          * @return
506          */
507         public DateTimeSheet add(FormattedRecord record) {
508             if(formattedRecords == null) {
509                 this.formattedRecords = new ArrayList<>();
510             }
511             this.formattedRecords.add(record);
512             record.no(formattedRecords.size());
513             return this;
514         }

523                 this.formulaRecords = new ArrayList<>();
