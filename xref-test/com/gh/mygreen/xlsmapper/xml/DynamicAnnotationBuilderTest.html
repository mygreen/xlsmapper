41      @Test
42      public void test_buildAnnotation1() throws Exception {
43          
44          AnnotationInfo info = new AnnotationInfo();
45          info.addAttribute("row", "10");
46          info.addAttribute("column", "99");
47          
48          DynamicAnnotationBuilder builder = DynamicAnnotationBuilder.getInstance();
49          Annotation ann = builder.buildAnnotation(XlsCell.class, info);
50          
51          assertThat(ann.annotationType(), is(typeCompatibleWith(XlsCell.class)));
52          
53          XlsCell cell = (XlsCell) ann;
54          assertThat(cell.row(), is(10));
55          assertThat(cell.column(), is(99));
56          
57      }

59      @Test
60      public void test_buildAnnotation2() throws Exception {
61          
62          DynamicAnnotationBuilder builder = DynamicAnnotationBuilder.getInstance();
63          Annotation ann =  builder.buildAnnotation(XlsHorizontalRecords.class, new AnnotationInfo());
64          
65          assertThat(ann.annotationType(), is(typeCompatibleWith(XlsHorizontalRecords.class)));
66          
67          XlsHorizontalRecords records = (XlsHorizontalRecords) ann;
68          
69          assertThat(records.optional(), is(false));
70          assertThat(records.tableLabel(), is(""));
71          assertThat(records.terminateLabel(), is(""));
72          assertThat(records.headerRow(), is(-1));
73          assertThat(records.headerColumn(), is(-1));
74          assertThat(records.recordClass(), is(typeCompatibleWith(Object.class)));
75          assertThat(records.terminal(), is(RecordTerminal.Empty));
76          assertThat(records.range(), is(1));
77          assertThat(records.bottom(), is(1));
78          
79          assertThat(records.headerAddress(), is(""));
80          
81      }
