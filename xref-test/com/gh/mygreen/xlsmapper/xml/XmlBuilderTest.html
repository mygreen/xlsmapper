61      public void test_simple() throws Exception {
62  
63          AnnotationMappingInfo xmlInfo = createXml()
64                  .classInfo(createClass(SimpleSheet.class)
65                          .annotation(createAnnotation(XlsSheet.class)
66                                  .attribute("name", "単純なシート")
67                                  .buildAnnotation())
68                          .field(createField("sheetName")
69                                  .annotation(createAnnotation(XlsSheetName.class)
70                                          .buildAnnotation())
71                                  .buildField())
72                          .field(createField("name")
73                                  .annotation(createAnnotation(XlsLabelledCell.class)
74                                          .attribute("label", "名称")
75                                          .attribute("type", LabelledCellType.Right)
76                                          .buildAnnotation())
77                                  .annotation(createAnnotation(XlsTrim.class)
78                                          .buildAnnotation())
79                                  .annotation(createAnnotation(XlsDefaultValue.class)
80                                          .attribute("value", "－")
81                                          .buildAnnotation())
82                                  .buildField())
83                          .method(createMethod("setRecords")
84                                  .annotation(createAnnotation(XlsHorizontalRecords.class)
85                                          .attribute("tableLabel", "名簿一覧")
86                                          .attribute("terminal", RecordTerminal.Border)
87                                          .buildAnnotation())
88                                  .buildMethod())
89                          .buildClass())
90                  .buildXml();
91  
92  
93          System.out.println(xmlInfo.toXml());
94  
95          AnnotationReader reader = new AnnotationReader(xmlInfo);
96  
97          // クラス定義の読み込み
98          Annotation[] classAnnos = reader.getAnnotations(SimpleSheet.class);
99          XlsSheet sheetAnno = select(classAnnos, XlsSheet.class);
100         assertThat(sheetAnno.name(), is("単純なシート"));
101 
102         // クラス定義の読み込み（アノテーションを指定）
103         sheetAnno = reader.getAnnotation(SimpleSheet.class, XlsSheet.class);
104         assertThat(sheetAnno.name(), is("単純なシート"));
105 
106         // フィールド定義の読み込み
107         Annotation[] nameAnnos = reader.getAnnotations(SimpleSheet.class.getDeclaredField("name"));
108 
109         XlsLabelledCell labeldCellAnno = select(nameAnnos, XlsLabelledCell.class);
110         assertThat(labeldCellAnno.label(), is("名称"));
111         assertThat(labeldCellAnno.type(), is(LabelledCellType.Right));
112 
113         XlsTrim trimAnno = select(nameAnnos, XlsTrim.class);
114         assertThat(trimAnno, is(not(nullValue())));
115 
116         XlsDefaultValue defaultValueAnno = select(nameAnnos, XlsDefaultValue.class);
117         assertThat(defaultValueAnno.value(), is("－"));
118 
119 
120         // フィールド定義の読み込み（アノテーションを指定）
121         labeldCellAnno = reader.getAnnotation(SimpleSheet.class.getDeclaredField("name"), XlsLabelledCell.class);
122         assertThat(labeldCellAnno.label(), is("名称"));
123         assertThat(labeldCellAnno.type(), is(LabelledCellType.Right));
124 
125         // メソッドの定義の読み込み
126         Annotation[] recordsAnnos = reader.getAnnotations(SimpleSheet.class.getDeclaredMethod("setRecords", List.class));
127 
128         XlsHorizontalRecords horizontalRecordsAnno = select(recordsAnnos, XlsHorizontalRecords.class);
129         assertThat(horizontalRecordsAnno.tableLabel(), is("名簿一覧"));
130         assertThat(horizontalRecordsAnno.terminal(), is(RecordTerminal.Border));
131 
132         // メソッドの定義の読み込み（アノテーションを指定）
133         horizontalRecordsAnno = reader.getAnnotation(SimpleSheet.class.getDeclaredMethod("setRecords", List.class), XlsHorizontalRecords.class);
134         assertThat(horizontalRecordsAnno.tableLabel(), is("名簿一覧"));
135         assertThat(horizontalRecordsAnno.terminal(), is(RecordTerminal.Border));

144     @Test
145     public void test_override() throws Exception {
146 
147         AnnotationMappingInfo xmlInfo = createXml()
148                 .classInfo(createClass(OrverrideSheet.class)
149                         .override(true)
150                         .annotation(createAnnotation(XlsSheet.class)
151                                 .attribute("name", "")
152                                 .attribute("regex", "リスト.+")
153                                 .buildAnnotation())
154                         .field(createField("name")
155                                 .override(true)
156                                 .annotation(createAnnotation(XlsLabelledCell.class)
157                                         .attribute("label", "クラス名")
158                                         .attribute("type", LabelledCellType.Bottom)
159                                         .buildAnnotation())
160                                 .buildField())
161                         .method(createMethod("setRecords")
162                                 .override(true)
163                                 .annotation(createAnnotation(XlsHorizontalRecords.class)
164                                         .attribute("tableLabel", "名簿一覧")
165                                         .attribute("terminal", RecordTerminal.Border)
166                                         .buildAnnotation())
167                                 .buildMethod())
168                         .buildClass())
169                 .buildXml();
170 
171         System.out.println(xmlInfo.toXml());
172 
173         AnnotationReader reader = new AnnotationReader(xmlInfo);
174 
175         // クラス定義の読み込み
176         XlsSheet sheetAnno = reader.getAnnotation(OrverrideSheet.class, XlsSheet.class);
177         assertThat(sheetAnno.name(), is(""));
178         assertThat(sheetAnno.regex(), is("リスト.+"));
179 
180         // フィールド定義の読み込み
181         Annotation[] nameAnnos = reader.getAnnotations(OrverrideSheet.class.getDeclaredField("name"));
182 
183         // フィールド - XMLに定義している
184         XlsLabelledCell labeldCellAnno = select(nameAnnos, XlsLabelledCell.class);
185         assertThat(labeldCellAnno.label(), is("クラス名"));
186         assertThat(labeldCellAnno.type(), is(LabelledCellType.Bottom));
187 
188         {
189             // フィールド - XMLに定義していない
190             XlsCellOption cellOptionAnno = select(nameAnnos, XlsCellOption.class);
191             assertThat(cellOptionAnno.shrinkToFit(), is(true));
192 
193             XlsTrim trimAnno = select(nameAnnos, XlsTrim.class);
194             assertThat(trimAnno, is(not(nullValue())));
195 
196             XlsDefaultValue defaultValueANno = select(nameAnnos, XlsDefaultValue.class);
197             assertThat(defaultValueANno.value(), is("－"));
198 
199         }
200 
201         XlsOrder hintAnno1 = select(nameAnnos, XlsOrder.class);
202         assertThat(hintAnno1.value(), is(1));
203 
204         // メソッド定義の読み込み
205         Annotation[] recordsAnnos = reader.getAnnotations(OrverrideSheet.class.getDeclaredMethod("setRecords", List.class));
206 
207         // メソッド - XMLに定義している
208         XlsHorizontalRecords horizontalRecordsAnno = select(recordsAnnos, XlsHorizontalRecords.class);
209         assertThat(horizontalRecordsAnno.tableLabel(), is("名簿一覧"));
210         assertThat(horizontalRecordsAnno.terminal(), is(RecordTerminal.Border));
211 
212         // メソッド - XMLに定義していない
213         XlsOrder hintAnno2 = select(recordsAnnos, XlsOrder.class);
214         assertThat(hintAnno2.value(), is(2));
215     }

471     private <A extends Annotation> A select(Annotation[] annos, Class<A> clazz) {
472 
473         for(Annotation anno : annos) {
474             if(anno.annotationType().equals(clazz)) {
475                 return (A) anno;
476             }
477         }
478 
479         return null;
