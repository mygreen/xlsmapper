275         // XlsMapColumnsのマッピング用のセルを作成する
276         @XlsPreSave
277         public void onPreSave(final Sheet sheet, final Configuration config) {
278 
279             try {
280                 final Workbook workbook = sheet.getWorkbook();
281 
282                 // 基準となる日付のセル[日付]を取得する
283                 Cell baseHeaderCell = CellFinder.query(sheet, "[日付]", config).findWhenNotFoundException();
284 
285                 // 書き換えるための見出しの値の取得
286                 List<String> dateHeaders = new ArrayList<>(records.get(0).attendedMap.keySet());
287 
288                 // 1つ目の見出しの書き換え
289                 baseHeaderCell.setCellValue(dateHeaders.get(0));
290 
291                 // ２つ目以降の見出し列の追加
292                 Row headerRow = baseHeaderCell.getRow();
293                 for(int i=1; i < dateHeaders.size(); i++) {
294                     Cell headerCell = headerRow.createCell(baseHeaderCell.getColumnIndex() + i);
295 
296                     CellStyle style = workbook.createCellStyle();
297                     style.cloneStyleFrom(baseHeaderCell.getCellStyle());
298                     headerCell.setCellStyle(style);
299                     headerCell.setCellValue(dateHeaders.get(i));
300 
301                 }
302 
303                 // 2つ目以降のデータ行の列の追加
304                 Row valueRow = sheet.getRow(baseHeaderCell.getRowIndex() + 1);
305                 Cell baseValueCell = valueRow.getCell(baseHeaderCell.getColumnIndex());
306                 for(int i=1; + i < dateHeaders.size(); i++) {
307                     Cell valueCell = valueRow.createCell(baseValueCell.getColumnIndex() + i);
308 
309                     CellStyle style = workbook.createCellStyle();
310                     style.cloneStyleFrom(baseValueCell.getCellStyle());
311                     valueCell.setCellStyle(style);
312 
313                 }
314 
315             } catch (Exception e) {
316                 throw new RuntimeException(e);
317             }
318

348      * {@link XlsSheet} - 正規表現の場合のシートのコピー
349      */
350     @Test
351     public void test_Sheet_regex_clone() throws Exception {
352 
353         // 正規表現による複数のシートを出力する場合。
354         // 書き込み時に、シート名を設定して、一意に関連づけます。
355         SheetRegexClone sheet1 = new SheetRegexClone();
356         sheet1.sheetName = "Sheet_1"; // シート名の設定
357 
358         SheetRegexClone sheet2 = new SheetRegexClone();
359         sheet2.sheetName = "Sheet_2"; // シート名の設定
360 
361         SheetRegexClone sheet3 = new SheetRegexClone();
362         sheet3.sheetName = "Sheet_3"; // シート名の設定
363 
364         SheetRegexClone[] sheets = new SheetRegexClone[]{sheet1, sheet2, sheet3};
365 
366         // シートのクローン
367         Workbook workbook = WorkbookFactory.create(new FileInputStream("src/test/data/sample_template.xlsx"));
368         Sheet templateSheet = workbook.getSheet("XlsSheet(regexp)");
369         for(SheetRegexClone sheetObj : sheets) {
370             int sheetIndex = workbook.getSheetIndex(templateSheet);
371             Sheet cloneSheet = workbook.cloneSheet(sheetIndex);
372             workbook.setSheetName(workbook.getSheetIndex(cloneSheet), sheetObj.sheetName);
373         }
374 
375         // コピー元のシートを削除する
376         workbook.removeSheetAt(workbook.getSheetIndex(templateSheet));
377 
378         // クローンしたシートファイルを、一時ファイルに一旦出力する。
379         File cloneTemplateFile = File.createTempFile("template", ".xlsx");
380         workbook.write(new FileOutputStream(cloneTemplateFile));
381 
382         // 複数のシートの書き込み
383         XlsMapper xlsMapper = new XlsMapper();
384         xlsMapper.saveMultiple(
385                 new FileInputStream(cloneTemplateFile), // クローンしたシートを持つファイルを指定する
386                 new FileOutputStream(new File(OUT_DIR, "sample_out.xlsx")),
387                 sheets);
388 
389     }

417         // XmlMapperクラスに直接渡せます。
