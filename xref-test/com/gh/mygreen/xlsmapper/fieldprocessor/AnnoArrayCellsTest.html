341             assertThat(sheet.labels).isNull();
342 
343             assertThat(sheet.text).containsExactly("あいう", "かきく", "", "");

350 
351             assertThat(sheet.number_double).containsExactly(1234.567d, 0.0d, 0.0d, 0.0d);
352             assertThat(cellFieldError(errors, cellAddress(sheet.positions.get("number_double[3]"))).isConversionFailure()).isTrue();
353 
354 
355             // converter
356             {
357                 FieldFormatter<String> formatter = errors.getFieldFormatterRegistry().getFormatter("text", String.class);
358                 assertThat(formatter).isNotNull();
359                 assertThat(formatter.format(" あいう  ")).isEqualTo(" あいう  ");
360             }
361 
362             {
363                 FieldFormatter<Boolean> formatter = errors.getFieldFormatterRegistry().getFormatter("bool", Boolean.class);
364                 assertThat(formatter).isNotNull();
365                 assertThat(formatter.format(true)).isEqualTo("○");
366             }
367 
368             {
369                 FieldFormatter<Integer> formatter = errors.getFieldFormatterRegistry().getFormatter("number_int", Integer.class);
370                 assertThat(formatter).isNotNull();
371                 assertThat(formatter.format(123456)).isEqualTo("123,456");
372 
373             }
374 
375             {
376                 FieldFormatter<Double> formatter = errors.getFieldFormatterRegistry().getFormatter("number_double", double.class);
377                 assertThat(formatter).isNotNull();

379 
380             }
381
