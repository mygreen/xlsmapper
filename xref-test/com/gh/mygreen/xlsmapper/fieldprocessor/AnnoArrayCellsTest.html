344

352

355 
356 
357             // converter
358             {
359                 FieldFormatter<String> formatter = errors.getFieldFormatterRegistry().getFormatter("text", String.class);
360                 assertThat(formatter).isNotNull();
361                 assertThat(formatter.format(" あいう  ")).isEqualTo(" あいう  ");
362             }
363 
364             {
365                 FieldFormatter<Boolean> formatter = errors.getFieldFormatterRegistry().getFormatter("bool", Boolean.class);

370             {
371                 FieldFormatter<Integer> formatter = errors.getFieldFormatterRegistry().getFormatter("number_int", Integer.class);
372                 assertThat(formatter).isNotNull();
373                 assertThat(formatter.format(123456)).isEqualTo("123,456");
374 
375             }
376 
377             {
378                 FieldFormatter<Double> formatter = errors.getFieldFormatterRegistry().getFormatter("number_double", double.class);
