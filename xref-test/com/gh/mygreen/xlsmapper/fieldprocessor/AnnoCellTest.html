176         
177         // テストデータの作成
178         final NormalSheet outSheet = new NormalSheet();
179         
180         outSheet.c1("文字列です。改行あり")
181                 .c2(12.345)
182                 .c3(toUtilDate(toTimestamp("2015-06-06 10:12:13.000")))
183                 .c4(-12345);
184         
185         // ファイルへの書き込み
186         XlsMapper mapper = new XlsMapper();
187         mapper.getConig().setContinueTypeBindFailure(true);
188         
189         File outFile = new File(OUT_DIR, "anno_Cell_out.xlsx");
190         try(InputStream template = new FileInputStream("src/test/data/anno_Cell_template.xlsx");
191                 OutputStream out = new FileOutputStream(outFile)) {
192             
193             mapper.save(template, out, outSheet);
194         }
195         
196         // 書き込んだファイルを読み込み値の検証を行う。
197         try(InputStream in = new FileInputStream(outFile)) {
198             
199             SheetBindingErrors errors = new SheetBindingErrors(NormalSheet.class);
200             
201             NormalSheet sheet = mapper.load(in, NormalSheet.class, errors);
202             
203             assertThat(sheet.positions, is(outSheet.positions));
204             assertThat(sheet.labels, is(outSheet.labels));
205             
206             assertThat(sheet.c1, is(outSheet.c1));
207             assertThat(sheet.c2, is(outSheet.c2));
208             assertThat(sheet.c3, is(outSheet.c3));
209             assertThat(sheet.c4, is(outSheet.c4));
210             
211         }

524     @XlsSheet(name="数式を指定")
525     private static class FormulaSheet {
526         
527         private Map<String, Point> positions;
528         
529         private Map<String, String> labels;
530         
531         /**
532          * 数式の指定
533          */
534         @XlsCell(address="B4")
535         @XlsFormula("UPPER(F4)")
536         private String c1;
537         
538         /**
539          * メソッドで指定
540          */
541         @XlsCell(address="C7")
542         @XlsFormula(methodName="getC2Formula")
543         @XlsNumberConverter(excelPattern="0.000;\"▲ \"0.000")
544         private Double c2;
545         
546         /**
547          * 優先 = true
548          */
549         @XlsCell(address="B10")
550         @XlsFormula(value="F10-G10", primary=true)
551         private Date c3;
552         
553         /**
554          * 優先 = false
555          */
556         @XlsCell(address="D12")
557         @XlsFormula(value="F12+G12", primary=false)
558         private Integer c4;
559         
560         public FormulaSheet c1(String c1) {
561             this.c1 = c1;
562             return this;
563         }
564         
565         public FormulaSheet c2(Double c2) {
566             this.c2 = c2;
567             return this;
568         }
569         
570         public FormulaSheet c3(Date c3) {
571             this.c3 = c3;
572             return this;
573         }
574         
575         public FormulaSheet c4(Integer c4) {
576             this.c4 = c4;
577             return this;
578         }
579         
580         private String getC2Formula(final Point address) {
581             
582             String formula = String.format("SUM(%s%s:%s%s)",
583                     CellReference.convertNumToColString(address.x + 3),
584                     address.y + 1,
585                     CellReference.convertNumToColString(address.x + 6),
586                     address.y + 1);
587             
588             return formula;
589         }
