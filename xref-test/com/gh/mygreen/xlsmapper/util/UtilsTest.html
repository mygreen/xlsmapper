113     @Test
114     public void testConvertListToCollection() {
115         
116         final BeanFactory<Class<?>, Object> beanFactory = new Configuration().getBeanFactory();
117         
118         List<String> list = new ArrayList<>();
119         list.add("abc");
120         list.add("efg");
121         
122         Class toClass;
123         Collection<String> value;
124         
125         // 変換先がCollectionインタフェースの場合
126         toClass = Collection.class;
127         value = Utils.convertListToCollection(list, toClass, beanFactory);
128         assertThat(value, is(instanceOf(ArrayList.class)));
129         
130         // 変換先がListインタフェースの場合
131         toClass = List.class;
132         value = Utils.convertListToCollection(list, toClass, beanFactory);
133         assertThat(value, is(instanceOf(ArrayList.class)));
134         
135         // 変換先がArrayListで元のインスタンスと同じ場合。
136         toClass = ArrayList.class;
137         value = Utils.convertListToCollection(list, toClass, beanFactory);
138         assertThat(value, is(instanceOf(ArrayList.class)));
139         
140         // 変換先がVectorの場合
141         toClass = Vector.class;
142         value = Utils.convertListToCollection(list, toClass, beanFactory);
143         assertThat(value, is(instanceOf(Vector.class)));
144         
145         // 変換先がSetインタフェースの場合
146         toClass = Set.class;
147         value = Utils.convertListToCollection(list, toClass, beanFactory);
148         assertThat(value, is(instanceOf(LinkedHashSet.class)));
149         
150         // 変換先がQueueインタフェースの場合
151         toClass = Queue.class;
152         value = Utils.convertListToCollection(list, toClass, beanFactory);
153         assertThat(value, is(instanceOf(LinkedList.class)));
154         
155         // 変換先がDequeインタフェースの場合
156         toClass = Deque.class;
157         value = Utils.convertListToCollection(list, toClass, beanFactory);
158         assertThat(value, is(instanceOf(LinkedList.class)));
159         
160         // 変換先がLinkedListの場合
161         toClass = LinkedList.class;
162         value = Utils.convertListToCollection(list, toClass, beanFactory);
163         assertThat(value, is(instanceOf(LinkedList.class)));
164         
165         
166     }

168     @Test
169     public void testConvertCollectionToList() {
170         
171         List<String> data = Arrays.asList("abc", "efg");
172         
173         Collection<String> collection;
174         List<String> value;
175         
176         // 変換元がリストの場合
177         collection = new ArrayList<>(data);
178         value = Utils.convertCollectionToList(collection);
179         assertThat(value, is(instanceOf(ArrayList.class)));
180         
181         // 変換元がSetの場合
182         collection = new LinkedHashSet<>(data);
183         value = Utils.convertCollectionToList(collection);
184         assertThat(value, is(instanceOf(ArrayList.class)));
185         
186         // 変換元がLinkedListの場合
187         collection = new LinkedList<>(data);
188         value = Utils.convertCollectionToList(collection);
189         assertThat(value, is(instanceOf(LinkedList.class)));
190         
191         // 変換元がArrayDequeの場合
192         collection = new ArrayDeque<>(data);
193         value = Utils.convertCollectionToList(collection);
194         assertThat(value, is(instanceOf(ArrayList.class)));
195

198     @Test
199     public void testAddListWithIndex() {
200         
201         {
202             // 引数の確認
203             List<String> list = null;
204             assertThatThrownBy(() -> Utils.addListWithIndex(list, "abc", 0)).isInstanceOf(IllegalArgumentException.class);
205         
206         }
207         
208         {
209             // 引数の確認 - インデックスが-1
210             List<String> list = new ArrayList<>();
211             assertThatThrownBy(() -> Utils.addListWithIndex(list, "abc", -1)).isInstanceOf(IllegalArgumentException.class);
212         
213         }
214         
215         {
216             List<String> list = new ArrayList<>();
217             
218             // サイズが1つ足りない - サイズとインデックス番号が同じ
219             Utils.addListWithIndex(list, "abc", 0);
220             assertThat(list).hasSize(1).containsExactly("abc");
221             
222             // サイズが足りない
223             Utils.addListWithIndex(list, "efg", 2);
224             assertThat(list).hasSize(3).containsExactly("abc", null, "efg");
225             
226             // サイズが足りている
227             Utils.addListWithIndex(list, "hij", 1);
228             assertThat(list).hasSize(3).containsExactly("abc", "hij", "efg");
229             
230             
231         }

266     public void testAsList() {
267         
268         {
269             // 引数 - データがnullの場合
270             assertThat(Utils.asList(null, Integer.class))
271                 .isNotNull()
272                 .hasSize(0);
273         }
274         
275         {
276             // 引数 - 型がnullの場合
277             assertThatThrownBy(() -> Utils.asList(new int[]{123, 456}, null))
278                 .isInstanceOf(IllegalArgumentException.class);
279         }
280         
281         {
282             // 引数が配列出ない場合
283             assertThatThrownBy(() -> Utils.asList("text", String.class))
